import { Component, OnInit, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { CustomFormComponent } from 'src/app/shared/components/custom-form/custom-form.component';
import { CustomFormActionModel, CustomFormModel } from 'src/app/shared/components/custom-form/custom-form.model';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { DropdownOptionModel } from 'src/app/shared/models/dropdown-option.model';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { LookupsEnum } from 'src/app/shared/models/lookups.enum';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { DistributionDetails } from './assign.config';
import { DistributionService } from '../distribution.service';
import { HomeService } from '../../home.service';

@Component({
  selector: 'wfps-assign-distribution',
  templateUrl: './assign-distribution.component.html',
  styleUrls: ['./assign-distribution.component.sass']
})
export class AssignDistributionComponent implements OnInit {
  @ViewChild(CustomFormComponent) customFormComponent?: CustomFormComponent;

  submitAction: CustomFormActionModel = new CustomFormActionModel('home.distribution.assign', false, false, true, true);
  cancelAction: CustomFormActionModel = new CustomFormActionModel('', true);
  details = DistributionDetails;
  formDetails: CustomFormModel = new CustomFormModel(this.details);
  lookups: { [key: string]: DropdownOptionModel[] } = {};
  applicationCountEntity: { id?: number, count?: number, type?: string } = {};

  constructor(
    private utilsService: UtilsService,
    public translateService: TranslateService,
    public _service: DistributionService,
    public _homeService: HomeService,
    private dialog: MatDialog) { }

  ngOnInit(): void {
    this.getLookups();
  }

  public apply(event) {
    if (event.apply) {
      this._service.assign(event.formVal).subscribe(res => {
        if (res && res.body) {
          let text = '';
          if (res.body.applicationCount > 0) {
            text =
              `${this.translateService.instant("home.distribution.done1")} ${res.body.applicationCount} ${this.translateService.instant("home.distribution.done2")}`;
          }
          else {
            text = this.translateService.instant("home.distribution.done0");
          }
          const dialogRef = this.dialog.open(WarningDialogComponent, {
            width: '400px',
            data: { arabic: text, canClose: true }
          });
          dialogRef.afterClosed().subscribe(s => {
            this.customFormComponent.resetForm();
            this.applicationCountEntity = {};
          })
        }
      });
    }
  }

  change(event) {
    if (event.field == "kadaa" || event.field == "town") {
      if (this.customFormComponent.form.controls.town.value) {
        this.getAssignableForms('town', this.customFormComponent.form.controls.town.value);
      }
      else if (this.customFormComponent.form.controls.kadaa.value) {
        this.getAssignableForms('kadaa', this.customFormComponent.form.controls.kadaa.value);
      }
      else {
        const ind = this.details.findIndex(f => f.prop === 'applicationCount');
        this.details[ind].label = 'home.distribution.cases_nb';
      }
    }
  }

  private getAssignableForms(type, value) {
    if (this.applicationCountEntity.type != type || this.applicationCountEntity.id != value) {
      this.applicationCountEntity = { id: value, type: type };
      this._service.getEligableApplications(type, value).subscribe(x => {
        if (x && x.body) {
          this.applicationCountEntity.count = x.body.applicationCount;
          const ind = this.details.findIndex(f => f.prop === 'applicationCount');
          this.details[ind].label = `${this.translateService.instant('home.distribution.cases_nb')} (الاجمالي: ${x.body.applicationCount})`;
        }
      });
    }
  }

  getLookups() {
    let dropdownFields = this.details.filter(f => f.type === FieldType.dropdown && f.lookupKey !== LookupsEnum.Superviors);
    let keys = dropdownFields.map(m => { return m.lookupKey })

    this.utilsService.getLookupsService(keys).subscribe(s => {
      if (s.body) {
        keys.forEach(k => {
          this.lookups[k] = s.body[k];
        });
      }
      this.formDetails.lookups = this.lookups;
    });

    this._homeService.getActiveSupervisors().subscribe(s => {
      if (s.body) {
        this.lookups[LookupsEnum.Superviors] = s.body;
      }
      this.formDetails.lookups = this.lookups;
    });
  };

}
