import { formatDate } from '@angular/common';
import { HttpResponse } from '@angular/common/http';
import { Component, OnInit, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { finalize } from 'rxjs/operators';
import { CustomFormComponent } from 'src/app/shared/components/custom-form/custom-form.component';
import { CustomFormActionModel, CustomFormModel } from 'src/app/shared/components/custom-form/custom-form.model';
import { DetailsDialogComponent } from 'src/app/shared/components/details-dialog/details-dialog.component';
import { DropdownOptionModel } from 'src/app/shared/models/dropdown-option.model';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { LookupsEnum } from 'src/app/shared/models/lookups.enum';
import { SnackbarService } from 'src/app/shared/services/snackbar.service';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { HomeService } from '../../home.service';
import { DistributionService } from '../distribution.service';
import { AssignSpotcheckFormType, AssignSpotchecksFields, calculatePercentageFields, ChangeEventType, DialogResult, prepareAssignSpotcheckPayload } from './assign-spotchecks.config';

@Component({
  selector: 'wfps-assign-spotchecks',
  templateUrl: './assign-spotchecks.component.html',
  styleUrls: ['./assign-spotchecks.component.sass']
})
export class AssignSpotchecksComponent implements OnInit {

  @ViewChild(CustomFormComponent) customFormComponent?: CustomFormComponent;

  submitAction: CustomFormActionModel = new CustomFormActionModel('home.assign_spotchecks.assign_forms', false, false, true, true);
  cancelAction: CustomFormActionModel = new CustomFormActionModel('', true);
  details = AssignSpotchecksFields;
  percentageFields = calculatePercentageFields
  formDetails: CustomFormModel = new CustomFormModel(this.details);
  lookups: { [key: string]: DropdownOptionModel[] } = {};
  applicationCount: number;
  totalCount: number;
  assignableCount: number;
  version: number;
  isLookupsLoading: boolean;
  isSupervisorsLoading: boolean;
  constructor(
    private _homeService: HomeService,
    public translateService: TranslateService,
    private _service: DistributionService,
    private utilsService: UtilsService,
    private dialog: MatDialog,
    private snacks: SnackbarService
  ) { }

  ngOnInit(): void {
    this.getLookups();

  }

  getLookups() {
    let dropdownFields = this.details.filter(f => f.type === FieldType.dropdown && f.lookupKey !== LookupsEnum.Superviors);
    let keys = dropdownFields.map(m => { return m.lookupKey })
    this.isLookupsLoading = true;
    this.utilsService.getLookupsService(keys)
      .pipe(finalize(() => this.isLookupsLoading = false))
      .subscribe(s => {
        if (s.body) {
          keys.forEach(k => {
            this.lookups[k] = s.body[k];
          });
        }
        this.formDetails.lookups = this.lookups;
      });

    this.isSupervisorsLoading = true;
    this._homeService.getActiveSupervisors()
      .pipe(finalize(() => this.isSupervisorsLoading = false))
      .subscribe(s => {
        if (s.body) {
          this.lookups[LookupsEnum.Superviors] = s.body;
          this.lookups[LookupsEnum.FilteredSuperviors] = s.body;
        }
        this.formDetails.lookups = this.lookups;
      });
  }

  public apply() {
    this.getAssignableForms();
  }

  change(event) {
    const { field, value } = event;
    if (field === "kadaa") {
      this.onKadaaChange(value);
    }
  }

  private getAssignableForms() {
    const formValue: AssignSpotcheckFormType = this.customFormComponent.form.getRawValue();
    const filter = {
      toDate: formatDate(formValue.toDate, 'dd-MM-yyyy', 'en'),
      fromDate: formatDate(formValue.fromDate, 'dd-MM-yyyy', 'en')
    };

    this._service.getEligableSpotCheckApplications(formValue.kadaa, filter)
      .subscribe(this.handleResponseAndOpenDialog.bind(this));
  }

  private handleResponseAndOpenDialog(response: HttpResponse<{ assignableCount: number, totalCount: number, version: number }>) {
    if (response && response.body) {
      const formValue: AssignSpotcheckFormType = this.customFormComponent.form.getRawValue();
      this.totalCount = response.body.totalCount;
      this.assignableCount = response.body.assignableCount;
      this.version = response.body.version;
      const detailsTitle = `${this.translateService.instant("home.assign_spotchecks.total_forms_count")} ${this.totalCount} | ${this.translateService.instant("home.assign_spotchecks.eligible_forms_count")} ${this.assignableCount}`;
      if (this.assignableCount > 0) {
        this.openDetails(formValue, detailsTitle);
      } else {
        this.snacks.openWarningSnackBar(this.translateService.instant("home.assign_spotchecks.no_forms_found"));
      }
    }
  }

  public openDetails(element?: any, detailsTitle?: string) {
    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: "details-dialog",
      disableClose: true,
      width: '80%',
      
      data: {
        data: element,
        fields: this.percentageFields,
        lookups: this.lookups,
        detailsTitle
      },
    });

    dialogRef.componentInstance.fieldChange.subscribe((fieldChange) => {
      if (fieldChange.field === "applicationPercentage") {
        const { value } = fieldChange;
        // check if value is valid number
        if (!isNaN(value) && parseInt(value)) {
          const result = Math.min((this.totalCount * parseInt(value)) / 100, this.assignableCount);
          fieldChange.form.controls.calculatedApplications.setValue(Math.round(result));
        } else {
          // if value is invalid number, clear fromControl
          fieldChange.form.controls.calculatedApplications.setValue(null);
        }
      }
    });

    dialogRef.componentInstance.result.subscribe((result: DialogResult) => {
      const payload = prepareAssignSpotcheckPayload(result, this.version);
      this._service.assignSpotcheck(payload).subscribe(() => {
        this.snacks.openSnackBar("snackbar.edit_success");
        dialogRef.close();
      })
    });
  };

  private onKadaaChange(value: number) {
    // if kadaa dropdown is cleared, show list of all supervisors
    if (!value) {
      this.lookups[LookupsEnum.FilteredSuperviors] = this.lookups[LookupsEnum.Superviors];
      return;
    };
    // filter supervisor list based on kadaa
    this.lookups[LookupsEnum.FilteredSuperviors] = this.lookups[LookupsEnum.Superviors].filter((x: any) => x.kadaa.id === value);
    
    // check if the already selected supervisors are in the same kadaa selected
    // if not, clear supervisorIds control value
    const control = this.customFormComponent.form.get('supervisorIds');
    const supervisorIds = [];
    if (control.value) {
      control.value.forEach(id => {
        const isSupervisorInKadaa = this.lookups[LookupsEnum.FilteredSuperviors].find(x => x.id === id);
        isSupervisorInKadaa ? supervisorIds.push(id) : null;
      });
      supervisorIds !== control.value ? control.setValue(supervisorIds) : null;
    }
  }

}
