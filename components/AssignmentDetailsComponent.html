import { Component, Injector, OnInit, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { ActivatedRoute, Router } from '@angular/router';
import { TranslateService } from '@ngx-translate/core';
import { forkJoin } from 'rxjs';
import { finalize } from 'rxjs/operators';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { CustomFormArrayComponent } from 'src/app/shared/components/custom-form-array/custom-form-array.component';
import { CustomFormActionModel, CustomFormModel } from 'src/app/shared/components/custom-form/custom-form.model';
import { DropdownOptionModel } from 'src/app/shared/models/dropdown-option.model';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { BECampaignFormModel, BEDbRequestModel, BEDbRequestTablesModel, FeCampaignFormModel } from '../call-center-admin.model';
import { CallCenterAdminService } from '../call-center-admin.service';
import { CallCenterAssignmentDetails } from './assignment-details.config';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';


@Component({
  selector: 'wfps-assignment-details',
  templateUrl: './assignment-details.component.html',
  styleUrls: ['./assignment-details.component.sass']
})
export class AssignmentDetailsComponent extends DataTableBaseClass<any> implements OnInit {
  @ViewChild(CustomFormArrayComponent) customFormArrayComponent?: CustomFormArrayComponent;
  submitAction: CustomFormActionModel = new CustomFormActionModel('home.call_center.details.save_assignment', false, false, true, true);
  cancelAction: CustomFormActionModel = new CustomFormActionModel('', true);
  details = CallCenterAssignmentDetails;
  formDetails: CustomFormModel = new CustomFormModel(this.details);
  lookups: { [key: string]: DropdownOptionModel[] } = {};
  result;
  isDataLoaded = false;
  private id: number;
  isLookupsLoaded: boolean;

  constructor(
    public injector: Injector,
    public _service: CallCenterAdminService,
    public router: Router,
    public route: ActivatedRoute,
    public utilsService: UtilsService,
    private _snackbar: MatSnackBar,
    private translateService: TranslateService,
    private dialog: MatDialog
  ) {
    super(injector);
  }

  ngOnInit(): void {
    super.ngOnInit();
    this.getLookups();
    this.id = this.route.snapshot.params['id'];
    if (this.id) {
      this._service.getAssignmentById(this.id)
        .pipe(finalize(() => this.isDataLoaded = true))
        .subscribe(res => {
          this.result = res.body;
          this.formDetails.data = this.prepareForm(res.body);
          this.formDetails = { ...this.formDetails };
        })
    } else {
      this.isDataLoaded = true;
    }
  }

  public apply(event) {
    if (event.apply) {
      const payload = this.preparePayload(event.formVal)
      if (payload.dbRequest.isManual) {
        payload.dbRequest.dbRequestFields = [];
        payload.dbRequest.dbRequestTableFilterCriteria = [];
        payload.dbRequest.dbRequestTables = [];
      }
      this._service.saveAssignment(payload).subscribe(res => {
        if (res && res.body) {
          this._snackbar.open(this.translateService.instant('snackbar.edit_success'), null, { duration: 1500 });
          if (!this.id) {
            this.router.navigate(['../'], { relativeTo: this.route })
          }
        }
      }, _ => {
        this.dialog.open(WarningDialogComponent, {
          width: '400px',
          data: { arabic: this.translateService.instant('home.call_center.list.execute_error'), canClose: true }
        });
      });
    }
  }

  preparePayload(data: any): BECampaignFormModel {

    let dbRequestTablesArray: Array<BEDbRequestTablesModel> = [];

    const primaryTable: BEDbRequestTablesModel = {
      tableName: data.mainTable,
      isPrimary: true,
      joinFromTable: null,
      joinToTable: null,
      joinFromField: null,
      joinToField: null,
    }
    const mainTableId = this.result?.dbRequest?.dbRequestTables?.find(x => x.isPrimary == true)?.id;
    if (mainTableId) primaryTable.id = mainTableId;
    dbRequestTablesArray.push(primaryTable);

    data.dbRequestTables.forEach((t) => {
      const tableInfo = t;
      t.isPrimary = false;
      dbRequestTablesArray.push(tableInfo);
    })

    const dbRequestData: BEDbRequestModel = {
      name: null,
      isManual: data.isManual,
      query: data.query,
      dbRequestTables: dbRequestTablesArray,
      dbRequestFields: data.dbRequestFields,
      dbRequestTableFilterCriteria: data.dbRequestTableFilterCriteria,
    };

    if (this.result?.dbRequest.id) dbRequestData.id = this.result.dbRequest.id;
    const allInfo: BECampaignFormModel = {
      title: data.title,
      instructions: data.instructions,
      sampleSize: data.sampleSize,
      operators: data.operators,
      dbRequest: dbRequestData,
    }
    if (data.id) allInfo.id = data.id;

    return allInfo;
  }

  prepareForm(data: any): FeCampaignFormModel {
    const operators = data.operators.map(x => x.id);
    const formDetails: FeCampaignFormModel = {
      id: data.id,
      title: data.title,
      sampleSize: data.sampleSize,
      instructions: data.instructions,
      isManual: data.dbRequest.isManual,
      query: data.dbRequest.query,
      operators,
      mainTable: data.dbRequest.dbRequestTables.find(x => x.isPrimary == true)?.tableName,
      dbRequestFields: data.dbRequest.dbRequestFields,
      dbRequestTables: data.dbRequest.dbRequestTables.filter(x => x.isPrimary == false),
      dbRequestTableFilterCriteria: data.dbRequest.dbRequestTableFilterCriteria,
    }
    return formDetails;
  }

  getLookups() {
    this.formDetails.lookups = {};
    forkJoin([
      this._service.getOperatorsList(),
      this._service.getOperationTypes(),
      this._service.getTablesList(), this._service.getfieldsList()
    ])
      .pipe(finalize(() => this.isLookupsLoaded = true))
      .subscribe((data) => {
        this.lookups.operators = data[0].body?.map(m => {
          m.name = `${m.firstName} ${m.fatherName} ${m.lastName}`;
          return m;
        });
        this.lookups.campaignOperationTypes = data[1].body;
        this.lookups.tableNames = data[2].body.map(m => {
          return { id: m.name, name: m.name }
        });

        let fieldsList = [];
        data[3].body.forEach(m => {
          fieldsList = fieldsList.concat(m.fields);
        })

        this.lookups.fieldsName = fieldsList.map(m => {
          return { id: m.fieldName, name: m.fieldName, nameEn: '', type: m.fieldType, parentTable: m.parentTable }
        });

        this.lookups.fieldsNameTo = this.lookups.fieldsName;
        this.lookups.fieldsNameFrom = this.lookups.fieldsName;
        this.lookups.fieldsNameCriteria = this.lookups.fieldsName;
        this.lookups.fieldsNameRes = this.lookups.fieldsName;

        this.formDetails.lookups = this.lookups;
      });
  }

  clickBack() {
    this.router.navigate(['./'], { relativeTo: this.route.parent })
  }
}
