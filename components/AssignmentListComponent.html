import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { ActivatedRoute } from '@angular/router';
import { TranslateService } from '@ngx-translate/core';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { ConfirmComponent } from 'src/app/shared/components/confirm/confirm-component';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { FilterConfig } from 'src/app/shared/models/filter-config.model';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { PaginatorConfig } from 'src/app/shared/models/paginator-config.model';
import { UserModulesEnum, UserRolesActionsEnum } from 'src/app/shared/models/user-role.enum';
import { CallCenterAdminService } from '../call-center-admin.service';
import { AssignmentFilterbyOptions, CallCenterAssignmentColumnDef } from './assignment-list.config';

@Component({
  selector: 'wfps-assignment-list',
  templateUrl: './assignment-list.component.html',
  styleUrls: ['./assignment-list.component.sass']
})
export class AssignmentListComponent extends DataTableBaseClass<any> implements OnInit {

  dataTableColumnsDefinition = CallCenterAssignmentColumnDef;
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  filterConfig = new FilterConfig(ConfigTypeEnum.NONE);
  lookups: any = {};
  moduleName = UserModulesEnum.SUPPORT_ASSIGNMENTS;
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.call_center.list.empty_table'
  };
  filterbyOptions = AssignmentFilterbyOptions;
  canAdd: boolean = true;
  canExecute: boolean = false;

  constructor(
    public _service: CallCenterAdminService,
    public injector: Injector,
    public route: ActivatedRoute,
    public dialog: MatDialog,
    public translateService: TranslateService,
    private _snackbar: MatSnackBar
  ) {
    super(injector);
  }

  ngOnInit(): void {
    super.ngOnInit();
    this.getLookups();
    this.updateAccess();
    this.updateColumns();
    this.getListing(null, true);
  }

  private updateAccess() {
    this.canExecute = this.authService.isActionAllowed(this.moduleName, UserRolesActionsEnum.EXECUTE);
  }

  private updateColumns() {
    this.dataTableColumnsDefinition?.forEach(e => {
      if (e.property === 'execute') {
        if (!this.canExecute)
          e.visible = false;
      }
      else if (e.property === 'edit') {
        if (!this.canViewDetails)
          e.visible = false;
      }
    });
  }

  getLookups() {
    this._service.getOperatorsList().subscribe((data) => {
      this.lookups.operators = data.body?.map(m => {
        m.name = `${m.firstName} ${m.fatherName} ${m.lastName}`;
        return m;
      });
    });
  }

  getListing(paginationObj: PageChangeEvent, start: boolean, filter?: string) {
    this._service.getList(paginationObj, start, filter).subscribe(
      (data) => {
        if (data && data.body) {
          this.updateListData(data.body.content);
          this.setDataTableData(data.body, start);
        }
      },
      (err) => { }
    );
  }

  updateListData(data) {
    data.forEach(e => {
      this.updateActions(e);
    })
  }

  updateActions(element) {
    if (element.assignmentTasksCount > 0) {
      element.execute_icon = "assets/icons/domain_verification-green.svg";
      element.isExecuteDisabled = true;
      element.isEditDisabled = true;
    }
    else {
      element.execute_icon = "assets/icons/domain_verification-grey.svg";
    }
  }

  public openDetails(id?: any) {
    this.router.navigate([`${id}`], { relativeTo: this.route });
  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getListing(pageChangeEvent, false, this.filter);
  }

  public newClicked() {
    this.router.navigate(['new'], { relativeTo: this.route })
  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'edit':
        this.openDetails(ev.data.id);
        break;
      case 'execute':
        this.executeAssignment(ev.data);
        break;
      default:
        break;
    }
  };

  executeAssignment(data) {
    let dialogRef = this.dialog.open(ConfirmComponent, {
      panelClass: "confirm-dialog",
      data: {
        confirmationMessage: 'home.call_center.list.execute_confirmation',
        icon: 'no_accounts'
      }
    });
    dialogRef.afterClosed().subscribe((isConfirmed) => {
      if (isConfirmed) {
        this._service.executeAssignment(data.id).subscribe(s => {
          if (s && s.body) {
            let index = this.dataTableData.visibleRows.findIndex(x => x.id === data.id);
            if (index >= 0) {
              this.dataTableData.visibleRows[index].assignmentTasksCount = s.body.assignmentTasksCount;
              this.updateActions(this.dataTableData.visibleRows[index]);
            }
            this.dialog.open(WarningDialogComponent, {
              width: '400px',
              data: { arabic: this.translateService.instant('home.call_center.list.execute_success', { count: s.body?.assignmentTasksCount }), canClose: true }
            });
          }
        }, err => {
          this.dialog.open(WarningDialogComponent, {
            width: '400px',
            data: { arabic: this.translateService.instant('home.call_center.list.execute_error'), canClose: true }
          });
        });
      }
    });
  }

  searchFilter(events) {
    //The filter is a string of key:value seperated by comma
    this.filter = this.frameworkHelper.getFilterString(events, 'suspended');

    //Call List API
    this.getListing(null, true, this.filter);

    //Clear filters
    if (events == null) {
      this.header.togglePanel();
    }
  }

}
