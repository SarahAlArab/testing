import { Component, OnInit } from '@angular/core';
import { DataType } from 'src/app/shared/models/data-list.model';
import { HouseholdsService } from '../../households.service';
import { AuditModel } from './audit.config'
import { UtilsService } from 'src/app/shared/services/utils.service';
import { ActivatedRoute, Router } from '@angular/router';
import { forkJoin } from 'rxjs';
import { TABLE_NAME_UPLOAD, UPLOAD_SECTION_ENUM } from './audit.model';
import { MatSnackBar } from '@angular/material/snack-bar';

@Component({
  selector: 'wfps-audit',
  templateUrl: './audit.component.html',
  styleUrls: ['./audit.component.sass']
})
export class AuditComponent implements OnInit {
  application: any = {};
  auditModel = AuditModel;
  lookupKeyList = [];
  lookups: any = {};
  formId: any;
  isCIData;

  constructor(
    public _service: HouseholdsService,
    private UtilsService: UtilsService,
    private route: ActivatedRoute,
    private _snackbar: MatSnackBar,
  ) { }

  ngOnInit(): void {
   
    this.formId = this.route.snapshot.params.id;
    this.isCIData = this.route.snapshot?.data?.isCIData;
   
    this.getData();
    this.getLookups();
  }

  getData() {
    if (this.isCIData){
      this._service.getAudit(this.formId).subscribe( data => 
        {
          this.application = data.body;
        })

    }
     else if (!this.isCIData ){
      forkJoin([this._service.getDaemApplication(this.formId), this._service.getDaemPersons(this.formId)]).subscribe(res => {
        if (res[0]) {
          res[0].body.familyMembersNumber = res[1].body.length;
          this.application = res[0].body;
          this.application.persons = res[1].body;
          this.getFiles(this.application.persons);
        }
      });
    }
    
    
  };

  getFiles(persons) {
    if (persons && persons.length) {
      const uploadSections = Object.values(UPLOAD_SECTION_ENUM);
      let filesToGet = [];

      persons.forEach(per => {
        filesToGet = filesToGet.concat(uploadSections.map(u => {
          return { personId: per.id, section: u }
        }));
      });
      const filesObservables = filesToGet.map(u => {
        return this._service.getUploadedFiles(u.personId, TABLE_NAME_UPLOAD, u.section);
      });
      forkJoin(filesObservables).subscribe(s => {
        filesToGet = filesToGet.map((m, i) => { m.fileIndex = i; return m; });
        persons.forEach(per => {
          let uploadedFolders = filesToGet.filter(f => f.personId == per.id);
          uploadedFolders.forEach(f => {
            let uploadedFiles = s[f.fileIndex]?.body?.documents;
            if (uploadedFiles && uploadedFiles.length) {
              if (!per.files) { per.files = {}; }
              per.files[f.section] = uploadedFiles.map(m => {
                return {
                  fileName: m.IMAGE_NAME,
                  key: m.URL_KEY
                }
              });
            }

          });
        });
      },
        err => {
          this._snackbar.open('لم يتم تنزيل الملفات بنجاح', null, { duration: 1500 });
        });
    }

  }

  findLookupkeys(array, length, index) {
    if (index >= array.length) return;
    if (array[index].type === DataType.array) {

      this.findLookupkeys(array[index].children, array[index].children.length, 0);

      if ((array[index].children?.lookupKey)) {
        this.lookupKeyList.push(array[index].children.lookupKey);
      }
    }

    this.findLookupkeys(array, length, index + 1);

    if ((array[index]?.lookupKey)) {
      this.lookupKeyList.push(array[index].lookupKey);
    }
  }

  getLookups() {
    let allFields = [];
    this.auditModel.forEach(conf => {
      allFields = allFields.concat(conf.content);
    });

    this.findLookupkeys(allFields, allFields.length, 0);

    this.UtilsService.getLookupsService(this.lookupKeyList).subscribe(s => {
      if (s.body) {
        this.lookupKeyList.forEach(k => {
          this.lookups[k] = s.body[k];
        });
      }
    });
  }

  isEmptyObject(obj) {
    return (obj && (Object.keys(obj).length === 0));
  }

}


<div *ngIf="!isEmptyObject(lookups)">
    <wfps-details-report [report]="application" [config]="auditModel" [lookups]="lookups"></wfps-details-report>
</div>
