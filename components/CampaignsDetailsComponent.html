import { Component, OnInit } from '@angular/core';
import { CustomFormActionModel, CustomFormModel } from 'src/app/shared/components/custom-form/custom-form.model';
import { CampaignsDetailsService } from './campaigns-details.service';
import { FormDetails } from './campaigns-details.config';
import { forkJoin } from 'rxjs';
import { finalize } from 'rxjs/operators';
import { BECampaignFormModel, FeCampaignFormModel } from './campaigns-details.model';
import { ActivatedRoute, Router } from '@angular/router';

import { formatDate } from "@angular/common";

@Component({
  selector: 'app-campaigns-details',
  templateUrl: './campaigns-details.component.html',

})
export class CampaignsDetailsComponent implements OnInit {
  formDetails: CustomFormModel = new CustomFormModel(FormDetails);
  public essnFormData: FeCampaignFormModel;
  lookups: any = {};
  result;

  submitBtn: CustomFormActionModel = new CustomFormActionModel('home.households.saveAndContinue', false, false);
  constructor(public _service: CampaignsDetailsService,
    private route: ActivatedRoute,
    private router: Router,
    private location: Location,
    private utilsService: UtilsService) { }

  private id: number;

  ngOnInit(): void {
    this.id = this.route.snapshot.params['id'];
    this.getLookups();
    if (this.id) {
      this._service.getCampaignById(this.id).subscribe(res => {
        this.result = res.body;
        this.formDetails.data = this.prepareForm(res.body);
        this.formDetails = { ...this.formDetails };
      })
    }
  }

  submit(ev) {
    if(ev.apply){   
      const payload = this.preparePayload(ev.formVal)
    // if (payload.dbRequest.isManual) {
    //   payload.dbRequest.dbRequestFields = [];
    //   payload.dbRequest.dbRequestTableFilterCriteria = [];
    //   payload.dbRequest.dbRequestTables = [];
    // }
    this._service.addCampaign(payload).subscribe(res => {
      this.router.navigate([`campaigns`], { relativeTo: this.route.parent });
    })
  }
  else
    this.router.navigate([`campaigns`], { relativeTo: this.route.parent });
  }

  fieldChange(data) {
    if (data.field == "dbRequestId") {
      data.form.get("statusId").setValue(null);
      data.form.get('statusId').updateValueAndValidity();
      data.form.get('kadaaId').setValue(null);
      data.form.get('kadaaId').updateValueAndValidity();
      data.form.get('mophStatusId').setValue(null);
      data.form.get('mophStatusId').updateValueAndValidity();
      data.form.get("statusId").enable();  
      this.handleMophStatus(data);
  }
}

  handleMophStatus(data){
    //put status readonly and set value to pay enrolled
    let currentIdValue = data.value;
    const result = this.lookups.dbRequests.find(obj => {
      return obj.id === currentIdValue;
    });
    if(!!result){ 
      const name = result.name;
      //const {name} = result;
      if(!!name && name === "Kadaa And MophEnrollmentStatus"){
       const obj = this.lookups.ApplicationStatus.find(obj =>{
            return obj.type ==="PAY_ENROLLED";
         });

         const beirutKadaa = this.lookups.Kadaa.find(obj =>{
          return obj.type ==="BEIRUT";
       });
         
        if(!!obj){
          data.form.get("statusId").setValue(null);
          data.form.get("statusId").disable();  
          data.form.get("statusId").setValue(obj.id);          
          //disable the field 
          //this.formDetails.fields.find(key => {if (key.prop==='statusId') return key.isDisabled=true});
        } 
        if(!!beirutKadaa){
          data.form.get("kadaaId").setValue(null);
          data.form.get("kadaaId").disable();  
          data.form.get("kadaaId").setValue(beirutKadaa.id); 
        }

    }
  }
}

  getLookups() {
    this.isLookupsLoading = true;
    this.formDetails.lookups = {};
    forkJoin([
      this._service.getTemplatesList(),
      this._service.getSmsServerList(),
      this._service.getDbRequestList(),
      this.utilsService.getLookupsService(['ApplicationStatus','Kadaa','MophEnrollmentStatus'])
    ])
      .pipe(finalize(() => this.isLookupsLoading = false))
      .subscribe((data) => {
        this.lookups.campaignOperationTypes = [];
        this.lookups.fieldsName = [];
        this.lookups.tableNames = [];
        this.lookups.fieldsName = [];
        this.lookups.fieldsNameTo = [];
        this.lookups.joinType = [];
        this.lookups.fieldsNameTo = [];
        this.lookups.fieldsNameFrom = [];
        this.lookups.fieldsNameCriteria = [];
        this.lookups.fieldsNameRes = [];

        this.lookups.templates = data[0].body
        this.lookups.smsServers = data[1].body.sms_server.map(m => {
          return { id: m.NAME, name: m.NAME }
        });
        this.lookups.dbRequests =  data[2].body;
        this.lookups.ApplicationStatus = data[3].body.ApplicationStatus;
        this.lookups.Kadaa = data[3].body.Kadaa;
        this.lookups.MophEnrollmentStatus = data[3].body.MophEnrollmentStatus;
        this.formDetails.lookups = this.lookups;
        this.addStaticContentToMophLookup();
      });
  }

  addStaticContentToMophLookup()
  {
      const all = {"id":-1,"name":"للجميع","nameEn":"For All","type":"ALL","order":3};
      const NoStatus = {"id":-2,"name":"بلا حالة","nameEn":"No Status","type":"ALL","order":4};
      this.lookups.MophEnrollmentStatus.push(all,NoStatus);
      this.formDetails.lookups = this.lookups;
  }

  addChild(ev) {
    if (ev.name == 'dbRequestTables') {
      let lastAlias = ev.array.length > 1 ? ev.array?.controls[ev.array.length -2].get('alias').value : 0;
      ev.array?.controls[ev.index].get('alias').setValue(lastAlias+1);
      if (!this.formDetails.lookups.aliases) {
        this.formDetails.lookups.aliases = [];
      }
      this.formDetails.lookups.aliases.push({ id: lastAlias+1, name: `JoinTable${lastAlias+1}` });
    }
  }

  deleteChild(ev) {
    if (ev.name == 'dbRequestTables') {
      this.formDetails.lookups.aliases = this.formDetails.lookups.aliases.filter(x => x.id != ev.oldDetails.get('alias').value)
    }
  }

  goBack() {
      this.location.back();
  }
}
