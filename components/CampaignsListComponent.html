import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { PaginatorConfig } from 'src/app/shared/models/paginator-config.model';
import { FilterConfig } from 'src/app/shared/models/filter-config.model';
import { CampaignsWarningDialogComponent } from '../campaigns-warning-dialog/campaigns-warning-dialog.component';
import { CampaignsColumnDef } from './campaigns-list.config';
import { CampaignsService } from './campaigns.service';
import { ConfirmationDialogComponent } from 'src/app/shared/components/dialogs/confirmation-dialog/confirmation-dialog.component';
import { TranslateService } from '@ngx-translate/core';

@Component({
  selector: 'app-campaigns-list',
  templateUrl: './campaigns-list.component.html',
  styleUrls: ['./campaigns-list.component.sass']
})
export class CampaignsListComponent extends DataTableBaseClass<any> implements OnInit {
  dataTableColumnsDefinition = CampaignsColumnDef;
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  filterConfig = new FilterConfig(ConfigTypeEnum.NONE);
  paginationIndex: any;
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.call_center.list.empty_table'
  };

  constructor(public dialog: MatDialog,
    public injector: Injector,
    public _service: CampaignsService,
    public translateService: TranslateService,
    ) {
    super(injector);
  }

  ngOnInit(): void {
    this.paginationIndex = this._service.getPaginationIndexValue();
    const withResett =
      this.paginationIndex && this.paginationIndex.length != 0 ? false : true;

    this.getCampaignListing(this.paginationIndex, withResett)
  }

  getCampaignListing(paginationObj: PageChangeEvent, start: boolean) {
    if (paginationObj) {
      this._service.setPaginationIndexValue(paginationObj);
    }
    this._service.getCampaignsList(paginationObj, start).subscribe((data) => {
      if (data && data.body) {
        this.setDataTableData(data.body, start);
      }
    },
      (err) => { }
    );
  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getCampaignListing(pageChangeEvent, false);
  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'executeThenSend':
        this.sendText(ev.data.id);
        break;
      case 'send':
        this.sendDirectly(ev.data.id);
      case 'edit':
        this.openDetails(ev.data.id);
        break;
      case 'delete':
        this.delete(ev.data);
        break;
      default:
        break;
    }
  }

  public openDetails(id?: any) {
    this.router.navigate([`${id}`], { relativeTo: this.route });
  }

  public sendText(id: any) {
    this._service.executeQuery(id).subscribe(res => {
      const dialogRef = this.dialog.open(CampaignsWarningDialogComponent, {
        width: '400px',
        data: { arabic: '', english: res.body.infoMessage }
      });

      dialogRef.afterClosed().subscribe(res => {
        if (res) {
          this._service.executeSendQuery(id).subscribe(res => {
          this.snacks.openSnackBar("snackbar.send_success");
          }, error => {
            this.snacks.openWarningSnackBar("snackbar.delete_error");
          })
        }
      })
    },
    (err) => {
    this.snacks.openWarningSnackBar("snackbar.send_error");});
  }

  private delete(row) {
    this._service.deleteCampaign(row.id).subscribe(s => {
      this.getCampaignListing(null, true)
      this.snacks.openSnackBar("snackbar.delete_success");
    }, error => {
      this.snacks.openWarningSnackBar("snackbar.delete_error");
    });
  }

  private sendDirectly(id: any){
    const dialogRefConfirmation = this.dialog.open(ConfirmationDialogComponent, {
      width: '400px',
      data: {
          body: `${this.translateService.instant(
                    'confirmation_dialog.are_you_sure'
                  )}`,
        saveButtonText: 'confirmation_dialog.confirm',
        discardButtonText: 'confirmation_dialog.reject'
        }
    })

    dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
      if (confirmation) {
        this._service.executeSendQuery(id).subscribe(res => {
          this.snacks.openSnackBar("snackbar.send_success");
          }, error => {
            this.snacks.openWarningSnackBar("snackbar.delete_error");
          })
      }
    }) 
  }
}
