import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { FrameworkHelper } from 'src/app/shared/helpers/framework.helper';
import { DropdownOptionModel } from 'src/app/shared/models/dropdown-option.model';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { FilterConfig } from 'src/app/shared/models/filter-config.model';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { PaginatorConfig } from 'src/app/shared/models/paginator-config.model';
import { UserModulesEnum } from 'src/app/shared/models/user-role.enum';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { ApplicationStatusEnum } from '../../assigned-households/application-status.enum';
import { HouseholdsService } from '../../households.service';
import { familiesColumnDef, FilterbyOptions } from '.././cases-inspection.config';

@Component({
  selector: 'wfps-cases-inspection-list',
  templateUrl: './cases-inspection-list.component.html',
  styleUrls: ['./cases-inspection-list.component.sass']
})
export class CasesInspectionListComponent extends DataTableBaseClass<any> implements OnInit {

  inspectAudit = 'INSPECT_GET_AID_CARD_APP';
  inspectEssn = 'INSPECT_GET_ESSN_AID_CARD_APP';
  filterbyOptions = FilterbyOptions;
  filterLookups: any = {};
  element: any = {};
  frameworkHelper= FrameworkHelper;
  filter = null;
  canAdd: boolean = false;
  dataTableColumnsDefinition;
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  paginationIndex: any;
  filterConfig = new FilterConfig(ConfigTypeEnum.NONE);
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.households.empty_table',
  };
  assetsToLoad = [
    "assets/icons/person-gray.svg",
    "assets/icons/person-orange.svg",
    "assets/icons/person-red.svg",
    "assets/icons/person-green.svg",
    "assets/icons/house.svg",
    "assets/icons/house_icon_yellow.svg",
    "assets/icons/house_icon_red.svg",
    "assets/icons/house-green.svg",
  ];
  lookups: { [key: string]: DropdownOptionModel[] } = {};

  constructor(
      public _service: HouseholdsService,
      private translateService: TranslateService,
      public utilService: UtilsService,
      public dialog: MatDialog,
     public injector: Injector) {
    super(injector);
  }

  ngOnInit(): void {
    this.dataTableColumnsDefinition = JSON.parse(
      JSON.stringify(familiesColumnDef)
    );
    this.dataTableColumnsDefinition.find(x => x.property === 'audit').visible  =  this.isSectionAccessible(this.inspectAudit);
    this.dataTableColumnsDefinition.find(x => x.property === 'visit' ).visible = this.isSectionAccessible(this.inspectEssn);
   
    this.getLookups();
    this.paginationIndex = this._service.getCIPaginationIndexValue();
    const withResett =
      this.paginationIndex && this.paginationIndex.length != 0 ? false : true;
    this.setFilter();
   
  }

  isSectionAccessible(permission: string) {
    return this.authService.isPageAccessible(permission)
  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getListing(pageChangeEvent, false, this.filter);
  }

  setFilter() {
    let filterValues = this._service
      .getFilterVal()
      ?.find((f) => f.component === 'cases-inspection')?.filter;

    if (filterValues) {
      this.filter = this.frameworkHelper.getFilter2String(filterValues);
      
      this.element = this.frameworkHelper.formatElement(filterValues);
      this.getListing(this.paginationIndex, false, this.filter);
    }
  }

  getLookups() {
    this.utilService.getLookupsService(['Mouhafaza','Kadaa','Town','Locality','ApplicationStatus']).subscribe(res => {
      this.filterLookups.Mouhafaza = res.body.Mouhafaza;
      this.filterLookups.Kadaa = res.body.Kadaa;
      this.filterLookups.Town = res.body.Town;
      this.filterLookups.Locality = res.body.Locality;
      this.filterLookups.aidCardApplicationStatus = res.body.ApplicationStatus;
      
    })

  }

  

  searchFilter(events) {
    let numberOfFilter =0;
    numberOfFilter = Object.keys(events).filter( x => (events[x].value !=null)).length;
    if ( numberOfFilter > 1){
       //The filter is a string of key:value seperated by comma
    this.filter = this.frameworkHelper.getFilter2String(events);

    //save filter in global storage
    this._service.addFilter({ component: 'cases-inspection', filter: events });

   //Call List API
   this.getListing(this.paginatorConfig, true, this.filter);
    } else  {
      this.dialog.open(WarningDialogComponent, {
        width: '400px',
        data: { arabic: this.translateService.instant("home.households.choose_at_least_one_search_option")}
      });

    }
  
  
    //Clear filters
    if (events == null) {
      this.header.togglePanel();
    }
  }

  getListing( paginationObj: PageChangeEvent, start: boolean, filter?: string) {
    if (paginationObj) {
      this._service.setCIPaginationIndexValue(paginationObj);
    }

    this._service.getHouseholds(paginationObj.page, paginationObj.size, filter).subscribe(
      (data) => {
        if (data && data.body) {
          this.prepareListForDataTable(data.body.content);
          this.updateListData(data.body.content);
          this.setDataTableData(data.body, start);

        }
      },
      (err) => { }
    );
  }

  updateListData(data) {
    data.forEach((e) => {
      const kadaa = this.filterLookups?.Kadaa.find((f: { id: any; }) => f.id === e.kadaa.id);
      const town = this.filterLookups?.Town.find((f: { id: any; }) => f.id === e.town.id);

      e.fullName = `${e.firstName} ${e.lastName}`;
       e.address = `${kadaa?.name} - ${town?.name}`;
    });
  }

  prepareListForDataTable(list) {
    let newList = list;

    newList.forEach((l) => {
      const essnStatus = this.filterLookups.aidCardApplicationStatus.find((f: { id: any; }) => f.id === l.essnApplicationStatus.id);
      const auditStatus = this.filterLookups.aidCardApplicationStatus.find((f: { id: any; }) => f.id === l.aidCardApplicationStatus.id).name;
    
      if (essnStatus){
        l.visit_status = essnStatus.name;
      } else {
        l.visit_status = "لا توجد بيانات"
      }
      
      if (auditStatus){
        l.audit_status = auditStatus;
      } else {
        l.audit_status = "";
      }
    });
    return newList;

  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'audit':
        this.router.navigate([`${ev.data.aidCardApplication.id}/audit`], {
          state: { isCIData: true },
          relativeTo: this.route,
        });
        break;

      case 'visit':
        if (ev.data.essnApplication.id){
          this.router.navigate([`${ev.data.essnApplication.id}/visit`], {
            state: { isCIData: true },
            relativeTo: this.route,
           
          });
        } else {
          this.dialog.open(WarningDialogComponent, {
            width: '400px',
            data: { arabic: this.translateService.instant('home.households.access_denied'), canClose: true }
          });
        }
       
        break;
        
      default:
        break;
    }
  }




 

 
}
