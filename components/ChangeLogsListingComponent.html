import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { DataTableColumnDefinition } from 'src/app/shared/models/data-table-column-definition.model';
import { LookupsEnum ,LookupsAppStatusTypeEnum} from 'src/app/shared/models/lookups.enum';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';

@Component({
  selector: 'wfps-change-logs-listing',
  templateUrl: './change-logs-listing.component.html',
  styleUrls: ['./change-logs-listing.component.sass'],
})
export class ChangeLogsListingComponent
  extends DataTableBaseClass<any>
  implements OnInit

  constructor(
    private utilSerivce: UtilsService,
    public injector: Injector,
    public translateService: TranslateService,
    public _service: HouseholdsService,
    private dialog: MatDialog,
    private _homeService: HomeService,
    private utilsService: UtilsService

  ) {
    super(injector);
  }

  ngOnInit(): void {
    this.dataTableColumnsDefinition = JSON.parse(
      JSON.stringify(AuditedHistoryColumnDef)
    );

    this.getFilterLookups();
    this.setFilter();

    this.paginationIndex = this._service.getPaginationIndexValue();
    const withResett = this.paginationIndex && this.paginationIndex.length != 0 ? false : true;

    this.getListing(this.paginationIndex, true, this.filter, true);

  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getListing(pageChangeEvent, false, this.filter);
  }

  getListing(
    paginationObj: PageChangeEvent,
    start: boolean,
    filter?: string,
    openPanel?: boolean
  ) {
    if (paginationObj) {
      this._service.setPaginationIndexValue(paginationObj);
    }

    let filterArr = this.filter?.split(',');
    let adminChanges = this.filter?.includes('adminChanges');
    if(adminChanges && filterArr.length==1){
      this.snacks.openWarningSnackBar(
        `${this.translateService.instant(
          'home.households.please_fill_additional_flter_options'
        )}`
      );
      return;
    }
   else{
    this._service.getApplicationAuditHistory(paginationObj, start, filter).subscribe(
      (data) => {
        if (data && data.body) {
          this.updateListData(data.body.content);
          this.setDataTableData(data.body, start);

          //check for previous filter, and if we're calling getlisting onInit
          if (this.filter && openPanel) {
            this.header.togglePanel();
          }
        }
      },
      (err) => {}
    );
   }

  }

  updateListData(data) {
    data.forEach((e) => {
      this.updateActionsByPrevStatus(e);
      this.updateActionsByStatus(e);
      this.updateOnHoldStatus(e);
    });
  }
  updateOnHoldStatus(element) {
    const onHold =
      element.issues.filter((x) => x.status.type === 'ON_HOLD').length > 0;
    element.onHold = onHold ? 'نعم' : 'كلا';
  }

  updateActionsByPrevStatus(element){
    element.previousApplicationStatus =  this.translateService.instant('home.households.statuses.'+element.previousApplicationStatus.type);
    element.applicationStatus =  this.translateService.instant('home.households.statuses.'+element.applicationStatus.type);
  }

  enableDisable(value: string) {
    switch (value) {
      case 'houseHoldNameChange':
        this.houseHoldNameChange = !this.houseHoldNameChange;
        break;
      case 'nameChange':
        this.nameChange = !this.nameChange;
        break;
      case 'phoneChange':
        this.phoneChange = !this.phoneChange;
        break;
      case 'memberCountChange':
        this.memberCountChange = !this.memberCountChange;
        break;
      case 'appIdChange':
        this.appIdChange = !this.appIdChange;
        break;
      case 'addressChange':
        this.addressChange = !this.addressChange;
        break;

      default:
        break;
    }
  }

  searchFilter(events) {
    //The filter is a string of key:value seperated by comma
    this.filter = this.frameworkHelper.getFilter2String(events, 'issues');
    // //save filter in global storage
    this._service.addFilter({ component: 'fuzzyMatching', filter: events });
    //Call List API
    this.getListing(this.paginatorConfig, true, this.filter);
    //Clear filters
    if (events == null) {
      this.header.togglePanel();
    }
  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'audit':
        this.router.navigate([`${ev.data.id}/audit`], {
          relativeTo: this.route,
        });
        break;
      case 'essn':
        this.router.navigate([`${ev.data.id}/essn`], {
          relativeTo: this.route,
        });
        break;
      case 'changeLogs':
        this.openDetails(ev.data, 'home.households.surveyor');
        break;
      default:
        break;
    }
  }



  public openDetails(element?: any, title?: string) {
    const dialogRef = this.dialog.open(PreviousChangesComponent, {
      panelClass: 'details-dialog',
      disableClose: false,
      width: '80%',
      data: {

    });
  }

  private updateFilterStatus() {
    if (
      !this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS] ||
      !this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS]?.length
    ) {
      this.filterbyOptions = this.filterbyOptions.filter(
        (f) => f.prop != 'applicationStatus'
      );
    }
  }

  private getilterLookups() {
    this.utilsService.getLookupsService([LookupsEnum.AID_CARD_APPLICATION_STATUS,LookupsEnum.Surveyor,LookupsEnum.Superviors]).subscribe(s => {
      if (s.body) {
        this.filterLookups[LookupsEnum.AID_CARD_APPLICATION_STATUS] = s.body[LookupsEnum.AID_CARD_APPLICATION_STATUS];
        this.filterLookups[LookupsEnum.Surveyor] = s.body[LookupsEnum.Surveyor];
        this.filterLookups[LookupsEnum.Superviors] = s.body[LookupsEnum.Superviors];
      }
    });
  }
  private getFilterLookups() {
    let dropdownFields = this.filterbyOptions.filter(
      (f) => f.type == FieldType.dropdown
    );
    let keys = dropdownFields.map((m) => {
      return m.lookupKey;
    });
    this.utilsService.getLookupsService(keys).subscribe((s) => {
      if (s.body) {
        keys.forEach((k) => {
          switch (k) {
            case LookupsEnum.Surveyor:
              this._homeService.getAllSurveyors().subscribe((Surveyors) => {
                this.filterLookups[LookupsEnum.Surveyor] = Surveyors.body;
              });
              break;
            case LookupsEnum.Superviors:
              this._homeService.getAllSupervisors().subscribe((s) => {
                this.filterLookups[LookupsEnum.Superviors] = s.body;
              });
              break;
            case LookupsEnum.FILTERED_APPLICATION_STATUS:
              this._homeService
                .getFilteredApplicationStatus()
                .pipe(
                  map((response) => {
                    const lookuptsToReturn = response.body.map((x) => {
                      x.name = '';
                      return x;
                    });
                    return { body: lookuptsToReturn };
                  })
                )
                .subscribe((s) => {
                  this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS] = s.body;
                  this.updateFilterStatus();
                });
              break;
            default:
              this.filterLookups[k] = s.body[k];
              break;
          }
        });
      }
    });
  }

  setFilter() {
    let filterValues = this._service
      .getFilterVal()
      ?.find((f) => f.component === 'assigned')?.filter;

    if (filterValues) {
      this.filter = this.frameworkHelper.getFilterString(
        filterValues,
        'issues',
        'hasMobile'
      );
      this.element = this.frameworkHelper.formatElement(filterValues);
    }
  }
