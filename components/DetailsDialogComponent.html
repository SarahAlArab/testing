import { Component, EventEmitter, Inject, Injector, OnInit, Output, ViewChild } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { FrameworkHelper } from '../../helpers/framework.helper';
import { DropdownOptionModel } from '../../models/dropdown-option.model';
import { FieldType } from '../../models/fields-config.model';
import { CustomFormComponent } from '../custom-form/custom-form.component';
import { CustomFormActionModel } from '../custom-form/custom-form.model';
import { DetailsDialogModel } from './details-dialog.model';
import { DataTableBaseClass } from '../../bases/data-table.base';

@Component({
  selector: 'app-details-dialog',
  templateUrl: './details-dialog.component.html',
  styleUrls: ['./details-dialog.component.sass'],
})
export class DetailsDialogComponent extends DataTableBaseClass<any>  implements OnInit {
  @ViewChild(CustomFormComponent) customFormComponent?: CustomFormComponent;
  form: FormGroup;
  close = false;
  error: string;
  dataTableData;
  //this.setDataTableData(this.dataTableData, true);
  fieldTypeEnum = FieldType;
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.call_center.list.empty_table'
  };
  frameworkHelper = FrameworkHelper;
  @Output() result = new EventEmitter<any>();
  @Output() fieldChange = new EventEmitter<any>();
  submitAction: CustomFormActionModel = new CustomFormActionModel(this.data.submitLbl ? this.data.submitLbl : 'generic.send', false, false, true, true);
  cancelAction: CustomFormActionModel = new CustomFormActionModel(this.data.cancelLbl ? this.data.cancelLbl : 'generic.table.delete');

  constructor(
    private fb: FormBuilder,
    public dialogRef: MatDialogRef<DetailsDialogComponent>,
    public injector: Injector,
    @Inject(MAT_DIALOG_DATA) public data: DetailsDialogModel
  ) {
     super(injector);
  }

  ngOnInit(): void {
    this.dataTableData = this.data.dataTableData;
    this.setDataTableData(this.dataTableData, true);
  }

  public apply(event) {
    if (event.apply) {
      this.buttonClicked(event.formVal);
    }
    else {
      this.closeDialog(true);
    }
  }

  public buttonClicked(formValToSave) {
    let formVal;
    if (formValToSave) {
      formVal = Object.assign(this.data.data || {}, formValToSave);
      this.result.emit(formVal);
    } 
    // else {
    //   this.dialogRef.close(formVal);
    // }
    this.dialogRef.close(formVal);
  }

  public closeDialog(val: boolean) {
    if (this.customFormComponent.form.pristine) {
      this.buttonClicked(false);
    } else this.close = val;
  }

  changeField(event) {
    this.fieldChange.emit(event);
  }

  goBack() {
    this.dialogRef.close()
  }
}
