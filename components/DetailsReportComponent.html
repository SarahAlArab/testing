import { ChangeDetectionStrategy, Component, Input, OnInit, Optional } from '@angular/core';
import { LanguageHelper } from '../../helpers/lang.helper';
import { DataListSectionConfigModel, DataType } from '../../models/data-list.model';
import { HouseholdsService } from '../../../home/households/households.service';
import { FrameworkHelper } from '../../helpers/framework.helper';
import { ActivatedRoute, Router } from '@angular/router';
import { MatDialog, MatDialogRef } from '@angular/material/dialog';
import { MapMarkerComponent } from '../map-marker/map-marker.component';

@Component({
  selector: 'wfps-details-report',
  templateUrl: './details-report.component.html',
  styleUrls: ['./details-report.component.sass'],
  changeDetection: ChangeDetectionStrategy.OnPush // works like a charm, Make sure functions from html not updating
})
export class DetailsReportComponent implements OnInit {
  @Input() report: any;
  @Input() config: DataListSectionConfigModel[];
  @Input() lookups: any;
  @Input() withHeader: any = true;
  @Input() withBackButton: any = false;
  dataType = DataType;
  languageHelper = LanguageHelper;
  frameworkHelper = FrameworkHelper;
  constructor(
    private _service: HouseholdsService,
    private dialog: MatDialog,
    @Optional() private dialogRef: MatDialogRef<DetailsReportComponent>,

  ) {
  }

  ngOnInit(): void {
  }

  public getDisplay(data, config): any {
    return this.languageHelper.getPropValue(data, config.key);
  }

  public checkConditionality(data, config) {
    //if there is condition
    if (config.showOnKey) {

      let conditionValue = this.languageHelper.getPropValue(data, config.showOnKey);

      if (config.relatedFeildLookup) {
        conditionValue = this.lookups[config.relatedFeildLookup]?.find(x => x.id === this.languageHelper.getPropValue(data, config.showOnKey)?.id)?.type;
      }

      //check if condition is true
      if (conditionValue == config.relatedFeildValue) {
        return true;
      } else {
        return false
      }

    } else {
      return true;
    }

  }

  public getLookup(data, configItem) {
    return this.lookups[configItem.lookupKey]?.find(x => x.id === this.getDisplay(data, configItem)?.id)?.name;
  }

  public getLookupArray(data, configItem) {
    let list = [];
    this.getDisplay(data, configItem).forEach(element => {
      list.push(this.lookups[configItem.lookupKey]?.find(x => x.id === element?.id)?.name);
    });
    return list.join('، ');
  }

  public getValue(data, config) {
    const value = this.languageHelper.getPropValue(data, config.key);
    return value ? 'audit_module.yes' : 'audit_module.no';
  }

  navigateTo(data, config) {
    // const navigateTo = this.languageHelper.getPropValue(data, config.key);
    const dialogRef = this.dialog.open(MapMarkerComponent, {
      width: '600px',
      data: {
        coordinates: data.relatedProps,
        distance: data.desc
      }
    })

    dialogRef.afterClosed().subscribe(res => {
      if(res !== undefined){
        this.dialogRef.close({action: res, type: data.key, id: data.id})
      }
    });
  }

  public downloadImage(data,configItem){
    let key = this.languageHelper.getPropValue(data, configItem.imageUrl);
    let fileName = this.languageHelper.getPropValue(data, configItem.key);
    this._service.downloadUploadedFiles(key).subscribe(res=>{
      //this.frameworkHelper.openPdf(res, fileName)
      this.exportImage(res, fileName)
    });
  }

  public exportImage(res: any, imageName: string){
    let file = new Blob([res.body], { type: res.body.type });
    var fileURL = URL.createObjectURL(file);
    //Uglier hack to download the file as CSV. Stupid JS.
    let a = document.createElement('a');
    a.href = fileURL;
    a.download = imageName;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
  }
}
