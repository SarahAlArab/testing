import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { DataType } from 'src/app/shared/models/data-list.model';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { HouseholdsService } from './../../households.service';
import { EssnConfig } from './essn.config';

@Component({
  selector: 'wfps-essn',
  templateUrl: './essn.component.html',
  styleUrls: ['./essn.component.sass'],
})
export class EssnComponent implements OnInit {
  //lookup keys
  lookupKeyList = [];

  //lookups
  lookups: any = {};

  //Config data to be displayed
  essnConfig = EssnConfig;

  //Data from backend
  data;

  formId: any;
  isCIData: any;

  constructor(
    private _service: HouseholdsService,
    private UtilsService: UtilsService,
    private route: ActivatedRoute,
    private router: Router
  ) {}

  ngOnInit(): void {
    this.formId = this.route.snapshot.params.id;
    this.isCIData = this.route.snapshot?.data?.isCIData;
    this.getLookups();
  }

  getData() {
    if ( this.isCIData ) {
      this._service.getEssn(this.formId).subscribe((info) => {
        this.data = { ...info.body };
        this.handleFamilyRelationship();
      });
    } else {
      this._service.getEssnForm(this.formId).subscribe((info) => {
        this.data = { ...info.body };
        this.handleFamilyRelationship();
      });
    }
  }

  handleFamilyRelationship() {
    /******************************************************************************
     * Handling null family relationship becuase backend sucks
     */
    //default is id = 10
    let headOfHouseholdId = 10;

    //get id of HEAD_OF_HOUSEHOLD beacuse the above id may change one day
    this.lookups.FamilyRelation.forEach((x) => {
      if (x.type == 'HEAD_OF_HOUSEHOLD') headOfHouseholdId = x.id;
    });

    //Assign family relationship if null for each family member
    this.data.essnHousehold.essnPerson.forEach((element) => {
      if (element.person.familyRelation == null) {
        element.person.familyRelation = { id: headOfHouseholdId };
      }
    });

    //Assign family relationship if null for respondant
    if (this.data.essnHousehold.familyRelation == null) {
      this.data.essnHousehold.familyRelation = { id: headOfHouseholdId };
    }
  }

  /******************************************************************************************************
  //******************************************************************************************************
  //** Lookups    */

  //Recursive function that finds all lookupkeys in the nested array
  findLookupkeys(array, length, index) {
    if (index >= array.length) return;
    if (array[index].type == DataType.array) {
      this.findLookupkeys(
        array[index].children,
        array[index].children.length,
        0
      );

      if (array[index].children?.lookupKey) {
        this.lookupKeyList.push(array[index].children.lookupKey);
      }
    }

    this.findLookupkeys(array, length, index + 1);

    if (array[index]?.lookupKey) {
      this.lookupKeyList.push(array[index].lookupKey);
    }
  }

  getLookups() {
    let allFields = [];
    this.essnConfig.forEach((conf) => {
      allFields = allFields.concat(conf.content);
    });
    this.findLookupkeys(allFields, allFields.length, 0);

    this.UtilsService.getLookupsService(this.lookupKeyList).subscribe((s) => {
      if (s.body) {
        this.lookupKeyList.forEach((k) => {
          this.lookups[k] = s.body[k];
        });
      }
    this.getData();
    });
  }
  /*********************************************************************************************************/
  /** */

  isEmptyObject(obj) {
    return obj && Object.keys(obj).length === 0;
  }
}

<div *ngIf="!isEmptyObject(lookups)">
    <wfps-details-report [report]="data" [config]="essnConfig" [lookups]="lookups"></wfps-details-report>
</div>
