import { HttpClient } from '@angular/common/http';
import { Component, Inject, Injector, OnInit, ViewChild } from '@angular/core';
import { FormControl } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { HttpBaseClass } from '../../../shared/bases/http-base.class';
import { DetailsDialogModel } from '../../../shared/components/details-dialog/details-dialog.model';
import { UploadFieldComponent } from '../../../shared/components/upload-field/upload-field.component';
import { DocumentModel } from '../../../shared/models/document.model';
import { HomeService } from '../../home.service';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { SnackbarService } from 'src/app/shared/services/snackbar.service';
import { TranslateService } from '@ngx-translate/core';


@Component({
  selector: 'app-external-deduplication-dialog',
  templateUrl: './external-deduplication-dialog.component.html',
  styleUrls: ['./external-deduplication-dialog.component.sass'],
})
export class ExternalDeduplicationDialogComponent implements OnInit {

  @ViewChild(UploadFieldComponent) uploadElements: UploadFieldComponent;
  programType = new FormControl('');

  constructor(
    private homeService: HomeService,
    private dialogRef: MatDialogRef<ExternalDeduplicationDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private snacks: SnackbarService,
    private translateService: TranslateService
  ) {

  }

  ngOnInit(): void {

  }

  public submit() {

    if (!this.uploadElements?.file.length || !this.programType.value) {

      this.snacks.openWarningSnackBar(
        this.translateService.instant(
          'generic.checkMissing'
        )
      );

      return;
    }

    const ext = this.uploadElements.file[0]?.type;

    if (!UtilsService.excelCSVMimeTypes.includes(ext)) {

      this.snacks.openWarningSnackBar(
        this.translateService.instant(
          'generic.checkIfExcel'
        )
      );

      return;
    }

    this.dialogRef.close({ file: this.uploadElements.file[0], id: this.programType.value });
  }

  public closeDialog() {

    this.dialogRef.close();
  }

  downloadTemplate() {

    this.homeService.downloadtemplate().subscribe(data => {

      UtilsService.downloadFile(data);
    });
  }

}
