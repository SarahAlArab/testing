import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { SnackbarService } from 'src/app/shared/services/snackbar.service';
import { DetailsDialogComponent } from 'src/app/shared/components/details-dialog/details-dialog.component';
import { additionalInfoColumnDef, campaignStatusColumnDef, extProgramCampaing, extProgramField } from './program-generation.config';
import { ExternalProgramService } from '../external-program.service';
import { FormControl } from '@angular/forms';
import { DataTableColumnDefinition } from 'src/app/shared/models/data-table-column-definition.model';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { forkJoin } from 'rxjs';
import { EXTERNAL_STATUS, PROCESS_STATUS } from '../external-program.model';

@Component({
  selector: 'wfps-program-generation',
  templateUrl: './program-generation.component.html',
  styleUrls: ['./program-generation.component.sass']
})
export class ExternalProgramGenerateComponent implements OnInit {


  EXT_PROGRAMS = 'EXT_PROGRAMS';
  isAccessExtProgramAllowed: boolean;
  data: any;
  lookups: any = {};
  extProgramField = extProgramField
  externalProgram = new FormControl('');
  detailsFormReady: boolean = false;
  additionalInfoColumnDef = additionalInfoColumnDef;
  campaignStatusColumnDef = campaignStatusColumnDef;
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.households.empty_families_address_change',
  };
  extProgramCampaing = extProgramCampaing;
  HIDE_DETAILS_FOR_STATUS = [PROCESS_STATUS.IN_PROGRESS, PROCESS_STATUS.NEW, PROCESS_STATUS.BUILDING_ERROR];
  campaignStatusList: any;

  constructor(public dialog: MatDialog,
    public translateService: TranslateService,
    public _service: ExternalProgramService,
    public snacks: SnackbarService,
    public injector: Injector,
  ) {
  }

  ngOnInit(): void {
    this.getLookups();
  }

  getLookups() {
    this.lookups = {};
    forkJoin(
      this._service.getUserProgram(),
      this._service.getAllTemplatesList(),
    ).subscribe(([res1, res2]) => {
      this.lookups.programs = res1.body;
      this.lookups.programs.map(e => e.name = e.title)

      this.lookups.campaignTemplate = res2?.body?.content
      this.lookups.ext_status = Object.keys(EXTERNAL_STATUS).map(key => ({ id: EXTERNAL_STATUS[key], name: EXTERNAL_STATUS[key] }))
    })
  }

  selectionChange(ev) {
    if (ev && ev.value) {
      this.getExtProgramDetails();
    }
  }

  getExtProgramDetails() {
    this.detailsFormReady = false;
    this._service.getExtProgramDetails(this.externalProgram.value).subscribe(
      (res) => {
        if (res && res.body) {
          //TODO add a field in data to display or hide datatable
          this.data = {
            'extProgramSectionTitle': `home.external_program.status.status_${res.body.status}`,
            'extProgramStatus': res.body.status,
            'beneficiariesHHCount': res.body.beneficiariesHHCount,
            'beneficiariesCount': res.body.beneficiariesCount,
            'additionalInfo': res.body.additionalInfo,
            'campaignStatusList': res.body.campaignStatusList,
            'showDetails': !this.HIDE_DETAILS_FOR_STATUS.includes(res.body.status)
          }
          this.additionalInfoColumnDef = [];
          if (res.body.additionalInfo && res.body.additionalInfo.length > 0) {
            const keys = Object.keys(res.body?.additionalInfo[0]);
            keys.forEach(k => {
              this.additionalInfoColumnDef.push(
                new DataTableColumnDefinition(k, `home.external_program.additional_info.${k}`, `${(100 / keys.length)}%`)
              )
            });
          }

        }
        this.detailsFormReady = true;

      }
    )
  }

  buildExternalProgramDataset() {
    this.detailsFormReady = false;
    this._service.buildExternalProgramDataset(this.externalProgram.value).subscribe((res) => {
      this.getExtProgramDetails();
    },
      (error) => {
        this.snacks.openWarningSnackBar(error.message);
        if (this.data){
          //in case error keep the preexisting data and display them
          this.detailsFormReady = true;
        }
      })
  }

  issueSmsCampaign() {
    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: 'details-dialog',
      disableClose: true,
      width: '80%',
      data: {
        fields: this.extProgramCampaing,
        lookups: this.lookups,
        modalTitle: 'home.external_program.issue_sms_campaign',
      },
    });

    dialogRef.componentInstance.result.subscribe((result) => {
      this._service.issueSmsCampaign(this.externalProgram.value, result).subscribe(res => {
        this.snacks.openSnackBar("home.external_program.execute_campaign");
        dialogRef.close();
      })
    });
  }
}
