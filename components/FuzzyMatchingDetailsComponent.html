import { Component, OnInit,ElementRef } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { ActivatedRoute, Router } from '@angular/router';
import { TranslateService } from '@ngx-translate/core';
import { elementAt, finalize } from 'rxjs/operators';
import { DetailsDialogComponent } from 'src/app/shared/components/details-dialog/details-dialog.component';
import { ConfirmationDialogComponent } from 'src/app/shared/components/dialogs/confirmation-dialog/confirmation-dialog.component';
import { LanguageHelper } from 'src/app/shared/helpers/lang.helper';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { SnackbarService } from 'src/app/shared/services/snackbar.service';
import { FuzzyModel, MODERATION_KEY } from '../fuzzy-matching.config';
import { FuzzyMatchingService } from '../fuzzy-matching.service';
import { DuplicatePerson, DuplicateApplication, ModerateDuplicatePayload, DuplicatePersonList } from './fuzzy-matching-details.config';
import { CommentField } from 'src/app/home/households/fuzzy-matching/fuzzy-matching.config'
import { forkJoin } from 'rxjs';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { ConfirmComponent } from 'src/app/shared/components/confirm/confirm-component';
import { AuthService } from 'src/app/shared/services/auth.service';
import { ApplicationScheduledStatusEnum } from '../../assigned-households/application-status.enum';

@Component({
  selector: 'wfps-fuzzy-matching-details',
  templateUrl: './fuzzy-matching-details.component.html',
  styleUrls: ['./fuzzy-matching-details.component.sass']
})
export class FuzzyMatchingDetailsComponent implements OnInit {
  duplicateApplicationId: number;
  config = FuzzyModel;
  withHeader: any = true;
  lookups: any;
  dataType = FieldType;
  report: { person1: DuplicatePersonList, person2: DuplicatePersonList } = { person1: null, person2: null };
  data: DuplicateApplication;
  isSubmitting: boolean;
  languageHelper = LanguageHelper;
  commentField = CommentField;
  isLookupsLoaded = false;
  persondelete = 'DELETE_PERSON_ANY_STATUS';
  personsIdsArray = [];
  person1Details: any= {};
  person2Details: any= {};
  duplicateTypeLabel: string;
  isDeletePersonAllowed: boolean;
  internalDuplicateModeration: boolean;
  classStatus: boolean = false;

  constructor(
    public _service: FuzzyMatchingService,
    private route: ActivatedRoute,
    private router: Router,
    public dialog: MatDialog,
    private translateService: TranslateService,
    private snackbar: SnackbarService,
    private utilSerivce: UtilsService,
    private authService: AuthService,
    private el: ElementRef
  ) { }

  ngOnInit(): void {
    this.duplicateApplicationId = this.route.snapshot.params.id;
    this.getLookups();
    this.isDeletePersonAllowed = this.authService.isPageAccessible(
      this.persondelete
    );
  }

  private getData(): void {
    this._service.getDataById(this.duplicateApplicationId).subscribe(res => {
      this.data= res.body;
      if(this.data?.applicationId == this.data?.duplicatedApplicationId)
      {
        this.internalDuplicateModeration = true;
      }
      else{
        this.internalDuplicateModeration = false;
      }

      this.report.person1 = res.body.persons1;
      this.report.person2 = res.body.persons2;

      //---------------------------person1 Details : -----------------------------------------------
      const mappedPerson1 = Object.entries(this.report.person1).map(([type, value]) => ({type, value}));
      mappedPerson1.map(item => {
        this.personsIdsArray.push(item.value?.personId);
        item.value.family = 'person1'
      });

      const mappedPerson2 = Object.entries(this.report.person2).map(([type, value]) => ({type, value}));
      mappedPerson2.map(item => {
        this.personsIdsArray.push(item.value?.personId ? item.value?.personId : item.value?.personExtNbr);
        item.value.family = 'person2'
      });

      this.person1Details.applicationCaseNumber = this.data.applicationCaseNumber;
      if (this.data.lastActiveApplicationStatus){
        this.person1Details.lastActiveApplicationStatus = this.lookups?.ApplicationStatus?.find(x => x.id == this.data?.lastActiveApplicationStatus?.id).name;
      } else {
        this.person1Details.lastActiveApplicationStatus = '';
      }
      this.person1Details.isSuspended = this.data.suspendApplicationId === this.data.applicationId ? 'home.generic_text.yes': 'home.generic_text.no';

      if (this.data.currentApplicationStatus){
        this.person1Details.currentApplicationStatus = this.lookups?.ApplicationStatus?.find(x => x.id == this.data?.currentApplicationStatus?.id).name;
      } else {
        this.person1Details.currentApplicationStatus = '';
      }

      if(this.person1Details.currentApplicationStatus.toLowerCase() === ApplicationScheduledStatusEnum.Suspended.toLowerCase()){
        this.person1Details.showLastStatus = true;
      }
      else{
        this.person1Details.showLastStatus = false;
      }
      this.person1Details.duplicateTypeLabel = this.internalDuplicateModeration ? 'essn_module.form.form3.internal_self_duplicate' : 'essn_module.form.form3.internal_duplicate';

      this.person1Details.applicationDuplicateCount = this.data.applicationDuplicateCount;

      this.person1Details.applicationPayment = this.data.applicationPayment;

      //---------------------------person2 Details : -----------------------------------------------

      this.person2Details.dupApplicationPayment = this.data.dupApplicationPayment;
      this.person2Details.dupApplicationDuplicateCount = this.data.dupApplicationDuplicateCount;

      this.person2Details.isBeneficiary = this.report.person2[0]?.extStatus === "Beneficiary" ? 'home.generic_text.yes': 'home.generic_text.no';

      this.person2Details.applicationCaseNumber = this.data.duplicatedApplicationCaseNumber;
      if (this.data.dupLastActiveApplicationStatus){
        this.person2Details.dupLastActiveApplicationStatus = this.lookups.ApplicationStatus.find(x => x.id == this.data.dupLastActiveApplicationStatus.id)?.name;
      } else {
        this.person2Details.dupLastActiveApplicationStatus = '';
      }

      this.person2Details.duplicateTypeLabel = this.data.duplicatedExtApplicationId ? 'essn_module.form.form3.external_duplicate' : 'essn_module.form.form3.internal_duplicate';
      this.person2Details.duplicatedExtApplicationId = this.data?.duplicatedExtApplicationId;

      if( this.data.suspendApplicationId){
        this.person2Details.isSuspended = this.data.suspendApplicationId === this.data.duplicatedApplicationId ? 'home.generic_text.yes': 'home.generic_text.no';
      } else {
        this.person2Details.isSuspended = 'home.generic_text.no';
      }

      if (this.data.dupCurrentApplicationStatus){
        this.person2Details.dupCurrentApplicationStatus = this.lookups?.ApplicationStatus?.find(x => x.id == this.data?.dupCurrentApplicationStatus?.id).name;
      } else {
        this.person2Details.dupCurrentApplicationStatus = '';
      }

      if(this.person2Details.dupCurrentApplicationStatus.toLowerCase() === ApplicationScheduledStatusEnum.Suspended.toLowerCase()){
        this.person2Details.showLastStatus = true;
      }
      else{
        this.person2Details.showLastStatus = false;
      }

      if(this.data.duplicatedExtApplicationId){
        this.person2Details.showRevistedAndHHStatus = true;
        this.person2Details.revisited = this.data.revisited?.toLowerCase() === "revisited" ? "home.households.fuzzy_matching-details.revisited_true":"home.households.fuzzy_matching-details.revisited_false";
        this.person2Details.extHhStatus = this.data.extHhStatus?.toLowerCase() === "beneficiary"? "home.households.fuzzy_matching-details.beneficiary_true":"home.households.fuzzy_matching-details.beneficiary_false";
      }
      else{
        this.person2Details.showRevistedAndHHStatus = false;
      }

      })
  }

  public getDisplay(data, config): any {
    return this.languageHelper.getPropValue(data, config.key);
  }

  public checkConditionality(data, config) {
    //if there is condition
    if (config.showOnKey) {

      let conditionValue = this.languageHelper.getPropValue(data, config.showOnKey);

      if (config.relatedFeildLookup) {
        conditionValue = this.lookups[config.relatedFeildLookup]?.find(x => x.id === this.languageHelper.getPropValue(data, config.showOnKey)?.id)?.type;
      }

      //check if condition is true
      if (conditionValue == config.relatedFeildValue) {
        return true;
      } else {
        return false
      }

    } else {
      return true;
    }

  }

  public getLookup(data, configItem) {
    return this.lookups[configItem.lookupKey]?.find(x => x.id === this.getDisplay(data, configItem)?.id)?.name;
  }

  public getLookupArray(data, configItem) {
    let list = [];
    this.getDisplay(data, configItem).forEach(element => {
      list.push(this.lookups[configItem.lookupKey]?.find(x => x.id === element?.id)?.name);
    });
    return list.join(' ');
  }

  public getValue(data, config) {
    const value = this.languageHelper.getPropValue(data, config.key);
    return value ? 'audit_module.yes' : 'audit_module.no';
  }

  public moderateDuplicate(key: string) {
    this.isSubmitting = true;
    if (key === MODERATION_KEY.PERSON_1) {
      const payload: ModerateDuplicatePayload = {
        duplicateApproved: true,
        comment: null,
        suspendApplicationId: this.data.applicationId
      }
      this.sendRequestAndNavigateBack(payload);
    }

    if (key === MODERATION_KEY.PERSON_2) {
      const payload: ModerateDuplicatePayload = {
        duplicateApproved: true,
        comment: null,
        suspendApplicationId: this.data.duplicatedApplicationId
      }
      this.sendRequestAndNavigateBack(payload);
    }
  }

  public rejectDuplicate() {
    let message = 'confirmation_dialog.are_you_sure';
    if(this.data.duplicatedApplicationId==this.data.applicationId){
      message = 'confirmation_dialog.duplicateWithinTheFamily';
    }
    const dialogRef = this.dialog.open(ConfirmComponent, {
      panelClass: 'confirm-dialog',
      data: {
        confirmationMessage: message,
        width: '100%',
        maxWidth: '650px',
      },
    });

    dialogRef.afterClosed().subscribe((isConfirmed) => {
      if (isConfirmed === true) {
          this.isSubmitting = true;
          const payload: ModerateDuplicatePayload = {
          duplicateApproved: false,
          comment: null,
          suspendApplicationId: null
          }
          this.sendRequestAndNavigateBack(payload);
        }
    });
  }

  private sendRequestAndNavigateBack(payload: ModerateDuplicatePayload) {
    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: "details-dialog",
      disableClose: true,
      width: '400px',
      data: {

        lookups: this.lookups,
        fields: this.commentField,
        canClose: true,

      },
    });

    dialogRef.componentInstance.result.subscribe((result) => {
      if (!result) {
        this.isSubmitting = false;
        return;
      };
      payload.comment = result.comment;
      this._service.moderateDuplicate(this.duplicateApplicationId, payload)
        .pipe(finalize(() => this.isSubmitting = false))
        .subscribe(_ => {
          this.router.navigate(['../../'], { relativeTo: this.route });
          this.snackbar.openSnackBar(this.translateService.instant('snackbar.edit_success'), null, null, 2000);
          dialogRef.close();
        }, err => {
          if (err.code === 'conflict') {
            this.snackbar.openWarningSnackBar("snackbar.conflict_resolved");
            this.router.navigate(['../../'], { relativeTo: this.route });
            dialogRef.close();
          } else {
            this.snackbar.openWarningSnackBar('snackbar.edit_error');

          }
        });

    });
  }

  getLookups() {
    this.lookups = {};
    this.utilSerivce.getLookupsService(['ApplicationStatus']).subscribe(res => {
      this.lookups.ApplicationStatus = res.body.ApplicationStatus;
    })

    this.getData();

  }
  removePerson(id :number) {
    let appId : number = null;
    const dialogRef = this.dialog.open(ConfirmComponent, {
      panelClass: 'confirm-dialog',
      data: {
        confirmationMessage: 'confirmation_dialog.are_you_sure',
        width: '100%',
        maxWidth: '650px',
      },
    });

    const mappedPerson1 = Object.entries(this.data.persons1).map(([type, value]) => ({type, value}));
    mappedPerson1.map(item => {
      if(item.value?.personId == id){
        appId = this.data.applicationId
      }
    });

    if(appId == null){
      const mappedPerson2 = Object.entries(this.data.persons2).map(([type, value]) => ({type, value}));
      mappedPerson2.map(item => {
        if(item.value?.personId == id){
          appId = this.data.duplicatedApplicationId
        }
      });
    }

    if(appId && id){

    dialogRef.afterClosed().subscribe((result) => {
      if (result === true) {

        this._service.deletePerson(appId.toString(),id.toString()).subscribe(d=>{
          if(d['status']==200){
            this.getData();
          }
        })
      }

    });
  }
  }

  public clickEvent(child){

    this.personsIdsArray.forEach(e => {
      this.el.nativeElement.querySelector('.identifier_'+e)?.classList.remove('duplicateModerateClass')
    })

    this.el.nativeElement.querySelector('.identifier_'+(child.personId ? child.personId : child.personExtNbr))?.classList.add('duplicateModerateClass')
      let duplicates = child.duplicateIds;
      duplicates.forEach(e => {
        this.el.nativeElement.querySelector('.identifier_'+e)?.classList.add('duplicateModerateClass')
      })

  }

}
