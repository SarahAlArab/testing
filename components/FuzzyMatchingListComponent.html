import { ChangeDetectorRef, Component, Injector, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { HomeService } from 'src/app/home/home.service';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { LookupsAppStatusTypeEnum, LookupsEnum } from 'src/app/shared/models/lookups.enum';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { PaginatorConfig } from 'src/app/shared/models/paginator-config.model';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { HouseholdsService } from '../../households.service';
import { FuzzyMatchingService } from '../fuzzy-matching.service';
import { DuplicateListItem, FilterByType, FuzzyListColumnDef } from './fuzzy-matching-list.config';

@Component({
  selector: 'wfps-fuzzy-matching-list',
  templateUrl: './fuzzy-matching-list.component.html',
  styleUrls: ['./fuzzy-matching-list.component.sass']
})
export class FuzzyMatchingListComponent
  extends DataTableBaseClass<any>
  implements OnInit {
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.households.empty_table',
  };
  dataTableColumnsDefinition = FuzzyListColumnDef;
  filterByType=FilterByType;
  element: any;
  filterLookups: any = {};
  paginationIndex: any;
  
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  filterbyOptions: any;
  constructor(
    private _service: FuzzyMatchingService,
    public injector: Injector,
    public router: Router,
    public route: ActivatedRoute,
    private cdr: ChangeDetectorRef,
    public houseHoldService: HouseholdsService,
    private _homeService: HomeService,
    private utilsService: UtilsService,
  ) {
    super(injector);
  }

  ngOnInit(): void {
    this.getLookups(); 
    this.paginationIndex = this.houseHoldService.getFuzzyPaginationIndexValue();
    const withReset =
      this.paginationIndex && this.paginationIndex.length != 0 ? false : true;
   
    this.setFilter();
  
    this.getListing(this.paginationIndex, withReset, this.filter,true);  
          

    
   
  }

  getListing(
    paginationObj: PageChangeEvent,
    start: boolean,
    filter?: string,
    openPanel? :boolean,
  ) {
    if (paginationObj) {
      this.houseHoldService.setFuzzyPaginationIndexValue(paginationObj);
    }
    this._service.getDuplicateList(paginationObj,start, filter).subscribe(
      (data) => {
        if (data && data.body) {
          this.setDataTableData(data.body, start);
          if (this.filter && openPanel) {
            this.header.togglePanel();
          }
        }
      },
      (err) => { }
    );
  }

  actionClicked(ev: { name: string, data: DuplicateListItem }) {
    if (ev.name === 'moderate') {
      this.router.navigate([`${ev.data.id}/moderate`], {
        relativeTo: this.route,
      });
    }
  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getListing(pageChangeEvent, false, this.filter);
  }

  searchFilter(events) {
    //The filter is a string of key:value seperated by comma
    this.filter = this.frameworkHelper.getFilter2String(events, 'issues');

    //save filter in global storage
    this.houseHoldService.addFilter({ component: 'fuzzyMatching', filter: events });

    //Call List API
    this.getListing(this.paginatorConfig, true, this.filter);

    //Clear filters
    if (events == null) {
      this.header.togglePanel();
    }
  }

  setFilter() {
    let filterValues = this.houseHoldService
      .getFilterVal()
      ?.find((f) => f.component === 'fuzzyMatching')?.filter;

    if (filterValues) {
      this.filter = this.frameworkHelper.getFilter2String(
        filterValues
      );
      this.element = this.frameworkHelper.formatElement(filterValues);
    }
  }

  clickBack() {
    this.router.navigate(['../'], { relativeTo: this.route })
  }

  getLookups() {
    this.utilsService.getLookupsService(['ApplicationStatus']).subscribe(res => {
      this.filterLookups.ApplicationStatus = res.body.ApplicationStatus;

    })



  }



}
