import { Component, EventEmitter, Inject, Injector, OnInit, Output, ViewChild } from '@angular/core';

@Component({
  selector: 'wfps-payment-history-dialog',
  templateUrl: './history-payment-dialog.component.html',
  styleUrls: ['./history-payment-dialog.component.sass']
})
export class HistoryPaymentDialogComponent {
  @ViewChild(CustomFormComponent) customFormComponent?: CustomFormComponent;
  form: FormGroup;
  close = false;
  error: string;
  dataTableData;
  dataTableData2;
  fieldTypeEnum = FieldType;
  filterLookups: any = {};
  frameworkHelper = FrameworkHelper;
  @Output() result = new EventEmitter<any>();
  @Output() fieldChange = new EventEmitter<any>();

  cancelAction: CustomFormActionModel = new CustomFormActionModel('generic.table.delete');
  // emptyTable = {
  //   icon: 'assets/icons/institution_empty.svg',
  //   text: 'home.call_center.list.empty_table_payment'
  // };
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.call_center.list.empty_table'
  };
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  modificationHistoryColumnDef=ModificationHistoryColumnDef;
  dataTableColumnsDefinition=HistoryPaymentColumn;
  paymentDetailsColumndefinition=HistoryPaymentDetailsColumn;
  isLookupsLoaded: boolean;

  constructor(
    public router: Router,
    public  route: ActivatedRoute,
    private fb: FormBuilder,
    public dialogRef: MatDialogRef<DetailsDialogComponent>,
    public injector: Injector,
    public utilService: UtilsService,
    public _homeService: HomeService,
    public _service: HouseholdsService,
    public translateService: TranslateService,
    private dialog: MatDialog,
    @Inject(MAT_DIALOG_DATA) public data: HistoryDialogModel<any, any>
  ) {
  }

  ngOnInit(): void {
    this.isLookupsLoaded = false;
    this.getLookups().subscribe(() => {
      this.prepareDatatables();
      this.isLookupsLoaded = true;
    });
  }
  
  getLookups() {
    const displayedStatus = [
      LookupsAppStatusTypeEnum.SCHEDULED,
      LookupsAppStatusTypeEnum.ESSN_IN_PROGRESS,
      LookupsAppStatusTypeEnum.ESSN_SUBMITTED,
      LookupsAppStatusTypeEnum.ESSN_REJECTED,
      LookupsAppStatusTypeEnum.ESSN_APPROVED,
      LookupsAppStatusTypeEnum.ESSN_AUTHORIZED,
      LookupsAppStatusTypeEnum.ESSN_UNAUTHORIZED,
      LookupsAppStatusTypeEnum.PAY_ENROLLED,
      LookupsAppStatusTypeEnum.PAY_CANCELLED,
      LookupsAppStatusTypeEnum.SUSPENDED,
    ];
    
    return this._homeService.getFilteredApplicationStatus().pipe(
      map((response) => {
        const lookuptsToReturn = response.body.map((x) => {
          x.name = '';
          return x;
        });
        return { body: lookuptsToReturn };
      }),
      switchMap((s) => {
        this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS] =
          s.body?.filter((f) => displayedStatus.includes(f.type));
        
        return this.utilService.getLookupsService(['AidPaymentStatus']);
      }),
      map(res => {
        this.filterLookups.aidPaymentStatus = res.body.AidPaymentStatus;
        return res;
      })
    );
  }

  prepareDatatables(){
    this.dataTableData= this.prepareListForDataTable(this.data.res1DataTable.content)
    this.dataTableData2 = this.data.res2DataTable.content;
  }

  prepareListForDataTable(list) {
    let newList = list;
    newList.forEach((l) => {
      l.issueDateOfYear = formatDate(l.issueDate, 'yyyy-MM-dd', 'en');
      l.aidPaymentStatus = this.filterLookups.aidPaymentStatus.find(f => f.id === l.paymentStatusId).name;
    });
    return newList;

  }

  public apply(event) {
    if (event.apply) {
      this.buttonClicked(event.formVal);
    }
    else {
      this.closeDialog(true);
    }
  }

  public buttonClicked(formValToSave) {
    let formVal;
    if (formValToSave) {
      formVal = Object.assign(this.data.data || {}, formValToSave);
      this.result.emit(formVal);
    } else {
      this.dialogRef.close(formVal);
    }
  }

  public closeDialog(val: boolean) {
    if (this.customFormComponent.form.pristine) {
      this.buttonClicked(false);
    } else this.close = val;
  }

  changeField(event) {
    this.fieldChange.emit(event);
  }

  goBack() {
    this.dialogRef.close()
  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'details':
        this._service.getHistoryPaymentsDetails(ev).subscribe((resDataTable) => {
          const dialogRef = this.dialog.open(DetailsDialogComponent,{
            panelClass: 'details-dialog',
            disableClose: false,
            width: '60%', 
            data: {
              dataTableColumnsDefinition:  this.paymentDetailsColumndefinition,
              dataTableData: resDataTable, 
              modalTitle: `${this.translateService.instant(
                'history_table.paymentDetails'
              )}`,          
              showTable: resDataTable.content.length > 0
            },
          })
        });
    }
  }

}
