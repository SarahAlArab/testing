import { formatDate } from '@angular/common';
import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { Subscription } from 'rxjs';
import { HomeService } from 'src/app/home/home.service';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { DATA_TABLE_COLUMN_FORMAT } from 'src/app/shared/enums/data-table-column-format';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { FilterConfig } from 'src/app/shared/models/filter-config.model';
import {
  FilterbyOptions,
  ProcessColumnDef,
} from 'src/app/home/households/history-process/history-process.config';

import {
  LookupsAppStatusTypeEnum,
  LookupsEnum,
} from 'src/app/shared/models/lookups.enum';

} from 'src/app/shared/models/user-role.enum';

@Component({
  selector: 'wfps-history-process',
  templateUrl: './history-process.component.html',
  styleUrls: ['./history-process.component.sass'],
})
export class HistoryProcessComponent
  extends DataTableBaseClass<any>
  implements OnInit
{
  dataTableColumnsDefinition = ProcessColumnDef;
  filterbyOptions = FilterbyOptions;
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  filterConfig = new FilterConfig(ConfigTypeEnum.NONE);
  lookups: any = {};
  element: any = {};
  filterLookups: any = {};
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.households.empty_table',
  };
  moduleName = UserModulesEnum.HOUSEHOLDS;
  canSchedule: boolean = true;
  canReview: boolean = true;
  canViewPaymentsHistory: boolean = true;
  canAuthorize: boolean = true;
  canReportIssue: boolean = true;
  canResolveIssue: boolean = true;
  paginationIndex: any;
  report: any = {};
  viewComments: boolean = true;
  beneficiarySubscription: Subscription;
  canViewHistory: boolean;
  generateSubscription: Subscription;
  isLookupsLoading: boolean;
  isMeheUser: boolean;

  constructor(
    public _service: HouseholdsService,
    private dialog: MatDialog,
    public injector: Injector,
    public translateService: TranslateService,
    private utilsService: UtilsService,
    private _homeService: HomeService
  ) {
    super(injector);
  }

  ngOnInit(): void {
    this.isMeheUser =
      this.authService.getUserRole() === UserRolesEnum.MEHE_USER;
    this.getProcessTrackingTypesAndStatuses();
    this.dataTableColumnsDefinition = JSON.parse(
      JSON.stringify(ProcessColumnDef)
    );

    super.ngOnInit();
    this.setFilter();

    this.paginationIndex = this._service.getPaginationIndexValue();
    const withResett =
      this.paginationIndex && this.paginationIndex.length != 0 ? false : true;
    this.getTrackingProcess(
      this.paginationIndex,
      withResett,
      this.filter,
      true
    );
  }

  getTrackingProcess(
    paginationObj: PageChangeEvent,
    start: boolean,
    filter?: string,
    openPanel?: boolean
  ) {
    if (paginationObj) {
      this._service.setPaginationIndexValue(paginationObj);
    }

    this._service.getTrackingProcess(paginationObj, start, filter).subscribe(
      (data) => {
        if (data && data.body) {
          this.updateListData(data.body.content);

          for (var content in data.body.content) {
            let type = this.translateService.instant(
              'home.households.process_tracking.' +
                data.body.content[content].type.toString().toLowerCase() +
                '_en'
            );
            let status = this.translateService.instant(
              'home.households.process_tracking.' +
                data.body.content[content].status.toString().toLowerCase() +
                '_en'
            );
            data.body.content[content].type = type;
            data.body.content[content].status = status;
          }
          this.setDataTableData(data.body, start);
        }
      },
      (err) => {}
    );
  }

  updateListData(data) {}

  setFilter() {
    let filterValues = this._service
      .getFilterVal()
      ?.find((f) => f.component === 'assigned')?.filter;
    if (filterValues) {
      this.filter = this.frameworkHelper.getFilterString(
        filterValues,
        'issues',
        'hasMobile'
      );
      this.element = this.frameworkHelper.formatElement(filterValues);
    }
  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getTrackingProcess(pageChangeEvent, false, this.filter);
  }

  searchFilter(events) {
    //The filter is a string of key:value seperated by comma
    this.filter = this.frameworkHelper.getFilterString(
      events,
      'issues',
      'hasMobile'
    );

    //save filter in global storage
    this._service.addFilter({ component: 'assigned', filter: events });

    //Call List API
    this.getTrackingProcess(null, true, this.filter);

    //Clear filters
    if (events == null) {
      this.header.togglePanel();
    }
  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'message':
        {
          this.dialog.open(WarningDialogComponent, {
            width: '400px',
            data: { arabic: ev.data.message, canClose: true },
          });
        }
        break;
      case 'logs':
        {
          this._service.getProcessDataLog(ev.data.id).subscribe((response) => {
            if (response) {
              UtilsService.downloadFile(response);
            } else {
              this.snacks.openWarningSnackBar('No data available to download.');
            }
            (error) => {
              this.snacks.openWarningSnackBar(error.message);
            };
          });
        }
        break;
      default:
        break;
    }
  }

  ngOnDestroy() {
    this.beneficiarySubscription && this.beneficiarySubscription.unsubscribe();
    if (this.generateSubscription) {
      this.generateSubscription.unsubscribe();
    }
  }

  getProcessTrackingTypesAndStatuses() {
    let dropdownFields = this.filterbyOptions.filter(
      (f) => f.type == FieldType.dropdown
    );
    let keys = dropdownFields.map((m) => {
      return m.lookupKey;
    });
    this.isLookupsLoading = true;
    this._service
      .processTrackingTypesAndStatuses()
      .pipe(
        finalize(() => {
          setTimeout(() => {
            this.getDataForMEHEUser();
          });
        })
      )
      .subscribe((s) => {
        for (var status in s.body.ProcessTrackingStatus) {
          let name = this.translateService.instant(
            'home.households.process_tracking.' +
              s.body.ProcessTrackingStatus[status].type.toString().toLowerCase()
          );
          let nameEn = this.translateService.instant(
            'home.households.process_tracking.' +
              s.body.ProcessTrackingStatus[status].type
                .toString()
                .toLowerCase() +
              '_en'
          );
          s.body.ProcessTrackingStatus[status].name = name;
          s.body.ProcessTrackingStatus[status].nameEn = nameEn;
        }

        for (var type in s.body.ProcessTrackingType) {
          let name = this.translateService.instant(
            'home.households.process_tracking.' +
              s.body.ProcessTrackingType[type].type.toString().toLowerCase()
          );
          let nameEn = this.translateService.instant(
            'home.households.process_tracking.' +
              s.body.ProcessTrackingType[type].type.toString().toLowerCase() +
              '_en'
          );
          s.body.ProcessTrackingType[type].name = name;
          s.body.ProcessTrackingType[type].nameEn = nameEn;
        }

        this.filterLookups = s.body;
      });
  }

  getDataForMEHEUser() {
    if (this.isMeheUser) {
      let type = this.filterbyOptions.find((f) => f.prop === 'type');
      let eduType = this.filterLookups.ProcessTrackingType.find(
        (e) => e.id === 'EDUCATIONAL_UPLOAD'
      );
      type.isDisabled = true;
      type.defaultValue = eduType.id;
      this.filter = 'type:' + eduType.id;
      this.paginationIndex = this._service.getPaginationIndexValue();
      const withResett =
        this.paginationIndex && this.paginationIndex.length != 0 ? false : true;
      this.getTrackingProcess(
        this.paginationIndex,
        withResett,
        this.filter,
        true
      );
      this.isLookupsLoading = false;
    } else {
      this.isLookupsLoading = false;
      setTimeout(() => {
        this.header.togglePanel();
      });
    }
  }
}
