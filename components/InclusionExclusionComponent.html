import { Component, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { ActivatedRoute, Router } from '@angular/router';
import { TranslateService } from '@ngx-translate/core';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { AuthService } from 'src/app/shared/services/auth.service';
import { SnackbarService } from 'src/app/shared/services/snackbar.service';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { HouseholdsService } from '../households.service';

@Component({
  selector: 'wfps-inclusion-exclusion',
  templateUrl: './inclusion-exclusion.component.html',
  styleUrls: ['./inclusion-exclusion.component.sass']
})
export class InclusionExclusionComponent implements OnInit {
  canRerunInclusionExclusion: boolean;
  canRunInclusionExclusion: boolean;
  canApplyInclusionExclusion: boolean;
  runIncExcFileTypeId: number;
  applyStatusFileType: number;
  canAccessReportGenerater: boolean;

  constructor(public dialog: MatDialog,
    private router: Router,
    private route: ActivatedRoute,
    public translateService: TranslateService,
    public _service: HouseholdsService,
    public snacks: SnackbarService,
    private utilSerivce: UtilsService,
    private authService: AuthService
    ) { }

  ngOnInit(): void {
    this.getLookups();

    this.canRerunInclusionExclusion = this.authService.isPageAccessible('RUN_INCLUSION_EXCLUSION_PROCESS');
    this.canRunInclusionExclusion   = this.authService.isPageAccessible('BACKUP_DELETE_RUN_INCLUSION_EXCLUSION_PROCESS');
    this.canApplyInclusionExclusion = this.authService.isPageAccessible('APPLY_INCLUSION_EXCLUSION_RESULTS');
    this.canAccessReportGenerater= this.authService.isPageAccessible('VIEW_REPORTS_LIST');

  }

  executeInclusionExclusion(fullProcess: boolean) {
    this._service.runInclusionExclusion(fullProcess).subscribe(res => {
      if(res && res.status == 200){
        if (this.canAccessReportGenerater && this.runIncExcFileTypeId > 0) {
          this.navigateToGeneratedReport('../generated-reports', this.runIncExcFileTypeId);
        }
        else {
          this.snacks.openSnackBar(
            `${this.translateService.instant(
              'home.households.report_generated_sucessfully'
            )}`
          );
        }
      }else{
        //returned 208 ALREADY_REPORTED, a successful series
        this.snacks.openSnackBar(
            `${this.translateService.instant(
              'snackbar.apply_inclusion_exclusion_done_previously'
            )}`
          );
        this.navigateToGeneratedReport('../generated-reports', this.runIncExcFileTypeId);
      }
      
    }, (err) => {
      this.dialog.open(WarningDialogComponent, {
        width: '400px',
        data: { arabic: this.translateService.instant(err.message), canClose: true }
      });
    });
  }

  getLookups() {
    this.utilSerivce.getLookupsService(['FilesType']).subscribe(s => {
      if (s.body && s.body.FilesType) {
        this.runIncExcFileTypeId = s.body.FilesType.find(e=>e.type == 'BACKUP_AND_RUN_INCLUSION_EXCLUSION')?.id;
        this.applyStatusFileType = s.body.FilesType.find(e=>e.type == 'STATUS_APPLY_INCLUSION_EXCLUSION')?.id;
      }
    });
  }

  applyInclusionExclusion(){
    this._service.applyInclusionExclusion().subscribe(res => {
      if (this.canAccessReportGenerater && this.runIncExcFileTypeId > 0) {
        this.navigateToGeneratedReport('../generated-reports', this.applyStatusFileType);
      }
      else {
        this.snacks.openSnackBar(
          `${this.translateService.instant(
            'home.households.report_generated_sucessfully'
          )}`
        );
      }
    }, (err) => {
      this.dialog.open(WarningDialogComponent, {
        width: '400px',
        data: { arabic: this.translateService.instant(err.message), canClose: true }
      });
    });
  }

  private dyanmicDownloadByHtmlTag(arg: {
    fileName: string,
    text: string
  }) {
    const element = document.createElement('a');
    const fileType = arg.fileName.indexOf('.json') > -1 ? 'text/json' : 'text/plain';
    element.setAttribute('href', `data:${fileType};charset=utf-8,${encodeURIComponent(arg.text)}`);
    element.setAttribute('download', arg.fileName);

    var event = new MouseEvent("click");
    element.dispatchEvent(event);
  }

  navigateToGeneratedReport(route,id) {
    this.router.navigate([route], { relativeTo: this.route, queryParams:{'id':id} });
  }
}
