import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { UserModulesEnum } from 'src/app/shared/models/user-role.enum';
import { AuthService } from 'src/app/shared/services/auth.service';

@Component({
  selector: 'wfps-landing-distribution',
  templateUrl: './landing-distribution.component.html',
  styleUrls: ['./landing-distribution.component.sass']
})
export class LandingDistributionComponent implements OnInit {
  assignPermission = UserModulesEnum.ASSIGN.module;
  reschedulePermission = UserModulesEnum.RESCHEDULE.module;
  reassignForAddresPermission = UserModulesEnum.REASSIGN_ADDRESS.module;
  assignSpotcheckPermission = "SPOT_ESSN_APP_MGT.SPOT_ESSN_APP_ASSIGN_BATCH";

  constructor(
    private router: Router,
    private route: ActivatedRoute,
    public authService: AuthService
  ) { }

  ngOnInit(): void {
  }

  isSectionAccessible(permission: string) {
    return this.authService.isPageAccessible(permission)
  }

  navigateTo(route) {
    this.router.navigate([route], { relativeTo: this.route });
  }
}
