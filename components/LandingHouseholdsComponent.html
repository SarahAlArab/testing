  templateUrl: './landing-households.component.html',
  styleUrls: ['./landing-households.component.sass'],

export class LandingHouseholdsComponent implements OnInit {
  assignSpotcheckPermission = 'SPOT_ESSN_APP_MGT.SPOT_ESSN_APP_SCHEDULE';
  externalDupPermission = 'DUPLICATE_PROGRAM_OPERATION';
  inspectSearch = 'INSPECT_SEARCH_AID_CARD_APP';
  reportDownload = 'ESSN_PAYMENT_REPORT_DOWNLOAD';
  applicationsList = 'AIDCARD_APP_LIST';
  ciAudit = "AID_CARD_APP_AUDIT_HISTORY";
  nptpMigration = 'NPTP_MIGRATION_MGT';
  createAppUser = "CREATE_APP_USERS";
  isMigrationAllowed: boolean;
  enrollEducationYear = 'ENROLL_ELIGIBLE_STUDENTS_IN_PAYMENT_PLAN';
  meheOperations = 'MEHE_OPERATIONS';

  downloadReportDetails = DownloadReportDetails;

  isDuplicateModerateAccessible: boolean;
  isCaseInspectionAccessible: boolean;
  isDownloadDataAllowed: boolean;
  isViewApplicationsAllowed: boolean;
  isViewCiAuditAllowed: boolean;
  educationalId = null;
  canAccessReportGenerater: boolean;
  canAccessInclusionExclusion: boolean;
  showProcessTrackingHistory: boolean;
  enrollEducationYearAllowed: boolean;
  meheOperationsAllowed: boolean;
  programTypes = [];
  fileTypeId;
  lookups: any = {};
  downloadEducationPersonsDetails = DownloadEducationPersons;
  formDetails: CustomFormModel = new CustomFormModel(
    this.downloadEducationPersonsDetails

  constructor(
    private router: Router,
    private dialog: MatDialog,
    private route: ActivatedRoute,
    private authService: AuthService,
    private utilsService: UtilsService,
    public snacks: SnackbarService,
    public translateService: TranslateService,
    public _service: HouseholdsService,
    private utilSerivce: UtilsService
  ) {}

  ngOnInit(): void {
    this.getLookups();

    this.isDuplicateModerateAccessible = this.authService.isPageAccessible(
      `${UserModulesEnum.DUPLICATE.module}.${UserModulesEnum.DUPLICATE.key}_${UserRolesActionsEnum.LIST}`
    );
    this.isCaseInspectionAccessible = this.authService.isPageAccessible(
      this.inspectSearch
    );
    this.isDownloadDataAllowed = this.authService.isPageAccessible(
      this.reportDownload
    );
    this.isViewApplicationsAllowed = this.authService.isPageAccessible(
      this.applicationsList
    );
    this.isViewCiAuditAllowed = this.authService.isPageAccessible(this.ciAudit);
    this.enrollEducationYearAllowed = this.authService.isPageAccessible(
      this.enrollEducationYear
    );
    this.meheOperationsAllowed = this.authService.isPageAccessible(
      this.meheOperations
    );
    this.canAccessReportGenerater =
      this.authService.isPageAccessible('VIEW_REPORTS_LIST');
    this.canAccessInclusionExclusion =
      this.isNptpUser() &&
      (this.isAllowed('BACKUP_DELETE_RUN_INCLUSION_EXCLUSION_PROCESS') ||
        this.isAllowed('RUN_INCLUSION_EXCLUSION_PROCESS') ||
        this.isAllowed('APPLY_INCLUSION_EXCLUSION_RESULTS'));

    this.showProcessTrackingHistory = this.authService.isPageAccessible(
      `${UserModulesEnum.PROCESS_TRACKING_HISTORY.module}`
    );
    this.isMigrationAllowed =  this.isNptpUser() && this.authService.isPageAccessible(this.nptpMigration);



    let FilesTypeArr = ['FilesType', 'ProgramType'];

    this.utilsService.getLookupsService(FilesTypeArr).subscribe((s) => {
      if (s.body) {
        FilesTypeArr.forEach((k) => {
          s.body[k].forEach((l) => {
            if (l.type == 'ESSN_PAYMENT') {
              this.educationalId = l.id;
            } else if (l.type === 'EXT_PROGRAM_DEDUPLICATE') {
              this.fileTypeId = l.id;
            }
          });
        });

        this.programTypes = s.body.ProgramType;
      }
    });

    this.canAccessInclusionExclusion =
      this.isNptpUser() &&
      (this.isAllowed('BACKUP_DELETE_RUN_INCLUSION_EXCLUSION_PROCESS') ||
        this.isAllowed('RUN_INCLUSION_EXCLUSION_PROCESS') ||
        this.isAllowed('APPLY_INCLUSION_EXCLUSION_RESULTS'));
  }

  navigateTo(route) {
    this.router.navigate([route], { relativeTo: this.route });
  }

  isSectionAccessible(isAssigned: boolean) {
    if (isAssigned) {
      return (
        this.authService.isPageAccessible(
          `${UserModulesEnum.HOUSEHOLDS.module}.${
            UserModulesEnum.HOUSEHOLDS.key
          }${'_ASSIGNED'}_${UserRolesActionsEnum.LIST}`
        ) || this.authService.isPageAccessible(`ACCESS_APPLICATIONS`)
      );
    } else {
      return this.authService.isPageAccessible(
        `${UserModulesEnum.HOUSEHOLDS.module}.${UserModulesEnum.HOUSEHOLDS.key}_${UserRolesActionsEnum.LIST}`
      );
    }
  }

  isAllowed(permission: string) {
    return this.authService.isPageAccessible(permission);
  }

  downloadFile() {
    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: 'details-dialog',
      disableClose: true,
      width: '80%',
      data: {
        fields: this.downloadReportDetails,
        modalTitle: 'home.households.download_history_files',
        detailsTitle: 'home.households.Payment_report_details',
      },
    });
    dialogRef.componentInstance.result.subscribe((result) => {
      if (result) {
        const dialogRefConfirmation = this.dialog.open(
          ConfirmationDialogComponent,
          {
            width: '400px',
            data: {
              body: `${this.translateService.instant(
                'confirmation_dialog.are_you_sure'
              )}`,
              saveButtonText: 'confirmation_dialog.confirm',
              discardButtonText: 'confirmation_dialog.reject',
            },
          }
        );

        dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
          if (confirmation) {
            if (!result.batchNumber) {
              delete result.batchNumber;
            }
            this._service.downloadReport(result).subscribe(
              (data) => {
                if (this.canAccessReportGenerater) {
                  this.navigateToGeneratedReport(
                    'generated-reports',
                    this.educationalId
                  );
                } else {
                  this.snacks.openSnackBar(
                    `${this.translateService.instant(
                      'home.households.report_generated_sucessfully'
                    )}`
                  );
                }
              },
              (err) => {
                this.snacks.openWarningSnackBar(err.description);
              }
            );
          }
        });

        dialogRef.close();
      }
    });
  }
  createNptpMigUser() {
    const dialogRefConfirmation = this.dialog.open(
      ConfirmationDialogComponent,
      {
        width: '400px',
        data: {
          body: `${this.translateService.instant(
            'confirmation_dialog.are_you_sure'
          )}`,
          saveButtonText: 'confirmation_dialog.confirm',
          discardButtonText: 'confirmation_dialog.reject',
        },
      }
    );
    dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
      if (confirmation) {
        this.authService.createNptpMigrationUsers().subscribe(
          _ => {
            this.snacks.openSnackBar("snackbar.send_success");
          }, error => {
            this.handleConflictError('conflict', error.message);
          })
      }
    })
   
  }
  isNptpOrWfpUser() {
    return (
      this.authService.getUserRole() == UserRolesEnum.NPTP_USER ||
      this.authService.getUserRole() == UserRolesEnum.WFP_ADMIN
    );
  }

  isNptpUser() {
    return this.authService.getUserRole() == UserRolesEnum.NPTP_USER;
  }

  isNptpOrWfpOrMeheUser() {
    return (
      this.authService.getUserRole() == UserRolesEnum.NPTP_USER ||
      this.authService.getUserRole() == UserRolesEnum.MEHE_USER
    );
  }

  navigateToGeneratedReport(route, id) {
    this.router.navigate([route], {
      relativeTo: this.route,
      queryParams: { id: id },
    });
  }

  isMophUser() {
    return this.authService.getUserRole() == UserRolesEnum.MOPH_USER;
  }

  createMophUsers() {
    const dialogRefConfirmation = this.dialog.open(
      ConfirmationDialogComponent,
      {
        width: '400px',
        data: {
          body: `${this.translateService.instant(
            'confirmation_dialog.are_you_sure'
          )}`,
          saveButtonText: 'confirmation_dialog.confirm',
          discardButtonText: 'confirmation_dialog.reject',
        },
      }
    );
    dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
      if (confirmation) {
        this.authService.createMophUsers().subscribe(
          _ => {
            this.snacks.openSnackBar("snackbar.send_success");
          }, error => {
            this.handleConflictError('conflict', error.message);
          })
      }
    })
  }
  getLookups() {
    let dropdownFields = this.downloadEducationPersonsDetails.filter(
      (f) => f.type === FieldType.dropdown
    );
    let keys = dropdownFields.map((m) => {
      return m.lookupKey;
    });
    this.utilSerivce
      .getLookupsService([LookupsEnum.EDUCATION_REGISTRATION_YEAR])
      .subscribe((result) => {
        if (result.body) {
          this.lookups[LookupsEnum.EDUCATION_REGISTRATION_YEAR] =
            result.body[LookupsEnum.EDUCATION_REGISTRATION_YEAR];
          this.formDetails.lookups = this.lookups;
        }
      });
  }
  enrollStudentEducationYear() {
    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: 'details-dialog',
      disableClose: true,
      width: '80%',
      data: {
        lookups: this.lookups,
        fields: this.downloadEducationPersonsDetails,
        canClose: true,
        modalTitle: 'home.households.download_history_files',
      },
    });
    dialogRef.componentInstance.result.subscribe((result) => {
      if (result) {
        const dialogRefConfirmation = this.dialog.open(
          ConfirmationDialogComponent,
          {
            width: '400px',
            data: {
              body: `${this.translateService.instant(
                'confirmation_dialog.are_you_sure'
              )}`,
              saveButtonText: 'confirmation_dialog.confirm',
              discardButtonText: 'confirmation_dialog.reject',
            },
          }
        );

        dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
          if (confirmation) {
            this._service.enrollStudents(result).subscribe(
              (data) => {
                dialogRef.close();

                this.snacks.openSnackBar(
                  `${this.translateService.instant(
                    'home.households.report_generated_sucessfully'
                  )}`
                );
              },
              (err) => {
                this.dialog.open(WarningDialogComponent, {
                  width: '400px',
                  data: {
                    arabic: this.translateService.instant(err.message),
                    canClose: true,
                  },
                });
              }
            );
          }
        });
      }
    });
  }

  scriptForNullRegistrationStatus() {
    let sql1 =
      '- update aid_card."EducationStatusHistory" esh set "EducationRegistrationStatus_Id" = ( select ers."Id" from aid_card."EducationRegistrationStatus" ers where ers."Type" = \'PRE_REGISTRATION_IN_PROGRESS\'), "StatusDetails" = \'This status was changed by the system from PRE_REGISTRATION_SUBMITTED to PRE_REGISTRATION_IN_PROGRESS due to requirment sent to allow the user to edit multiple times\' where esh."EducationRegistration_Id" in ( select per ."Id" from aid_card."PersonEducationRegistration" per where per."EducationRegistrationStatus_Id" is null and "EducationRegistrationYear_Id" = ( select "Id" from aid_card."EducationRegistrationYear" ery where ery."Type" = \'2022_2023\') ) and esh."EducationRegistrationStatus_Id" = ( select ers."Id" from aid_card."EducationRegistrationStatus" ers where ers."Type" = \'PRE_REGISTRATION_SUBMITTED\');';
    let sql2 =
      '- update aid_card."PersonEducationRegistration" set "EducationRegistrationStatus_Id" = ( select ers."Id" from aid_card."EducationRegistrationStatus" ers where ers."Type" = \'PRE_REGISTRATION_IN_PROGRESS\' ) where "EducationRegistrationStatus_Id" is null and "EducationRegistrationYear_Id" =( select "Id" from aid_card."EducationRegistrationYear" ery where ery."Type" = \'2022_2023\');';
    const dialogRef = this.dialog.open(ConfirmComponent, {
      panelClass: 'confirm-dialog',
      data: {
        confirmationMessage:
          'home.households.update_null_registrationedu_status_confirmation',
        confirmationMessage_en:
          'home.households.update_null_registrationedu_status_confirmation_en',
        additionalInfo: [{ text: sql1 }, { text: sql2 }],
        width: '100%',
        maxWidth: '650px',
      },
    });

    dialogRef.afterClosed().subscribe((result) => {
      if (result === true) {
        this._service.scriptForNullRegistrationStatus().subscribe(
          (res) => {
            this.dialog.open(InfoDialogComponent, {
              width: '60%',
              height: '25%',
              data: {
                texts: [
                  {
                    textString: JSON.stringify(res?.body, null, 2), // Use an indentation value of 2
                    isEnglish: true,
                  },
                ],
                actions: [
                  {
                    label: 'login_module.close',
                    isClose: true,
                    returnedValue: false,
                  },
                ],
              },
            });
          },
          (error) => {
            this.handleConflictError('conflict', error.message);
          }
        );
      }
    });
  }

  handleConflictError(error, message?: string) {
    if (error == 'conflict') {
      this.dialog.open(WarningDialogComponent, {
        width: '400px',
        data: {
          english: message,
          canClose: true,
        },
      });
      return;
    }
  }

  uploadFileForComparison() {
    const dialogRef = this.dialog.open(ExternalDeduplicationDialogComponent, {
      panelClass: 'details-dialog',
      disableClose: true,
      width: '80%',
      maxWidth: '650px',
      data: {
        programTypeData: this.programTypes,
      },
    });

    dialogRef.afterClosed().subscribe((result) => {
      if (result.file && result.id) {
        this._service
          .uploadExternalDuplicationDoc(result.file, result.id)
          .subscribe((res) => {
            this.snacks.openSnackBar(
              this.translateService.instant(
                'home.households.waitTillProcessed'
              ),
              'X',
              'bottom',
              4000
            );

            if (this.fileTypeId) {
              this.navigateToGeneratedReport(
                'generated-reports',
                this.fileTypeId
              );
            }
          });
      }
    });
  }
