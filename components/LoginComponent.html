import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { Router } from '@angular/router';
import { TranslateService } from '@ngx-translate/core';
import { Subscription } from 'rxjs';
import { InfoDialogComponent } from 'src/app/shared/components/dialogs/info-dialog/info-dialog.component';
import { CodeVerificationDialogComponent } from '../../shared/components/code-verification-dialog/code-verification-dialog.component';
import { WarningDialogComponent } from '../../shared/components/dialogs/warning-dialog/warning-dialog.component';
import { CONFIG_KEYS_ENUM } from '../../shared/models/config.enum';
import { AuthService } from '../../shared/services/auth.service';
import { SpinnerService } from '../../shared/services/spinner.service';
import { UtilsService } from '../../shared/services/utils.service';
import { LandingService } from '../landing.service';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.sass']
})
export class LoginComponent implements OnInit {
  form: FormGroup
  error: string;
  loginSubscription: Subscription;
  sectorDropDown: any;
  passwordError = false;
  isLoginVisited = false;
  showPass = false;
  isSubmitting = false;
  passwordFocused = false;
  wrongCredentials: boolean = false;

  constructor(
    private landingService: LandingService,
    private router: Router,
    private dialog: MatDialog,
    private authService: AuthService,
    private _snackbar: MatSnackBar,
    private translateService: TranslateService,
    public utilsService: UtilsService,
    private spinnerService: SpinnerService
  ) { }

  ngOnInit(): void {
    this.form = new FormGroup({
      username: new FormControl(null, Validators.required),
      password: new FormControl(null, Validators.required),
      legalCommitment: new FormControl(null, Validators.required)
    });
    this.form.controls.legalCommitment.valueChanges.subscribe(x => {
      if (x == false) {
        this.form.controls.legalCommitment.setValue(null);
      }
    });
    this.form.valueChanges.subscribe(x => {
      if (this.form.controls.username.errors && this.form.controls.username.hasError('incorrect')) {
        delete this.form.controls.username.errors['incorrect'];
        if (Object.keys(this.form.controls.username.errors).length === 0) {
          this.form.controls.username.setErrors(null);
        }
      }
      if (this.form.controls.password.errors && this.form.controls.password.hasError('incorrect')) {
        delete this.form.controls.password.errors['incorrect'];
        if (Object.keys(this.form.controls.password.errors).length === 0) {
          this.form.controls.password.setErrors(null);
        }
      }
    });
  }

  /**
   * If login enabled call submitLogin() else open a warning popup
   */
  submit() {

    let user = this.authService.getUser();
    if (user) {
      this.landingService.preventOverLogin(user);
      return;
    }

    if (this.form.invalid) {
      this.form.controls.legalCommitment.markAsTouched();
      return;
    }

    this.isSubmitting = true;

    this.utilsService.getConfig(CONFIG_KEYS_ENUM.LOGIN_ENABLED, true).subscribe(d => {
      if (d && d.body && d.body.value == "true") {
        this.submitLogin();
      }
      else {
        this.dialog.open(WarningDialogComponent, {
          width: '400px',
          data: { arabic: 'errors.login_blocked', english: '', canClose: true }
        });
        this.isSubmitting = false;
      }
    });
  }

  /**
   * Check recapthca and login, open phone verification if OTP requested
   */
  submitLogin() {
    if (this.form.valid) {
      const user = this.form.getRawValue();
      const username = user.username !== undefined && user.username !== null ? user.username.trim() : '';
      const password = user.password !== undefined && user.password !== null ? user.password.trim() : '';

      this.loginSubscription = this.authService.login(username, password).subscribe(data => {
        if (!data?.body.otpRequested) {
          let user = data.body;
          user.username = username;
          this.authService.setUser(user);
          this.landingService.getUser();
        } else {
          return this.openPhoneVerification();
        }
        this.wrongCredentials = false;
      }, err => {
        this.spinnerService.forceHideLoader();
        switch (err.code ? err.code : err) {
          case 'not_authorized':
            this.wrongCredentials = true;
            this.form.controls.username.setErrors({ 'incorrect': true });
            this.form.controls.password.setErrors({ 'incorrect': true });
            this._snackbar.open(this.translateService.instant('login_module.invalid_credentials'), null, { duration: 1500 });
            break;
          case 'account_suspended':
            this._snackbar.open(this.translateService.instant('login_module.account_suspended'), null, { duration: 1500 });
            break;

          case 'account_locked':
            this._snackbar.open(this.translateService.instant('login_module.account_locked'), null, { duration: 1500 });
            break;
        }
        this.isSubmitting = false;

      }, () => {
        this.isSubmitting = false;
      });
    }
  }

  /**
   * Open phone verification popup then login if OTP correct
   */
  openPhoneVerification() {
    const dialogRef = this.dialog.open(CodeVerificationDialogComponent, {
      disableClose: true,
      data: {
        title: 'otp_verification.title',
        title_en: 'otp_verification.title_en',
        submit: 'otp_verification.submit',
      }
    })
    dialogRef.afterClosed().subscribe((result) => {
      if (result) {
        this.authService.sendOTP(this.form.getRawValue().username, result).subscribe(otpResult => {
          if (otpResult) {
            let user = otpResult.body;
            user.username = this.form.getRawValue().username;
            this.authService.setUser(user);
            this.landingService.getUser();
          } else {
            this.openPhoneVerification();
          }
        }, error => this.onLoginError(error))
      }
    })
  }

  /**
   * reset password
   * @param error : error returned from backend
   */
  onLoginError(error?: any) {
    this.error = 'خطأ';
    if (error == 'INVALID_CREDENTIALS')
      this.passwordError = true;
    this.form.controls.password.reset();
  }

  ngOnDestroy() {
    if (this.loginSubscription) { this.loginSubscription.unsubscribe() }
  }

  /**
   * Navigate to a another page
   * @param path string of the path we want to navigate to
   */
  public navigateTo(path: String) {
    this.router.navigate([path]);
  }

  openLegalCommitment() {
    const root = window.location.href.replace(this.router.url, '');
    this.dialog.open(InfoDialogComponent, {
      width: '80%',
      height: '80%',
      data: {
        texts: [{
          textString: 'login_module.legal_commitment1_en',
          isEnglish: true
        },
        {
          textString: 'login_module.legal_commitment2_en',
          isEnglish: true
        },
        {
          textString: 'login_module.legal_commitment3_en',
          isEnglish: true
        },
        {
          textString: `${root}/assets/docs/WFP_Privacy_Policy.pdf`,
          isEnglish: true,
          isRelativeLink: false,
          link: `${root}/assets/docs/WFP_Privacy_Policy.pdf`
        },
        {
          textString: 'login_module.legal_commitment1',
          isEnglish: false
        },
        {
          textString: 'login_module.legal_commitment2',
          isEnglish: false
        },
        {
          textString: 'login_module.legal_commitment3',
          isEnglish: false
        },
        {
          textString: `${root}/assets/docs/WFP_Privacy_Policy.pdf`,
          isEnglish: false,
          isRelativeLink: false,
          link: `${root}/assets/docs/WFP_Privacy_Policy.pdf`
        }],
        actions: [{
          label: 'login_module.close',
          isClose: true,
          returnedValue: false
        }]
      }
    });
  }
}
