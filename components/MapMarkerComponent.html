import { AfterViewInit, Component, OnInit, Input, Inject } from '@angular/core';
import * as L from 'leaflet';
import { ReadOnlyDetailsDialogModel } from '../read-only-details-dialog/read-only-details-dialog.model';
import { ReadOnlyDetailsDialogComponent } from '../read-only-details-dialog/read-only-details-dialog.component';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { MapMarkerComponentModel } from './map-marker.model';
import { RedFlagIcon } from 'src/app/home/households/assigned-households/surveys-list/surveys-list.config';

@Component({
  selector: 'app-map-marker',
  templateUrl: './map-marker.component.html',
  styleUrls: ['./map-marker.component.sass']
})
export class MapMarkerComponent implements OnInit, AfterViewInit {
  private map: any;

  @Input() coordinates: any
  @Input() distance: any

  constructor(
    public dialogRef: MatDialogRef<MapMarkerComponent>,
    @Inject(MAT_DIALOG_DATA) public data: MapMarkerComponentModel) {
    if (data?.coordinates?.length) {
      this.coordinates = data.coordinates
      this.distance = RedFlagIcon + ' ' + data.distance + ' KM'
    }
  }

  ngOnInit() { }

  ngAfterViewInit() {
    this.initializeMap();
    this.addMarkers();
  }

  private initializeMap() {
    const avgLat = (this.coordinates[0].lat + this.coordinates[1].lat) / 2;
    const avgLng = (this.coordinates[0].lng + this.coordinates[1].lng) / 2;

    const bounds = L.latLngBounds(this.coordinates[0], this.coordinates[1]);

    this.map = L.map('map');

    const zoomLevel = this.map.getBoundsZoom(bounds);

    this.map = this.map.setView([avgLat, avgLng], zoomLevel);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '&copy; OpenStreetMap contributors'
    }).addTo(this.map);
  }

  private addMarkers() {
    this.coordinates.forEach(coord => {
      L.marker(coord, {
        icon: L.icon({
          iconUrl: coord.mainLocation ? 'assets/icons/family.svg' : 'assets/icons/surveyors_primary.svg',
          iconSize: [30, 30]
        })
      }).addTo(this.map);
    });
  }

  buttonClicked(value){
    this.dialogRef.close(value);
  }

}
