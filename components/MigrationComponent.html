import { Component, OnInit, ViewChild, Inject, HostListener } from '@angular/core';
import {
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import {
  MatDialogRef,
  MAT_DIALOG_DATA,
  MatDialog,
} from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { UploadFieldComponent } from 'src/app/shared/components/upload-field/upload-field.component';
import { HomeService } from '../../home.service';
import { SnackbarService } from 'src/app/shared/services/snackbar.service';
import { HouseholdsService } from '../households.service';
import { map, tap } from 'rxjs/operators';
import { FileDetailsColumn } from './migration-dialog.model';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { ConfirmationDialogComponent } from 'src/app/shared/components/dialogs/confirmation-dialog/confirmation-dialog.component';
import { AuthService } from 'src/app/shared/services/auth.service';
import { Subscription } from 'rxjs';
import { NavigationStart, Router } from '@angular/router';

@Component({
  selector: 'app-migration-dialog',
  templateUrl: './migration-dialog.component.html',
  styleUrls: ['./migration-dialog.component.sass'],
})
export class MigrationComponent implements OnInit {
  @ViewChild(UploadFieldComponent) uploadElements: UploadFieldComponent;
  lookups: any = {};
  fileImported = new FormControl('');
  filesList: any[] = [];
  selectedFile: any;
  FileDetailsColumn = FileDetailsColumn;
  showTabel = false;
  loading = false;
  uploadInProgress = false;

  routerSubscription: Subscription;
  navigationConfirmed = false;
  dialogOpen = false; // Guard to prevent multiple dialogs

  migrateData = 'NPTP_MIGRATION_MGT.MIGRATE_DATA_TO_DAEM';
  uploadData = 'NPTP_MIGRATION_MGT.UPLOAD_DB_BACKUP';
  importData = 'NPTP_MIGRATION_MGT.DATABASE_IMPORT';
  listAllFiles = 'NPTP_MIGRATION_MGT.LIST_ALL_FILES';
  migrationAllowed: boolean;
  uploadDataAllowed: boolean;
  importDataAllowed: boolean;
  listAllFilesAllowed: boolean;
  constructor(
    private fb: FormBuilder,
    private householdService: HouseholdsService,
    private snacks: SnackbarService,
    public translateService: TranslateService,
    private dialog: MatDialog,
    private authService: AuthService,
    private router: Router
  ) {
    {
      this.routerSubscription = this.router.events.subscribe((event) => {
        if (event instanceof NavigationStart) {
          if (this.uploadInProgress && !this.navigationConfirmed && !this.dialogOpen) {
            this.dialogOpen = true; // Prevent further dialogs
  
            // Prevent initial navigation
            this.router.navigateByUrl(this.router.url); // Prevent navigation
      
            // Show confirmation dialog
            const dialogRefConfirmation = this.dialog.open(
              ConfirmationDialogComponent,
              {
                width: '500px',
                data: {
                  body: `${this.translateService.instant(
                    'home.households.migration_form.leave_page_warning'
                  )}`,
                  saveButtonText: 'confirmation_dialog.confirm',
                  discardButtonText: 'confirmation_dialog.reject',
                },
              }
            );
      
            dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
              if (confirmation) {
                this.navigationConfirmed = true; // Mark navigation as confirmed
                this.router.navigateByUrl(event.url); // Proceed with navigation
              } else {
                this.dialogOpen = false; // Allow future dialogs
                this.navigationConfirmed = false;
  
                // Reset the browser history to prevent navigation back
                history.pushState(null, '', this.router.url);
              }
            });
          }
        }
      });
    }
  }

  ngOnInit(): void {
    this.getLookups();
    this.getFiles();
    this.migrationAllowed = this.authService.isPageAccessible(this.migrateData);
    this.uploadDataAllowed = this.authService.isPageAccessible(this.uploadData);
    this.importDataAllowed = this.authService.isPageAccessible(this.importData);
    this.listAllFilesAllowed = this.authService.isPageAccessible(
      this.listAllFiles
    );
  }
  @HostListener('window:beforeunload', ['$event'])
  preventPageUnload(event: BeforeUnloadEvent): void {
    if (this.uploadInProgress) {
      event.preventDefault();
      event.returnValue = ''; // Trigger the default confirmation dialog
    }
  }

  onUpload(): void {
    if (this.uploadElements?.file.length > 0) {
      const dialogRefConfirmation = this.dialog.open(
        ConfirmationDialogComponent,
        {
          width: '400px',
          data: {
            body: `${this.translateService.instant(
              'confirmation_dialog.are_you_sure'
            )}`,
            saveButtonText: 'confirmation_dialog.confirm',
            discardButtonText: 'confirmation_dialog.reject',
          },
        }
      );
      dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
        if (confirmation) {
          this.uploadInProgress = true;
          // Add event listener for beforeunload to warn the user about unsaved progress
          window.addEventListener('beforeunload', this.preventPageUnload);
          this.householdService
            .uploadMigrationFile(this.uploadElements.file[0])
            .subscribe(
              (response) => {
                this.uploadInProgress = false;
                this.snacks.openSnackBar(
                  'Upload file was successful!',
                  'Close'
                );
                this.getLookups();
                this.getFiles();
                this.uploadElements.reset();
                // Remove the event listener once the upload is complete
                window.removeEventListener(
                  'beforeunload',
                  this.preventPageUnload
                );
              },
              (error) => {
                this.uploadInProgress = false;
                this.snacks.openSnackBar('Upload file failed!', 'Close');
                window.removeEventListener(
                  'beforeunload',
                  this.preventPageUnload
                );
              }
            );
        }
      });
    } else {
      this.uploadInProgress = false;
      this.snacks.openSnackBar(
        'Please select a file and an import option',
        'Close'
      );
    }
  }

  getLookups() {
    this.householdService
      .getPendingMigrationFiles()
      .pipe(
        map((response: any) => {
          return response?.body.map((file: any) => ({
            name: file.fileName,
            id: file.fileName,
            nameEn: file.fileName,
          }));
        })
      )
      .subscribe((fileNames: any) => {
        this.lookups = fileNames;
      });
  }

  getFiles(): void {
    this.loading = true;
    this.householdService.getAllMigrationFiles().subscribe(
      (res) => {
        if (res.body && res.body.length > 0) {
          this.filesList = res.body;
        }
        this.showTabel = true;
        this.loading = false;
      },
      (error) => {
        this.snacks.openSnackBar('Failed to fetch files!', 'Close');
        this.loading = false;
      }
    );
  }

  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.call_center.list.empty_table',
  };

  onStore(): void {
    if (this.fileImported.value) {
      const dialogRefConfirmation = this.dialog.open(
        ConfirmationDialogComponent,
        {
          width: '400px',
          data: {
            body: `${this.translateService.instant(
              'confirmation_dialog.are_you_sure'
            )}`,
            saveButtonText: 'confirmation_dialog.confirm',
            discardButtonText: 'confirmation_dialog.reject',
          },
        }
      );
      dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
        if (confirmation) {
          if (this.fileImported.value) {
            this.householdService
              .uploadMigrationFileForProcessing(this.fileImported.value)
              .subscribe(
                (response) => {
                  this.snacks.openSnackBar(
                    'Store file is successful!',
                    'Close'
                  );
                  this.getFiles();
                  this.fileImported.reset('');
                },
                (error) => {
                  this.snacks.openSnackBar('Store file failed!', 'Close');
                }
              );
          }
        }
      });
    } else {
      this.snacks.openSnackBar('Please select a file', 'Close');
    }
  }
  onMigrateData(): void {
    const dialogRefConfirmation = this.dialog.open(
      ConfirmationDialogComponent,
      {
        width: '400px',
        data: {
          body: `${this.translateService.instant(
            'confirmation_dialog.are_you_sure'
          )}`,
          saveButtonText: 'confirmation_dialog.confirm',
          discardButtonText: 'confirmation_dialog.reject',
        },
      }
    );
    dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
      if (confirmation) {
        this.householdService.migrateRestoredData().subscribe(
          (res) => {
            this.snacks.openSnackBar(
              'Started data migration, check process traking interface for progress',
              'Close'
            );
          },
          (error) => {
            this.snacks.openSnackBar(
              'Failed to start data migration process!',
              'Close'
            );
          }
        );
      }
    });
  }

  showLogs(event: Event, file: any): void {
    event.preventDefault(); // Prevent the default anchor tag behavior
    this.selectedFile = file;
  }

  actionClicked(event: any) {
    switch (event.name) {
      case 'log':
        const logMessage = event.data.log ? event.data.log : 'No logs found';
        this.dialog.open(WarningDialogComponent, {
          width: '400px',
          data: { arabic: logMessage, canClose: true },
        });
        break;
      default:
        break;
    }
  }
  
  ngOnDestroy() {
    // Clean up the subscription when the component is destroyed
    if (this.routerSubscription) {
      this.routerSubscription.unsubscribe();
    }
  }
}
