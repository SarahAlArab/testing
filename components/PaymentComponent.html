import { formatDate } from '@angular/common';
import { Component, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { DetailsDialogComponent } from 'src/app/shared/components/details-dialog/details-dialog.component';
import { UtilsService } from './../../shared/services/utils.service';
import { PaymentFilterDetails } from './payment.config';
import { PaymentService } from './payment.service';
import { saveAs } from 'file-saver';

@Component({
  selector: 'wfps-payment',
  templateUrl: './payment.component.html',
  styleUrls: ['./payment.component.sass']
})
export class PaymentComponent implements OnInit {
  cases: string[];
  casesDetails: any[];
  //casesDetailsFile: Blob;
  payments: any[];
  //paymentsFile: Blob;
  filter: any;
  generatedFiles: any = {};

  constructor(
    public utilsService: UtilsService,
    public _service: PaymentService,
    public dialog: MatDialog
  ) {
  }

  ngOnInit(): void {
  }

  public export() {
    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: "details-dialog",
      disableClose: true,
      width: '80%',
      data: {
        fields: PaymentFilterDetails,
        modalTitle: 'home.payment.filter'
      },
    });

    dialogRef.componentInstance.result.subscribe((result) => {
      if (result) {
        this.filter = result;
        this.filter.fromDate = formatDate(result.fromDate, 'yyyy-MM-dd HH:mm:ss', 'en');
        this.filter.toDate = formatDate(result.toDate, 'yyyy-MM-dd HH:mm:ss', 'en');
        this.filter.paymentMonth = formatDate(result.paymentMonth, 'yyyy-MM', 'en');
        //let today = new Date();
        //this.filter.paymentMonth = formatDate(today, 'yyyy-MM', 'en');
        this._service.getCases(`${this.filter.fromDate}`, `${this.filter.toDate}`).subscribe(s => {
          if (s.body) {
            this.cases = s.body;
            this._service.getCasesDetails(s.body).subscribe(details => {
              if (details.body && details.body.length) {
                this.casesDetails = details.body;
                let data = details.body?.map(m => {
                  m.houseHoldHeadNameEnFirstName = m.houseHoldHeadNameEn.firstName;
                  m.houseHoldHeadNameEnFatherName = m.houseHoldHeadNameEn.fatherName;
                  m.houseHoldHeadNameEnLastName = m.houseHoldHeadNameEn.lastName;
                  m.houseHoldHeadNameEn = `${m.houseHoldHeadNameEn.firstName} ${m.houseHoldHeadNameEn.fatherName} ${m.houseHoldHeadNameEn.lastName}`;
                  m.houseHoldHeadNameArFirstName = m.houseHoldHeadNameAr.firstName;
                  m.houseHoldHeadNameArFatherName = m.houseHoldHeadNameAr.fatherName;
                  m.houseHoldHeadNameArLastName = m.houseHoldHeadNameAr.lastName;
                  m.houseHoldHeadNameAr = `${m.houseHoldHeadNameAr.firstName} ${m.houseHoldHeadNameAr.fatherName} ${m.houseHoldHeadNameAr.lastName}`;
                  return m;
                })
                this.generatedFiles.casesDetailsFile = { file: this.getFile(details.body), type: 'cases', generated: false };
              }
            });
            this._service.getPaymentDetails(this.filter.paymentMonth, s.body).subscribe(payment => {
              if (payment.body && payment.body.length) {
                this.payments = payment.body;
                this.generatedFiles.paymentsFile = { file: this.getFile(payment.body), type: 'payments', generated: false };
              }
            });

          }
          dialogRef.close();

        });

      }
    });
  };


  getFile(data: any) {
    const replacer = (key, value) => value === null ? '' : value; // specify how you want to handle null values here
    const header = Object.keys(data[0]);
    let csv = data.map(row => header.map(fieldName => JSON.stringify(row[fieldName], replacer)).join(','));
    csv.unshift(header.join(','));
    let csvArray = csv.join('\r\n');

    return new Blob(["\ufeff", csvArray], { type: 'text/csv' })

  }

  generateFile(file) {
    if (this.generatedFiles[file].generated) {
      return;
    }
    this.generatedFiles[file].generated = true;
    saveAs(this.generatedFiles[file].file, `${this.generatedFiles[file].type}.csv`);
  }

  clear() {
    this.cases = null;
    this.casesDetails = null;
    this.generatedFiles = {};
    this.payments = null;
    this.filter = null;
  }
}
