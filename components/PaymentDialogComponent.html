import { HttpClient } from '@angular/common/http';
import { Component, EventEmitter, Inject, Injector, OnInit, Output, QueryList, SimpleChanges, ViewChild, ViewChildren } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { HttpBaseClass } from '../../bases/http-base.class';
import { Microservice } from '../../enums/microservice.enum';
import { FrameworkHelper } from '../../helpers/framework.helper';
import { DocumentModel } from '../../models/document.model';
import { FieldType } from '../../models/fields-config.model';
import { HttpRequestOptions } from '../../models/http-request-options.model';
import { CustomFormComponent } from '../custom-form/custom-form.component';
import { CustomFormActionModel } from '../custom-form/custom-form.model';
import { DetailsDialogModel } from '../details-dialog/details-dialog.model';
import { UploadFieldComponent } from '../upload-field/upload-field.component';


@Component({
  selector: 'app-payment-dialog',
  templateUrl: './payment-dialog.component.html',
  styleUrls: ['./payment-dialog.component.sass'],
})
export class PaymentDialogComponent extends HttpBaseClass implements OnInit  {
  @ViewChild(CustomFormComponent) customFormComponent?: CustomFormComponent;
  
  @ViewChild(UploadFieldComponent) uploadElements: UploadFieldComponent;
  close = false;
  error: string;
  canUpload: true;
  fieldTypeEnum = FieldType;
  frameworkHelper = FrameworkHelper;
  @Output() result = new EventEmitter<any>();
  @Output() fieldChange = new EventEmitter<any>();
  submitAction: CustomFormActionModel = new CustomFormActionModel('home.households.saveAndContinue', false, false);
 
  public file: any[] = [];
  public documentsList: DocumentModel[] = [];
    
  constructor(
    public http: HttpClient,
    public injector: Injector,
    public dialogRef: MatDialogRef<PaymentDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: DetailsDialogModel
  ) {
    super(http, injector);

  }

  ngOnInit(): void {
   
  }

  public apply(event) {
    if (event) {
      this.buttonClicked(event.formVal);
    }
    else {
      this.closeDialog(true);
    }
  }

  public buttonClicked(formValToSave) {
    let formVal;
    if (formValToSave) {
      formVal = Object.assign(this.data.data || {}, formValToSave);
      if (this.uploadElements.file){
        formVal.file = this.uploadElements.file[0];
      }
 
      //this.result.emit(formVal);
    }
    this.dialogRef.close(formVal);
  }

  public closeDialog(val: boolean) {
    if (this.customFormComponent.form.pristine) {
      this.buttonClicked(false);
    } else this.close = val;
  }

  changeField(event) {
    this.fieldChange.emit(event);
  }

  downloadtemplate(){
    const reqOptions = new HttpRequestOptions(
      undefined,
      undefined,
      undefined,
      'response',
      undefined,
      'blob'
    );
    return this.get(`nptp-uploadtemplate`,reqOptions,undefined,  this.settingsService.settings.prioritizationUrl);
  
    }
  
    exportNptpExcel() {
  
      this.downloadtemplate().subscribe(data => {
  
          const link = document.createElement('a');
  
          const blob = new Blob([data.body], {type: 'application/octet-stream'});
  
          link.setAttribute('href', window.URL.createObjectURL(blob));
         
          link.setAttribute('download', data.headers.get('Content-Disposition').split('=')[1]);
  
          link.style.visibility = 'hidden';
  
          document.body.appendChild(link);
  
          link.click();
  
          document.body.removeChild(link);
  
      })
  
  }

  
}
