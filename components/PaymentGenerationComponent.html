  ngOnInit(): void {
    this.canReviewGeneratedPayment = this.authService.isPageAccessible(
      `${UserModulesEnum.PAYMENT_GENERATE.module}.AID_PAYMENTS_GENERATE`
    );

    this.canUploadExtraPayments  = this.authService.isPageAccessible(
      `${UserModulesEnum.PAYMENT_GENERATE.module}.UPLOAD_EXTRA_PAYMENTS`
    );

    this.showProcessTrackingHistory = this.authService.isPageAccessible(
      `${UserModulesEnum.PROCESS_TRACKING_HISTORY.module}`
    );

    this.canAccessReportGenerater= this.authService.isPageAccessible('VIEW_REPORTS_LIST');
    this.canUnsuspendEssn =  this.authService.isPageAccessible('UNSUSPEND_ESSN');
    this.canEnrolPayCanceled =  this.authService.isPageAccessible('ENROLL_PAY_CANCELED');
    this.canModifyPaymentPlans = this.authService.isPageAccessible('MODIFY_AID_PAYMENT_PLANS');

    let FilesTypeArr = ['FilesType']
    this.utilsService.getLookupsService(FilesTypeArr).subscribe(s => {
      if (s.body) {
        FilesTypeArr.forEach(k => {
          s.body[k].forEach(l => {
            if(l.type == "NPTP_DUPLICATE"){
              this.educationalId = l.id;
            }
          });
        });
      }
    });
    this.getLookups();
  }

  openUploadFileDialog(){
    const dialogRef = this.dialog.open(UploadFileDialogComponent, {
      panelClass: 'details-dialog',
      disableClose: true,
      width: '80%',
      maxWidth: '650px'
    });


    dialogRef.afterClosed().subscribe((result) => {
      if (result) {
        this.openCancelPaymentInfromativeWarningPopup(result);
      }
    });
  }



  openCancelPaymentInfromativeWarningPopup(file : any){
    const dialogRef = this.dialog.open(WarningDialogComponent, {
      width: '400px',
      data: {
        arabic: 'home.households.payments.cancel_payment_warning',
        canClose: true,
        action : 'confirmation_dialog.yes'
      },
    });


    dialogRef.afterClosed().subscribe((result) => {
      if (result) {
        var extension = file.name.split('.')[1].toLowerCase();
        if(extension == 'csv' || extension == 'xls' || extension == 'xlsx'){
          this._service.uploadCancelUploadedPayments(file)
          .subscribe(
            res => {
              if(res && res.body && res.body.errors?.length > 0){
                //display partial error happened with limitation for 10 errors only
                const errors = res.body.errors.slice(0, res.body.errors.length > 10 ? 10 : res.body.errors.length);

                this.dialog.open(MultiLineTextDialogComponent, {
                  width: '800',
                  maxHeight: '95%',
                  data: {
                    title:'',
                    arabic:  'home.households.payments.cancel_payment_partial_error',
                    listOfTexts: errors,
                    canClose: true,
                  },
                });
              }else{
                this.snacks.openSnackBar(
                  `${this.translateService.instant(
                    'home.households.payments.payment_canceled_successfully'
                  )}`
                );
              }
            },
            error => {
              this.handleConflictError('ERROR_UPLOAD');
            }
          );
        }
        else{
          this.dialog.open(WarningDialogComponent, {
            width: '400px',
            data: {
              arabic: 'home.households.unsupported_file_format',
              canClose: true,
            },
          });
        }

      }
    });
  }

  essnUnsuspedUploadFileDialog(){
    const dialogRef = this.dialog.open(UploadFileDialogComponent, {
      panelClass: 'details-dialog',
      disableClose: true,
      width: '80%',
      maxWidth: '650px'
    });


    dialogRef.afterClosed().subscribe((result) => {
      if (result) {
        const dialogRefConfirmation = this.dialog.open(ConfirmationDialogComponent, {

             body: `${this.translateService.instant(

        dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
          if (confirmation) {
            this.ProcessEssnToUnsuspend(result);
          }
        })       
      }
    });
  }

  ProcessEssnToUnsuspend(file : any){
      var extension = file.name.split('.')[1].toLowerCase();
      if(extension == 'csv' || extension == 'xls' || extension == 'xlsx'){
        this._service.essnUploadToUnsuspend(file)
        .subscribe(
          _ => {
            this.snacks.openSnackBar(
              `${this.translateService.instant(
                'home.households.essn_upload_done'
              )}`
            );
          },
          error => {
            this.handleConflictError('ERROR_UPLOAD');
          }
        );
      }
      else{
        this.dialog.open(WarningDialogComponent, {
          width: '400px',
          data: {
            arabic: 'home.households.unsupported_file_format',
            canClose: true,
          },
        });
      }
  }


  updateListData(data) {
  }

  checkInProgressProcess(ev){
    //Call API to check if process in progress or no
    //checking if PROCESS_TRACKING_HISTORY permisission given to go with this flow else directly to the switch cases
    if(!this.showProcessTrackingHistory){
      this.callingProcesses(ev);
    }
    else{
     this._service
    .getByStatus('IN_PROGRESS')
    .subscribe((res) => {
      if (res.body) {
        const element = res.body;
        element.runTime = res.body.runDateTime;
        if(element.type!= null && element.type!= ''){
          this.dialog.open(InfoDialogComponent, {
            width: '80%',
            height: '20%',
            data: {
              texts: [{
                textString: 'home.households.process_in_progress'+element.type,
                isEnglish: true
              }],
              actions: [{
                label: 'login_module.close',
                isClose: true,
                returnedValue: false
              }]
            }
          });
        }
      }
        else{
          this.callingProcesses(ev)
        }
    });
  }
  }

  callingProcesses(ev){
    switch(ev){
      case 'checkInternalDuplication':
        this.checkInternalDuplication();
        break;
      case 'checkNPTPDuplication':
        this.checkNPTPDuplication();
        break;
      case 'downloadNptpFile':
        this.downloadNptpFile();
        break;
      case 'generatePayment':
        this.generatePayment();
        break;
      case 'generatePaymentBatch':
        this.generatePaymentBatch();
        break;
      case 'getEligibleApplications':
        this.getEligibleApplications();
        break;
      case 'getPayCanceledApplications':
        this.getPayCanceledApplications();
        break;
      case 'batchAuthorize':
        this.batchAuthorize();
        break;
      case 'reviewOldExternalDuplicates':
        this.reviewOldExternalDuplicates();
        break;
      case 'reviewOldInternalDuplicates':
        this.reviewOldInternalDuplicates();
        break;
      case 'reviewPayment':
        this.reviewPayment();
        break;
      case 'uploadExtraPaymentFile':
        this.uploadExtraPaymentFile();
        break;
      default:
        break;
    }
  }
  checkInternalDuplication() {
    this.paginationIndex = this._service.getPaginationIndexValue();
    this.withResett =  this.paginationIndex && this.paginationIndex.length != 0 ? false : true;
   this._service
      .getLatestProcessInformation(ProcessType.InternalDuplication)
      .subscribe((res) => {
        if (res.body) {
          const element = res.body;
          element.runTime = res.body.runDateTime;

          const dialogRef = this.dialog.open(DetailsDialogComponent, {
            panelClass: 'details-dialog',
            disableClose: true,
            width: '80%',
            maxWidth: '650px',
            data: {
              data: element,
              fields: this.latestProcessDetails,
              detailsTitle: this.translateService.instant(
                'home.households.latest_nptp_duplication'
              ),
              modalTitle: `${this.translateService.instant(
                'home.households.nptp_duplication'
              )}`
            },
          });

          dialogRef.afterClosed().subscribe((result) => {
            if (result) {
              const dialogRefConfirmation = this.dialog.open(ConfirmationDialogComponent, {
                width: '400px',
                data: {
                  body: `${this.translateService.instant(
                    'confirmation_dialog.are_you_sure'
                  )}`,
