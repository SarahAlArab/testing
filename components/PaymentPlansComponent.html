import {
  Component,
  Injector,
  Input,
  OnInit,
  ViewChild,
  Inject,
} from '@angular/core';
import { MatTableDataSource } from '@angular/material/table';
import { TranslateService } from '@ngx-translate/core';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { PaginatorConfig } from 'src/app/shared/models/paginator-config.model';
// import { LookupsService } from '../lookups.service';
import { configurationsDef, lookupsColumnDef } from './payment-plans.config';
import { CustomFormComponent } from 'src/app/shared/components/custom-form/custom-form.component';
import { MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { DetailsDialogModel } from 'src/app/shared/components/details-dialog/details-dialog.model';
import { HouseholdsService } from '../../households.service';
import { ConfirmationDialogComponent } from 'src/app/shared/components/dialogs/confirmation-dialog/confirmation-dialog.component';
import { DetailsDialogComponent } from 'src/app/shared/components/details-dialog/details-dialog.component';
import { ReadOnlyDetailsDialogComponent } from 'src/app/shared/components/read-only-details-dialog/read-only-details-dialog.component';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { LookupsEnum } from 'src/app/shared/models/lookups.enum';
import { PaymentDetailsDialogComponent } from '../payment-details-dialog/payment-details-dialog.component';

@Component({
  selector: 'app-payment-plans',
  templateUrl: './payment-plans.component.html',
  styleUrls: ['./payment-plans.component.sass'],
})
export class PaymentPlansComponent
  extends DataTableBaseClass<any>
  implements OnInit
{
  @ViewChild(CustomFormComponent) customFormComponent?: CustomFormComponent;
  displayedColumns = [
    'id',
    'name_in_english',
    'name_in_arabic',
    'edit',
    'deprecated',
  ];

  methodEndpoint: string;

  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.call_center.list.empty_table',
  };
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);

  dataTableColumnsDefinition = lookupsColumnDef;
  dataSource = new MatTableDataSource();
  configurationsDef = configurationsDef;
  lookups: any;
  constructor(
    private _service: HouseholdsService,
    private translateService: TranslateService,
    public injector: Injector,
    private dialog: MatDialog,
    private utilsService: UtilsService
  ) {
    super(injector);
  }

  ngOnInit(): void {
    this.getLookups();
    this.getListing(null, true);
  }

  getLookups() {
    this.lookups = {};
    this.utilsService
      .getLookupsService([LookupsEnum.EDUCATION_REGISTRATION_YEAR])
      .subscribe((s) => {
        if (s.body) {
          this.lookups[LookupsEnum.EDUCATION_REGISTRATION_YEAR] =
            s.body[LookupsEnum.EDUCATION_REGISTRATION_YEAR];
        }
      });
  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getListing(pageChangeEvent, false);
  }

  viewConfigurations(element) {
    const dialogRef = this.dialog.open(ReadOnlyDetailsDialogComponent, {
      panelClass: 'details-dialog',
      disableClose: false,
      width: '80%',
      data: {
        report: element.configuration,
        config: this.configurationsDef,
        title: 'home.lookups.lookup_module.aid_payment.plan_configuration',
      },
    });
  }

  viewConfigurationsWithEdit(element) {
    const dialogRef = this.dialog.open(PaymentDetailsDialogComponent, {
      data: {
        element: element,
        report: element.configuration,
        config: this.configurationsDef,
        id: element.id,
      },
    });
  }

  clickToggle(ev) {
    switch (ev.name) {
      case 'enabled':
        this.enabledToggle(ev.data, ev.value);
        break;
      default:
        break;
    }
  }

  private enabledToggle(object, value) {
    object.enabled = value;
    const dialogRefConfirmation = this.dialog.open(
      ConfirmationDialogComponent,
      {
        width: '400px',
        data: {
          body: `${this.translateService.instant(
            'confirmation_dialog.are_you_sure'
          )}`,
          saveButtonText: 'confirmation_dialog.confirm',
          discardButtonText: 'confirmation_dialog.reject',
        },
      }
    );

    dialogRefConfirmation.afterClosed().subscribe((res) => {
      if (res) {
        this._service.putLookupOption(object.id, object).subscribe((res) => {
          let index = this.dataTableData.visibleRows.findIndex(
            (x) => x.id === object.id
          );
          index >= 0
            ? (this.dataTableData.visibleRows[index].enabled = value)
            : null;
          this.snacks.openSnackBar('snackbar.edit_success');
        });
      } else {
        object.enabled = !value;
      }
    });
  }

  getListing(paginationObj: PageChangeEvent, start: boolean) {
    this._service.getOptionsOfList(paginationObj, start).subscribe(
      (data) => {
        if (data) {
          let dataTable = data.body;

          data.body.content.map((e) => {
            if (e.educationRegistrationYear && e.educationRegistrationYear.id) {
              e.educationRegistrationYearName = this.lookups[
                LookupsEnum.EDUCATION_REGISTRATION_YEAR
              ].find(
                (eduYear) => eduYear.id == e.educationRegistrationYear.id
              ).nameEn;
            } else {
              e.educationRegistrationYearName = '-';
            }
            return e;
          });
          this.setDataTableData(dataTable, start);
        }
      },
      (err) => {}
    );
  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'configuration':
        if (ev.data.planType !== 'Edu_Transportation') {
          this.viewConfigurations(ev.data);
        } else this.viewConfigurationsWithEdit(ev.data);
        break;
      default:
        break;
    }
  }
}
