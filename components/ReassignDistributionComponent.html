import { Component, OnInit, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { CustomFormComponent } from 'src/app/shared/components/custom-form/custom-form.component';
import { CustomFormActionModel, CustomFormModel } from 'src/app/shared/components/custom-form/custom-form.model';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { LookupsEnum } from 'src/app/shared/models/lookups.enum';
import { HomeService } from '../../home.service';
import { DistributionService } from '../distribution.service';
import { ReassignmentDetails } from './reassign.config';

@Component({
  selector: 'wfps-reassign-distribution',
  templateUrl: './reassign-distribution.component.html',
  styleUrls: ['./reassign-distribution.component.sass']
})
export class ReassignDistributionComponent implements OnInit {
  @ViewChild(CustomFormComponent) customFormComponent?: CustomFormComponent;

  reassignDetails = ReassignmentDetails;
  reassignFormDetails: CustomFormModel = new CustomFormModel(this.reassignDetails);
  submitReassign: CustomFormActionModel = new CustomFormActionModel('home.distribution.reassign', false, false, true, true);
  cancelReassign: CustomFormActionModel = new CustomFormActionModel('', true);
  lookups: any = {};

  constructor(
    public translateService: TranslateService,
    public _service: DistributionService,
    public _homeService: HomeService,
    private dialog: MatDialog) { }

  ngOnInit(): void {
    this.getLookups();
  }

  public apply(event) {
    if (event.apply) {
      this._service.reassign(event.formVal).subscribe(res => {
        if (res && res.body) {
          let text = '';
          if (res.body.applicationCount > 0) {
            text =
              `${this.translateService.instant("home.distribution.done3")} ${this.lookups[LookupsEnum.SupervisorsByEntity].find(f => f.id === res.body.supervisorUser?.id)?.name}`;
          }
          else {
            text = this.translateService.instant("home.distribution.done4");
          }
          const dialogRef = this.dialog.open(WarningDialogComponent, {
            width: '400px',
            data: { arabic: text, canClose: true }
          });
          dialogRef.afterClosed().subscribe(s => {
            this.customFormComponent.resetForm();
            this.lookups[LookupsEnum.SupervisorsByEntity] = [];
          })
        }
      });
    }
  }

  getLookups() {
    this._homeService.getAllSupervisors().subscribe(s => {
      this.lookups[LookupsEnum.Superviors] = s.body;
      this.reassignFormDetails.lookups = this.lookups;
    })
  }

  getToSupervisorLookups(control) {
    if (control.field === 'fromSupervisorUser') {
      if (control.value) {
        let surveyEntityId = this.lookups[LookupsEnum.Superviors].find(f => f.id === control.value)?.surveyEntity.id
        this._homeService.getActiveSupervisors().subscribe(s => {
          if (s.body) {
            this.lookups[LookupsEnum.SupervisorsByEntity] = s.body.filter(f => f.surveyEntity.id === surveyEntityId && f.id !== control.value);
          }
        });
      } else {
        this.lookups[LookupsEnum.SupervisorsByEntity] = [];
      }
    }
  };
}
