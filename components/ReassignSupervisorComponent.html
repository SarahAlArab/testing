import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { TranslateService } from '@ngx-translate/core';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { ConfirmationDialogComponent } from 'src/app/shared/components/dialogs/confirmation-dialog/confirmation-dialog.component';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { DropdownOptionModel } from 'src/app/shared/models/dropdown-option.model';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { PaginatorConfig } from 'src/app/shared/models/paginator-config.model';
import { UserModulesEnum, UserRolesActionsEnum } from 'src/app/shared/models/user-role.enum';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { HomeService } from '../../home.service';
import { DistributionService } from '../distribution.service';
import { Decision, FamiliesAddresChangeColumnDef } from './reassign-supervisor.config';

@Component({
  selector: 'wfps-reassign-supervisor',
  templateUrl: './reassign-supervisor.component.html',
  styleUrls: ['./reassign-supervisor.component.sass'],
})
export class ReassignSupervisorComponent extends DataTableBaseClass<any> implements OnInit {
  //table rows and cols
 
  submitChangeRequest ='SUBMIT_APP_REASSIGNATION_RESPONSE';
  dataTableColumnsDefinition = FamiliesAddresChangeColumnDef;
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  lookups: { [key: string]: DropdownOptionModel[] } = {};
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.households.empty_families_address_change',
  };

  constructor(
    public _homeService: HomeService,
    private distributionService: DistributionService,
    private utilSerivce: UtilsService,
    private snackbar: MatSnackBar,
    public injector: Injector,
    private translateService: TranslateService,
    private dialog: MatDialog,
  ) {
    super(injector);
  }

  ngOnInit(): void {
    super.ngOnInit();
    this.getLookups();
    this.dataTableColumnsDefinition.find(x => x.property === 'acceptChange'  ).visible = this.isSectionAccessible(this.submitChangeRequest);
    this.dataTableColumnsDefinition.find(x => x.property === 'declineChange' ).visible = this.isSectionAccessible(this.submitChangeRequest);
   
    
  }

  isSectionAccessible(permission: string) {
    return this.authService.isPageAccessible(permission)
  }

  getLookups() {
    this.utilSerivce.getLookupsService(['Kadaa','Town','Locality']).subscribe(res => {
      this.lookups.Kadaa = res.body.Kadaa;
      this.lookups.Town = res.body.Town;
      this.lookups.Locality = res.body.Locality;
      this.getListing(null, true);
      
    })
  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getListing(pageChangeEvent, false);
  }

  getListing(paginationObj: PageChangeEvent, start: boolean) {
    this.distributionService
      .getFamiliesWithAddressChange(paginationObj, start)
      .subscribe((data) => {
        if (data && data.body) {
          this.updateListData(data.body.content);
          this.setDataTableData(data.body, start);
        }
      }, 
      (err) => { });
  }

  updateListData(data) {
    data.forEach(e => {
    this.updateHHHDisplay(e);
    this.updateSurveyorDisplay(e);
     this.updateNewAddress(e);
     this.updateCurrentAddress(e);
    })
  }

  updateHHHDisplay(e) {
    if (e.headOfHouseHoldFirstName && e.headOfHouseHoldLastName) {
      e.headOfHousehold = `${e.headOfHouseHoldFirstName}  ${e.headOfHouseHoldLastName}`;
    }
  }

  updateSurveyorDisplay(e) {
    if (e.surveyorFirstName && e.surveyorLastName) {
      e.surveyor = `${e.surveyorFirstName} ${e.surveyorLastName}`;
    }
  }

  updateNewAddress(e) {
    if (this.lookups){
      e.newKadaa= this.lookups.Kadaa.find(x => x.id === e.newAddress.kadaa.id).name;
      e.newTown= this.lookups.Town.find(x => x.id === e.newAddress.town.id).name;
      e.newLocality= this.lookups.Locality.find(x => x.id === e.newAddress.locality.id).name;
      e.newAddress= `${e.newKadaa}-${e.newTown}-${e.newLocality}`;
    }
   
  }
  
  updateCurrentAddress(e) {
    if (this.lookups){
      e.currentKadaa= this.lookups.Kadaa.find(x => x.id === e.currentAddress.kadaa.id).name;
      e.currentTown= this.lookups.Town.find(x => x.id === e.currentAddress.town.id).name;
      e.currentLocality= this.lookups.Locality.find(x => x.id === e.currentAddress.locality.id).name;
      e.currentAddress= `${e.currentKadaa}-${e.currentTown}-${e.currentLocality}`;
    }
  
  }
  
  actionClicked(ev) {
    switch (ev.name) {
      
      case 'acceptChange':
        this.decisionRequest(ev.data.id,'confirmation_dialog.change_supervisor',true);
        break;

      case 'declineChange':
        this.decisionRequest(ev.data.id,'confirmation_dialog.no_change_supervisor',false );
        break;
       
      default:
        break;
    }
  }

  decisionRequest(id: number, text: string, decision: boolean){
    const dialogRef = this.dialog.open(ConfirmationDialogComponent, {
      width: '400px',
      data: {
        body:  text,
        saveButtonText: 'confirmation_dialog.confirm',
        discardButtonText: 'confirmation_dialog.reject'
       }
    })
    dialogRef.afterClosed().subscribe((result) => {
      if (result){

        this.distributionService.submitAddressChangeDecision(id,decision).subscribe(
          (res) => {
            this.snackbar.open(this.translateService.instant('snackbar.submit_success'), "X", {
              duration: 10000,
              panelClass: ['successful-snackbar'],
            });
            this.getListing(this.paginatorConfig, true)
          },
          (err) => {
            this.snackbar.open(this.translateService.instant('snackbar.edit_error'), "X", {
              duration: 10000,
              panelClass: ['warn-snackbar'],
            });
          }
        )
      }
    })
  }

}
