import { Component, OnInit, ViewChild } from '@angular/core';
import { FormControl, FormGroup, FormGroupDirective, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { ActivatedRoute, Router } from '@angular/router';
import { TranslateService } from '@ngx-translate/core';
import { ResetPasswordService } from './reset-password.service';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { AuthService } from 'src/app/shared/services/auth.service';
import { SpinnerService } from 'src/app/shared/services/spinner.service';
import { matchingFields } from 'src/app/shared/validators/customLengthValidator.validator';
import { containsSpaces } from 'src/app/shared/validators/spaceValidator';

@Component({
  selector: 'app-reset-password',
  templateUrl: './reset-password.component.html',
  styleUrls: ['./reset-password.component.sass']
})
export class ResetPasswordComponent implements OnInit {
  @ViewChild(FormGroupDirective) formGroupDirective: FormGroupDirective;
  error: string;
  resetForm: FormGroup;
  showAllInfo: Boolean = false;
  isFirst: boolean = false;


  constructor(public dialog: MatDialog,
    private resetPasswordService: ResetPasswordService,
    private route: ActivatedRoute,
    private router: Router,
    private _snackbar: MatSnackBar,
    private translateService: TranslateService,
    private spinnerService: SpinnerService,
    private authService: AuthService) {
    this.isFirst = this.route.snapshot.data.isFirst;
  }

  ngOnInit(): void {
    this.resetForm = new FormGroup({
      username: new FormControl(null, Validators.required),
      password: new FormControl(null),
      confirmPassword: new FormControl(null),
      code: new FormControl(null)
    });

    this.resetForm.controls.password.valueChanges.subscribe(() => {
      setTimeout(() => {
        this.onPasswordChange();
      });
    });
  }

  public onPasswordChange() {
    if (this.resetForm.controls.confirmPassword.dirty) {
      this.resetForm.controls.confirmPassword.updateValueAndValidity();
    }
  }

  submit() {
    if (this.resetForm.valid) {
      if (this.showAllInfo) {
        this.reset();
      }
      else {
        this.sendCode();
      }
    }
  }

  /**
   * Send username to backend to send OTP then show remaining fields (passowrd and OTP) if there are no errors
   */
  sendCode() {
    if (this.resetForm.valid) {
      const form = this.resetForm.getRawValue();
      this.spinnerService.forceShowLoader();
      delete form.confirmPassword;
      delete form.password;
      delete form.code;
      this.resetPasswordService.sendCode(form).subscribe(x => {
        this.showAllInfo = true;
        this.formGroupDirective.resetForm();
        this.resetForm.controls.username.setValue(form.username);
        this.resetForm.controls.username.disable();
        this.resetForm.controls.code.setValidators(Validators.required);
        this.resetForm.controls.password.setValidators([Validators.required, Validators.pattern('(?=.*(\\d|[\u0660-\u0669]))(?=.*[a-z]|[A-Z]|[\u0621-\u064A])(?=.*[^A-Za-z0-9\u0621-\u064A\u0660-\u0669]).{12,50}'), containsSpaces(false)]);
        this.resetForm.controls.confirmPassword.setValidators([Validators.required, matchingFields('password')]);
        this.resetForm.controls.code.updateValueAndValidity();
        this.resetForm.controls.password.updateValueAndValidity();
        this.resetForm.controls.confirmPassword.updateValueAndValidity();
      }, err => {
        this.spinnerService.forceHideLoader();
        this.handleResetPassError(err);
      });
    }
  }

  /**
   * check that the 2 new passwords are equal  then reset password and show thank you dialog if there are no errors
   */
  reset() {
    if (this.resetForm.valid) {
      const form = this.resetForm.getRawValue();
      if (form.password !== form.confirmPassword) {
        this.dialog.open(WarningDialogComponent, {
          width: '400px',
          data: { arabic: 'reset_module.no_matching_passwords', english: 'reset_module.no_matching_passwords_en', canClose: true }
        });
        return;
      }
      delete form.confirmPassword;
      this.resetPasswordService.resetPassword(form).subscribe(x => {
        this.openThankYouDialog();
      }, err => {
        this.handleResetPassError(err)
      });
    }
  }

  handleResetPassError(err) {
    switch (err.code) {
     
      case 'username_not_found':
        this._snackbar.open(this.translateService.instant('login_module.username_not_found'), null, { duration: 1500 });
        break;

      case 'invalid_password_complexity':
        this._snackbar.open(this.translateService.instant('login_module.invalid_password_complexity'), null, { duration: 1500 });
        break;

      case 'account_locked':
        this._snackbar.open(this.translateService.instant('login_module.account_locked'), null, { duration: 1500 });
        break;

      case 'account_suspended':
        this._snackbar.open(this.translateService.instant('login_module.account_suspended'), null, { duration: 1500 });
        break;

      case 'invalid_auth_request':
        this._snackbar.open(this.translateService.instant('login_module.invalid_auth_request'), null, { duration: 1500 });
        break;
    }
  }

  /**
   * open thank you popup and navifgate to login when closed
   */
  private openThankYouDialog() {
    const dialogRef = this.dialog.open(WarningDialogComponent, {
      width: '400px',
      data: { arabic: 'reset_module.thank_you', english: 'reset_module.thank_you_en', canClose: true }
    });

    dialogRef.afterClosed().subscribe(result => {
      this.authService.onLogout();
    });
  }

  public goBack() {
    this.router.navigate(['./login'], { relativeTo: this.route });
  }

}
