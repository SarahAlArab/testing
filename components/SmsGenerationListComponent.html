import { Component, OnInit } from '@angular/core';

import { AuthService } from 'src/app/shared/services/auth.service';
import { ActivatedRoute,Router } from '@angular/router';
import { UserRolesEnum } from 'src/app/shared/models/user-role.enum';

@Component({
  selector: 'wfps-sms-generation-list',
  templateUrl: './sms-generation-list.component.html',
  styleUrls: ['./sms-generation-list.component.sass']
})
export class SmsGenerationListComponent implements OnInit {

  constructor(public dialog: MatDialog,
    private router: Router,
    private route: ActivatedRoute,
    private authService: AuthService) { }


ngOnInit(): void {
  }

isAllowed(permission: string) {
  return this.authService.isPageAccessible(permission);
}

isMophUser() {
  return (this.authService.getUserRole() == UserRolesEnum.MOPH_USER)
}


navigateTo(route) {
  this.router.navigate([route], { relativeTo: this.route });
}








}
