import { Component, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { TranslateService } from '@ngx-translate/core';
import { DetailsDialogComponent } from 'src/app/shared/components/details-dialog/details-dialog.component';
import { SnackbarService } from 'src/app/shared/services/snackbar.service';
import { LatestProcessDetails, NptpLatestProcessDetails, PaymentDetails } from '../../payment/payment.config';
import { HouseholdsService } from '../households.service';
import { Subscription } from 'rxjs/internal/Subscription';
import { SpinnerService } from 'src/app/shared/services/spinner.service';
import { AuthService } from 'src/app/shared/services/auth.service';
import { UserModulesEnum } from 'src/app/shared/models/user-role.enum';
import { DownloadEducationPersons } from 'src/app/home/households/landing-households/landing-hosueholds.config';
import { LookupsEnum } from 'src/app/shared/models/lookups.enum';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { CustomFormModel } from 'src/app/shared/components/custom-form/custom-form.model';
import { ActivatedRoute, Router } from '@angular/router';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { UploadFileDialogComponent } from 'src/app/shared/components/upload-file-dialog/upload-file-dialog.component';
import { ConfirmationDialogComponent } from 'src/app/shared/components/dialogs/confirmation-dialog/confirmation-dialog.component';
@Component({
  selector: 'wfps-student-data',
  templateUrl: './student-data.component.html',
  styleUrls: ['./student-data.component.sass']
})
export class StudentDataComponent implements OnInit {  latestProcessDetails = LatestProcessDetails;
  paymentDetails = PaymentDetails;
  nptpLatestProcessDetails = NptpLatestProcessDetails;
  canReviewGeneratedPayment: boolean
  canUploadExtraPayments: boolean

  lookups: any = {};
  details = DownloadEducationPersons;
  formDetails: CustomFormModel = new CustomFormModel(this.details);
  downloadEducationPersonsDetails = DownloadEducationPersons;
  reportDownloadWithNames = 'DOWNLOAD_EDUCATIONAL_INFO.DOWNLOAD_EDU_INFO_WITH_NAMES';
  reportDownloadWithoutNames = 'DOWNLOAD_EDUCATIONAL_INFO.DOWNLOAD_EDU_INFO_ANONYMOUS';
  educationalId = null;
  reportEducationalUpload ='UPLOAD_EDUCATIONAL_STATUS_UPDATE';



  isDownloadDataAllowed: boolean;
  canAccessReportGenerater: boolean;
  generateSubscription: Subscription;
  isUploadEducationalDataAllowed: boolean;
  constructor(public dialog: MatDialog,
    private router: Router,
    private route: ActivatedRoute,
    public translateService: TranslateService,
    public _service: HouseholdsService,
    public snacks: SnackbarService,
    private authService: AuthService,
    private spinnerService: SpinnerService,
    private utilSerivce: UtilsService) { }




  ngOnInit(): void {
    this.getLookups();

    this.canReviewGeneratedPayment = this.authService.isPageAccessible(
      `${UserModulesEnum.PAYMENT_GENERATE.module}.AID_PAYMENTS_GENERATE`
    );

    this.canUploadExtraPayments  = this.authService.isPageAccessible(
      `${UserModulesEnum.PAYMENT_GENERATE.module}.UPLOAD_EXTRA_PAYMENTS`
    );

    this.isDownloadDataAllowed = this.authService.isPageAccessible(this.reportDownloadWithNames) || this.authService.isPageAccessible(this.reportDownloadWithoutNames)
    this.canAccessReportGenerater= this.authService.isPageAccessible('VIEW_REPORTS_LIST');
    this.isUploadEducationalDataAllowed = this.authService.isPageAccessible(this.reportEducationalUpload);
  }

getLookups() {
  let dropdownFields = this.details.filter(f => f.type === FieldType.dropdown);
  let keys = dropdownFields.map(m => { return m.lookupKey })
  this.utilSerivce.getLookupsService([LookupsEnum.EDUCATION_REGISTRATION_YEAR]).subscribe(s => {
    if (s.body) {
      this.lookups[LookupsEnum.EDUCATION_REGISTRATION_YEAR] = s.body[LookupsEnum.EDUCATION_REGISTRATION_YEAR];
      this.formDetails.lookups = this.lookups;
    }
  });

  let FilesTypeArr = ['FilesType']
  this.utilSerivce.getLookupsService(FilesTypeArr).subscribe(s => {
    if (s.body) {
      FilesTypeArr.forEach(k => {
        s.body[k].forEach(l => {
          if(l.type == "EDUCATION_REGISTRATION"){
            this.educationalId = l.id;
          }
        });
      });
    }
  });
}

isAllowed(permission: string) {
  return this.authService.isPageAccessible(permission);
}


downloadEducationPersons() {

  const dialogRef = this.dialog.open(DetailsDialogComponent, {
    panelClass: 'details-dialog',
    disableClose: true,
    width: '80%',
    data: {
      lookups: this.lookups,
      fields: this.downloadEducationPersonsDetails,
      canClose: true,
      modalTitle: 'home.households.download_history_files',
    },
  });
  dialogRef.componentInstance.result.subscribe((result) => {

    if (result) {
      const dialogRefConfirmation = this.dialog.open(ConfirmationDialogComponent, {
        width: '400px',
        data: {
           body: `${this.translateService.instant(
                      'confirmation_dialog.are_you_sure'
                    )}`,
          saveButtonText: 'confirmation_dialog.confirm',
          discardButtonText: 'confirmation_dialog.reject'
         }
      })

      dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
        if (confirmation) {         
          this._service.downloadEducationPersonReport(result).subscribe(data => {
            dialogRef.close();
            if (this.canAccessReportGenerater) {
              this.navigateToGeneratedReport('../generated-reports', this.educationalId);
            }
            else {
              this.snacks.openSnackBar(
                `${this.translateService.instant(
                  'home.households.report_generated_sucessfully'
                )}`
              );
            }
          }, (err) => {
            this.dialog.open(WarningDialogComponent, {
              width: '400px',
              data: { arabic: this.translateService.instant(err.message), canClose: true }
            });
          });
            }
          })
    }
    });
}

navigateToGeneratedReport(route,id) {
  this.router.navigate([route], { relativeTo: this.route, queryParams:{'id':this.educationalId} });
}

openUploadFileDialog(){
  const dialogRef = this.dialog.open(UploadFileDialogComponent, {
    panelClass: 'details-dialog',
    disableClose: true,
    width: '80%',
    maxWidth: '650px'
  });


  dialogRef.afterClosed().subscribe((result) => {
    if (result) {
      const dialogRefConfirmation = this.dialog.open(ConfirmationDialogComponent, {
        width: '400px',
        data: {
           body: `${this.translateService.instant(
                      'confirmation_dialog.are_you_sure'
                    )}`,
          saveButtonText: 'confirmation_dialog.confirm',
          discardButtonText: 'confirmation_dialog.reject'
         }
      })
      dialogRefConfirmation.afterClosed().subscribe((confirmation) => {
        if (confirmation) {
          this.startUploadProcess(result);
        }
      })
  
    }
  });
}

startUploadProcess(file : any){
    var extension = file.name.split('.')[1].toLowerCase();
    if(extension == 'csv' || extension == 'xls' || extension == 'xlsx'){
      this._service.uploadEducationalDataStudentUpdate(file)
      .subscribe(
        _ => {
            this.snacks.openSnackBar(
            `${this.translateService.instant(
              'home.households.report_generated_sucessfully'
            )}`
          );
        },     
        error => {
          this.handleConflictError('ERROR_UPLOAD',error.message);
        }
      );
    }
    else{
      this.dialog.open(WarningDialogComponent, {
        width: '400px',
        data: {
          arabic: 'home.households.unsupported_file_format',
          canClose: true,
        },
      });
    }
  }

handleConflictError(error,message?: string) {
  if (error == 'conflict') {
    this.dialog.open(WarningDialogComponent, {
      width: '400px',
      data: {
        arabic: 'home.households.progress_running',
        canClose: true,
      },
    });
    return;
  }
  if ( error === 'ERROR_UPLOAD'){
    this.dialog.open(WarningDialogComponent, {
      width: '800',
      data: {
        title:'home.households.failed_upload',
        english: message,
        arabic: 'home.households.failed_upload',
        canClose: true,
      },
    });
    return;

  }

}


}
