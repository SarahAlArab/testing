import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { DetailsDialogComponent } from 'src/app/shared/components/details-dialog/details-dialog.component';
import { WarningDialogComponent } from 'src/app/shared/components/dialogs/warning-dialog/warning-dialog.component';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { FilterConfig } from 'src/app/shared/models/filter-config.model';
import { LookupsEnum } from 'src/app/shared/models/lookups.enum';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { PaginatorConfig } from 'src/app/shared/models/paginator-config.model';
import { UserModulesEnum, UserRolesActionsEnum } from 'src/app/shared/models/user-role.enum';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { HomeService } from '../home.service';
import { FilterbyOptions, SupervisorsColumnDef, SupervisorsDetails } from './supervisors.config';
import { SupervisorsService } from './supervisors.service';

@Component({
  selector: 'wfps-supervisors',
  templateUrl: './supervisors.component.html',
  styleUrls: ['./supervisors.component.sass']
})
export class SupervisorsComponent extends DataTableBaseClass<any> implements OnInit {

  dataTableColumnsDefinition = SupervisorsColumnDef;
  details = SupervisorsDetails;
  filterbyOptions = FilterbyOptions;
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  filterConfig = new FilterConfig(ConfigTypeEnum.NONE);
  lookups: any = {};
  moduleName = UserModulesEnum.SUPERVISORS;
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.supervisors.empty_table'
  };
  canUpdateStatus: boolean = true;
  editedDetails;
  disabledFields = ["email"];

  constructor(
    public _service: SupervisorsService,
    private dialog: MatDialog,
    public injector: Injector,
    private utilsService: UtilsService,
    private _homeService: HomeService
  ) {
    super(injector);
  }

  ngOnInit(): void {
    super.ngOnInit();
    this.updateAccess();
    this.updateColumns();
    this.getLookups();
    this.getListing(null, true);
    this.updateDetails();
  }

  private updateAccess() {
    this.canUpdateStatus = this.authService.isActionAllowed(this.moduleName, UserRolesActionsEnum.STATUS_UPDATE);
  }

  private updateDetails() {
    this.editedDetails = this.details.map(detail => {
      const newDetail = { ...detail };
      this.disabledFields.forEach(e => {
        if (newDetail.prop === e) {
          newDetail.isDisabled = true
        }
      })
      return newDetail
    });
  };

  private updateColumns() {
    this.dataTableColumnsDefinition?.forEach(e => {
      if (e.property === '!suspended') {
        if (!this.canUpdateStatus)
          e.disabled = true;
      }
      else if (e.property === 'edit') {
        if (!this.canViewDetails)
          e.visible = false;
      }
    });
  }

  private getLookups() {
    let dropdownFields = this.details.filter(f => f.type == FieldType.dropdown);
    let keys = dropdownFields.map(m => { return m.lookupKey })
    this.utilsService.getLookupsService(keys).subscribe(s => {
      if (s.body) {
        keys.forEach(k => {
          if (k === LookupsEnum.Entity) {
            this._homeService.getActiveInstitutions().subscribe(Entities => {
              this.lookups[LookupsEnum.Entity] = Entities.body
            })
          } else {
            this.lookups[k] = s.body[k];
          }
        });
      }
    });

    //Filter keys
    let dropdownFilterFields = this.filterbyOptions.filter(f => f.type == FieldType.dropdown);
    let filterkeys = dropdownFilterFields.map(m => { return m.lookupKey })

    filterkeys.forEach(k => {
      if (k === LookupsEnum.allEntities) {
        this._homeService.getAllInstitutions().subscribe(Entities => {
          this.lookups[LookupsEnum.allEntities] = Entities.body
        })
      }
    });

    this.lookups['itemStatus'] = this.utilsService.getBoolLookups('معلق', 'فعال');

  }

  getListing(paginationObj: PageChangeEvent, start: boolean, filter?: string) {
    this._service.getList(paginationObj, start, filter).subscribe(
      (data) => {
        if (data && data.body) {
          this.setDataTableData(data.body, start);
        }
      },
      (err) => { }
    );
  }

  public openDetails(element?: any, title?: string) {
    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: "details-dialog",
      disableClose: true,
      width: '90%',
      data: {
        data: element,
        fields: element ? this.editedDetails : this.details,
        lookups: this.lookups,
        modalTitle: title,
        isDisabled: element ? !this.canEdit : !this.canAdd
      },
    });

    dialogRef.componentInstance.result.subscribe((result) => {
      let objToPost = result;
      objToPost.name = `${result.firstName} ${result.lastName}`;
      objToPost.surveyEntity = {
        id: result.surveyEntity,
        name: this.lookups[LookupsEnum.Entity].find(f => f.id === result.surveyEntity).name
      };

      if (element && result) {
        this.applyEdit(element, objToPost, dialogRef);
      }
      else if (result) {
        this.applyAdd(objToPost, dialogRef);
      }
    });
  }

  private applyEdit(element, objToPost, dialogRef) {
    this._service.editRecord(element.id, objToPost).subscribe(res => {
      if (res?.body) {
        objToPost.id = res.body.id;
        let index = this.dataTableData.visibleRows.findIndex(x => x.id === element.id);
        index >= 0 ? this.dataTableData.visibleRows[index] = Object.assign(this.dataTableData.visibleRows[index], objToPost) : null;
      }
      this.snacks.openSnackBar("snackbar.edit_success");
      dialogRef.close();
    }, error => {
      this.onSaveError(true, error);
    })
  }

  private applyAdd(objToPost, dialogRef) {
    const confirmDialogRef = this.dialog.open(WarningDialogComponent, {
      width: '400px',
      disableClose: true,
      data: { arabic: 'home.generic_text.user_creation', canClose: true, action: 'home.generic_text.yes', closeAction: 'home.generic_text.no' }
    });

    confirmDialogRef.afterClosed().subscribe(s => {
      if (s) {
        objToPost.suspended = false;
        this._service.addRecord(objToPost).subscribe(res => {
          dialogRef.close();
          if (res?.body) {
            this.getListing(null, true);
            this.snacks.openSnackBar("snackbar.add_success");
          }
        }, error => {
          this.onSaveError(false, error);
        });
      }
    })

  }

  private onSaveError(isEdit, error) {
    //handle duplicate email
    if (error.code === 'duplicate_key') {
      this.snacks.openWarningSnackBar("snackbar.duplicate_email_error");
    } else {
      //general error
      this.snacks.openWarningSnackBar(`snackbar.${isEdit ? 'edit' : 'add'}_error`);
    }
  }

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getListing(pageChangeEvent, false, this.filter);
  }

  public newClicked() {
    this.openDetails(undefined, "home.supervisors.add");
  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'edit':
        this._service.getRecordById(ev.data.id).subscribe(result => {
          this.openDetails(result.body);
        })
        break;
      default:
        break;
    }
  };

  clickToggle(ev) {
    switch (ev.name) {
      case '!suspended':
        this.deactivateToggle(ev);
        break;
      default:
        break;
    };
  };

  private deactivateToggle(event) {
    let objToPost = event.data;
    objToPost.suspended = !event.value;

    this._service.editStatus(objToPost.id, objToPost.suspended).subscribe(res => {
      this.snacks.openSnackBar("snackbar.edit_success");
    }, error => {
      event.element.toggle(); //revert on error
      this.snacks.openWarningSnackBar("snackbar.edit_error");
    })
  };

  searchFilter(events) {
    //The filter is a string of key:value seperated by comma
    this.filter = this.frameworkHelper.getFilterString(events, 'suspended');

    //Call List API
    this.getListing(null, true, this.filter);

    //Clear filters
    if (events == null) {
      this.header.togglePanel();
    }
  }

}
