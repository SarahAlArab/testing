  constructor(
    public _service: HouseholdsService,
    private dialog: MatDialog,
    public injector: Injector,
    public translateService: TranslateService,
    private utilsService: UtilsService,
    private _homeService: HomeService,
  ) {
    super(injector);

    this.showBulkAuthUnauthorization = this.route.snapshot.data?.showBulkAuthUnauthorization;

    this.filterbyOptions = UtilsService.copyConfigs(this.showBulkAuthUnauthorization ? SurveyFilterbyOptions : FilterbyOptions);
  }

  ngOnInit(): void {
    this.dataTableColumnsDefinition = JSON.parse(
      JSON.stringify(this.showBulkAuthUnauthorization ? AssignedColumnDefWithoutActions : AssignedColumnDef)
    );
    this.dataTableColumnsDefinition.splice(0, 0, ConsistencyColumnDef)
    super.ngOnInit();
    this.updateAccess();
    this.updateColumns();
    this.updateDetails();

    this.isLookupsLoaded = false;

    this.getInitialData()
      .pipe(
        tap(() => {

          this.handleFilter();
        })
      )
      .subscribe({

        next: () => {

          this.isLookupsLoaded = true;
          this.popuplateList()
        }
      });
  }

  private handleFilter() {

    this.setFilter();

    if (this.showBulkAuthUnauthorization) {

      const essnApprovedStatus = this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS]?.find(obj => obj.type === 'ESSN_APPROVED');
      if (essnApprovedStatus) {
        const filterByStatus = this.filterbyOptions.find(f => f.prop == 'applicationStatus');
        if (filterByStatus) {
          filterByStatus.defaultValue = essnApprovedStatus.id;

          this.filterObject = { applicationStatus: { value: essnApprovedStatus.id, type: essnApprovedStatus.id } };
          this.filter = this.frameworkHelper.getFilterString(this.filterObject, this.booleanFilterFieldsThatAreActingAsDropdowns[0], this.booleanFilterFieldsThatAreActingAsDropdowns[1]);
        }
      }
    }
  }

  private popuplateList() {

    this.paginationIndex = this._service.getPaginationIndexValue();
    const withResett = this.paginationIndex && this.paginationIndex.length != 0 ? false : true;
    return this.getListing(this.paginationIndex, withResett, this.filter, true);
  }

  fieldChange(event) {
    if (event && event.action && event.action.field === "applicationStatus") {
      let myFilter = event.filterFormController.get('previousStatus');
      if (event.action.value) {
        const status = this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS].find(e => e.id == event.action.value);
        if (status && status.type === ApplicationScheduledStatusEnum.Suspended) {
          myFilter.enable();
        }
        else {
          myFilter.value = '';
          myFilter.disable();
        }
      }
      else {
        myFilter.value = '';
        myFilter.disable();
      }
    }
  }

  setFilter() {
    let filterValues = this._service
      .getFilterVal()
      ?.find((f) => f.component === 'assigned')?.filter;

    if (filterValues) {
      this.filter = this.frameworkHelper.getFilterString(
        filterValues,
        this.booleanFilterFieldsThatAreActingAsDropdowns[0],
        this.booleanFilterFieldsThatAreActingAsDropdowns[1]
      );

      this.element = this.frameworkHelper.formatElement(filterValues);
      this.filterObject = this.element;
    }
  }

  private updateAccess() {
    this.canViewPaymentsHistory = this.authService.isModuleAccessible(
      'AID_PAYMENTS_HISTORY'
    );
    this.canSchedule = this.authService.isActionAllowed(
      this.moduleName,
      UserRolesActionsEnum.SCHEDULE
    );
    this.bulkOperationsAllowed = this.authService.isPageAccessible(
     this.bulkOperations
    );
    this.canReview = this.authService.isActionAllowed(
      this.moduleName,
      UserRolesActionsEnum.REVIEW
    );
    this.canAuthorize = this.authService.isActionAllowed(
      this.moduleName,
      UserRolesActionsEnum.AUTHORIZE
    );
    this.canReportIssue = this.authService.isModuleAccessible(
      'REPORT_AID_CARD_ISSUE'
    );
    this.canResolveIssue = this.authService.isModuleAccessible(
      'RESOLVE_AID_CARD_ISSUE'
    );
    this.canResolveLocationIssue = this.authService.isModuleAccessible(
      'RESOLVE_FLAG_ISSUES'
    );
    this.viewComments = this.authService.isModuleAccessible('APP_STATUS_LIST');
    this.canViewHistory = this.authService.isPageAccessible(
      `${UserModulesEnum.ISSUE.module}.${UserModulesEnum.ISSUE.key}_${UserRolesActionsEnum.LIST}`
    );
  }

  private updateColumns() {
    this.updateColumn(!this.canSchedule, 'assign');
    this.updateColumn(!this.canReview, 'review');
    this.updateColumn(!this.canAuthorize, 'authorize');
    this.updateColumn(!this.canReportIssue, 'reportIssue');
    this.updateColumn(!this.canResolveIssue, 'resolveIssue');
    this.updateColumn(!this.canReportIssue, 'changeBeneficiary');
    this.updateColumn(!this.canResolveIssue, 'resolveChangeBeneficiary');
    this.updateColumn(!this.canResolveLocationIssue, 'resolveLocationIssue');
    this.updateFilter(!this.canReportIssue && !this.canResolveIssue, 'issues');
    this.updateColumn(
      this.authService.getUserRole() !== UserRolesEnum.NPTP_USER,
      'onHold'
    );
    this.updateColumn(!this.canViewPaymentsHistory, 'historyPayments');
    this.updateFilter(this.authService.getUserRole() !== UserRolesEnum.NPTP_USER, this.booleanFilterFieldsThatAreActingAsDropdowns[1]);
  }

  private updateColumn(condition: boolean, propertyName: string) {
    if (condition) {
      this.dataTableColumnsDefinition = this.dataTableColumnsDefinition.filter(
        (f) => f.property != propertyName
      );
      let actionsCol = this.dataTableColumnsDefinition.find(
        (f) => f.format == DATA_TABLE_COLUMN_FORMAT.Actions
      );
      if (actionsCol) {
        actionsCol.options = actionsCol.options.filter(
          (f) => f.property != propertyName
        );
      }
    }
  }

  private updateFilter(condition: boolean, propertyName: string) {
    if (condition) {
      this.filterbyOptions = this.filterbyOptions.filter(
        (f) => f.prop != propertyName
      );
    }
  }

  private updateFilterStatus() {

    if (
      !this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS] ||
      !this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS]?.length
    ) {
      this.filterbyOptions = this.filterbyOptions.filter(
        (f) => f.prop != 'applicationStatus'
      );
    }

  }

  private updatePreviousFilterStatus() {
    if (
      !this.filterLookups[LookupsEnum.FILTERED_PREVIOUS_STATUS] ||
      !this.filterLookups[LookupsEnum.FILTERED_PREVIOUS_STATUS]?.length
    ) {
      this.filterbyOptions = this.filterbyOptions.filter(
        (f) => f.prop != 'previousStatus'
      );
    }
  }

  private updateDetails() {
    //if my user is supervisor, remove entity and supervisor from my fields AND lookups
    if (this.authService.getUserRole() == UserRolesEnum.SUPERVISOR) {
      this.filterbyOptions = this.filterbyOptions.filter(
        (f) => f.prop != 'supervisorUser'
      );
    }
  }

  getListing(
    paginationObj: PageChangeEvent,
    start: boolean,
    filter?: string,
    openPanel?: boolean
  ) {

    this.totalElements = 0;

    if (paginationObj) {
      this._service.setPaginationIndexValue(paginationObj);
    }

    this._service.getAssignedList(paginationObj, start, filter).subscribe(
      (data) => {
        if (data && data.body) {
          this.updateListData(data.body.content);
          this.setDataTableData(data.body, start);

          //check for previous filter, and if we're calling getlisting onInit
          if (this.filter && openPanel) {
            this.header.togglePanel();
          }

          this.totalElements = data.body.totalElements;
        }
      },
      (err) => { }
    );
  }

  updateListData(data) {
    data.forEach((e) => {
      this.updateSurveyorDisplay(e);
      this.updateActionsByStatus(e);
      this.updateActionsByPrevStatus(e);
      this.updateActionByStatusDetails(e);
      this.updateActionByIssues(e);
      this.updateActionByBeneficiaryIssue(e);
      this.updateOnHoldStatus(e);
      this.updateMophStatusDisplay(e);
      this.updateKadaaDisplay(e);
      this.updateConsistencyFlagMarkAndActions(e)
    });
  }
  updateConsistencyFlagMarkAndActions(element: any) {
    const hasConsistencyIssues = element?.essnConsistencyChecks?.length > 0;
    if (hasConsistencyIssues && element?.applicationStatus?.type == ApplicationStatusEnum.EssnSubmitted && this.canResolveLocationIssue) {
      element.consistency = RedFlagIcon
      element.isResolveLocationIssueDisabled = false;
    } else {
      element.consistency = ''
      element.isResolveLocationIssueDisabled = true;
      if (element.actions_badge?.length) {
        element.actions_badge = element.actions_badge.filter(
          (f) => f !== 'resolveLocationIssue'
        );
      }
    }

  }

  updateActionByStatusDetails(element) {
    if (!element.hasStatusDetails) {
      element.isStatusDetailsDisabled = true;
      element.statusDetails_icon = 'assets/icons/comment-grey.svg';
    }
  }

  updateOnHoldStatus(element) {
    const onHold =
      element.issues.filter((x) => x.status.type === 'ON_HOLD').length > 0;
    element.onHold = onHold ? '' : '';
  }

  updateActionByIssues(element) {
    const contactIssue = element?.issues?.find((x) => x.type.type == 'CONTACT');
    if (contactIssue) {
      element.isResolveIssueDisabled = false;
      element.reportIssue_icon = 'assets/icons/dangerous-orange.svg';
      if (this.canResolveIssue) {
        if (!element.actions_badge) element.actions_badge = [];
        element.actions_badge.push('resolveIssue');
      }
    } else {
      element.isResolveIssueDisabled = true;
      element.resolveIssue_icon = 'assets/icons/report_prob-grey.svg';
      if (this.canResolveIssue) {
        if (element.actions_badge?.length) {
          element.actions_badge = element.actions_badge.filter(
            (f) => f !== 'resolveIssue'
          );
        }
      }
    }
  }
  updateActionByBeneficiaryIssue(element) {
    const beneficiaryIssue = element?.issues?.find(
      (x) => x.type.type == ISSUE_TYPE.BENEFICIARY && x.status.type != 'ON_HOLD'
    );

    const onHoldIssue = element?.issues?.find(
      (x) => x.type.type == ISSUE_TYPE.BENEFICIARY && x.status.type == 'ON_HOLD'
    );

    if (onHoldIssue) {
      element.changeBeneficiary_icon = 'assets/icons/dangerous-red.svg';
    }
    if (beneficiaryIssue) {
      element.isResolveBeneficiaryDisabled = false;
      element.changeBeneficiary_icon = 'assets/icons/dangerous-orange.svg';
      if (this.canResolveIssue) {
        if (!element.actions_badge) element.actions_badge = [];
        element.actions_badge.push('changeBeneficiary');
      }
    } else {
      element.isResolveBeneficiaryDisabled = true;
      element.resolveChangeBeneficiary_icon =
        'assets/icons/report_prob-grey.svg';
      if (this.canResolveIssue) {
        if (element.actions_badge?.length) {
          element.actions_badge = element.actions_badge.filter(
            (f) => f !== 'changeBeneficiary'
          );
        }
      }
    }
  }

  updateSurveyorDisplay(element) {
    if (element.surveyorUser) {
      element.surveyorUser.name = `${element.surveyorUser.name} - ${element.surveyorUser.email}`;
    }
  }

  updateMophStatusDisplay(element) {
    if (element.mophApplicationStatus && element.mophApplicationStatus.id !== null) {
      const mophStatus = this.filterLookups[LookupsEnum.MOPH_ENROLLMENT_STATUS]?.find(key => key.id === element.mophApplicationStatus.id);
      if (mophStatus) {
        element.mophApplicationStatus = `${mophStatus.name} - ${mophStatus.nameEn}`;
      }
    }
    else
      element.mophApplicationStatus = null;

  }

  updateKadaaDisplay(element) {
    if (element.kadaa && element.kadaa.id != null) {
      const kadaaObject = this.filterLookups[LookupsEnum.Kadaa]?.find(key => key.id === element.kadaa.id);
      if (kadaaObject)
        element.kadaa = `${kadaaObject.name}`;
    }
    else
      element.kadaa = null;

  }

          element.actions_badge.push('authorize');
        }

          if (element.actions_badge?.length) {

      case this.applicationScheduledStatusEnum.Rejected:

        element.essnToolTip = 'application_status.rejected';

        element.audit_icon = 'assets/icons/person-green.svg';

        element.isScheduleDisabled = true;

        element.isReviewDisabled = true;

        element.isReportIssueDisabled = false;
        element.isReportBeneficiaryDisabled = true;
        element.isPaymentHistoyDisabled = true;
        //element.statusAbv = 'Auth';
        if (this.canAuthorize) {
          if (element.actions_badge?.length) {
            element.actions_badge = element.actions_badge.filter(
              (f) => f !== 'authorize'
            );

    this.isLookupsLoaded = false;

      LookupsAppStatusTypeEnum.ESSN_IN_PROGRESS,
      LookupsAppStatusTypeEnum.ESSN_SUBMITTED,
      LookupsAppStatusTypeEnum.ESSN_REJECTED,

      LookupsAppStatusTypeEnum.PAY_ENROLLED,
      LookupsAppStatusTypeEnum.PAY_CANCELLED,
      LookupsAppStatusTypeEnum.SUSPENDED,
    ];

    if (this.showBulkAuthUnauthorization) {

      displayedStatus = [
        LookupsAppStatusTypeEnum.ESSN_APPROVED,
        LookupsAppStatusTypeEnum.ESSN_AUTHORIZED,
        LookupsAppStatusTypeEnum.ESSN_UNAUTHORIZED
      ]
    }

    const displayedPreviousStatus = [
      LookupsAppStatusTypeEnum.SCHEDULED,
      LookupsAppStatusTypeEnum.ESSN_IN_PROGRESS,
      LookupsAppStatusTypeEnum.ESSN_SUBMITTED,
      LookupsAppStatusTypeEnum.ESSN_REJECTED,
      LookupsAppStatusTypeEnum.ESSN_APPROVED,
      LookupsAppStatusTypeEnum.ESSN_AUTHORIZED,
      LookupsAppStatusTypeEnum.ESSN_UNAUTHORIZED,
      LookupsAppStatusTypeEnum.PAY_ENROLLED,
      LookupsAppStatusTypeEnum.PAY_CANCELLED,
    ];

    const keysToExclude = [
      LookupsEnum.BOOL_LIST_PHONE,
      LookupsEnum.BOOL_LIST_ISSUES,
      LookupsEnum.Surveyor,
      LookupsEnum.Superviors,
      LookupsEnum.FILTERED_APPLICATION_STATUS,
      LookupsEnum.FILTERED_PREVIOUS_STATUS
    ];

    let dropdownFields = this.filterbyOptions.filter(
      (f) => f.type == FieldType.dropdown
    );
    let keys = dropdownFields.map((m) => {
      return m.lookupKey;
    });

    keys = keys.filter((k) => !keysToExclude.includes(k));

    return forkJoin(
      [
        this.utilsService.getLookupsService(keys),
        this.getAllSurveyors(),
        this.getAllSupervisors(),
        this.getFilteredApplicationStatus(displayedStatus, displayedPreviousStatus)
      ]
    )
      .pipe(
        tap((res) => {

          keys.forEach((k) => {
            this.filterLookups[k] = res[0].body[k];
          });
          this.filterLookups[LookupsEnum.BOOL_LIST_ISSUES] = this.utilsService.getBoolLookups(' ', ' ');
          this.filterLookups[LookupsEnum.BOOL_LIST_PHONE] = this.utilsService.getBoolLookups('   ', '    ');
        }));
  }

  private getAllSurveyors() {
    return this._homeService.getAllSurveyors().pipe(tap((data) => {
      this.filterLookups[LookupsEnum.Surveyor] = data.body;
    }));
  }

  private getAllSupervisors() {
    const role = this.authService.getUserRole();
    return (role == UserRolesEnum.NPTP_USER || role == UserRolesEnum.WFP_ADMIN) ?
      this._homeService.getAllSupervisors().pipe(
        tap((data) => {
          this.filterLookups[LookupsEnum.Superviors] = data.body;
        })
      ) :
      of([]).pipe(
        tap(() => {
          this.filterLookups[LookupsEnum.Superviors] = [];
        })
      );
  }

  private getFilteredApplicationStatus(displayedStatus, displayedPreviousStatus) {

    return this._homeService
      .getFilteredApplicationStatus()
      .pipe(
        map((response) => {
          const lookuptsToReturn = response.body.map((x) => {
            x.name = '';
            return x;
          });
          return { body: lookuptsToReturn };
        }),
        tap((s) => {

          this.filterLookups[LookupsEnum.FILTERED_APPLICATION_STATUS] = s.body?.filter((f) => displayedStatus.includes(f.type));
          this.updateFilterStatus();

          this.filterLookups[LookupsEnum.FILTERED_PREVIOUS_STATUS] = s.body?.filter((f) => displayedPreviousStatus.includes(f.type));
          this.updatePreviousFilterStatus();
        })
      )
  }

  public openReview(element: any, type) {
    let approvalLookup: any = {};
    approvalLookup.decisions = [
      { id: 1, name: this.translateService.instant('home.households.approve') },
      { id: 2, name: this.translateService.instant('home.households.reject') },
    ];

    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: 'details-dialog',
      disableClose: true,
      width: '50%',
      data: {
        data: element,
        fields: this.approveField,
        lookups: approvalLookup,
        modalTitle: 'home.households.decision_title',
      },
    });

    //detect field change of decision, if rejected => make statusDetails required
    dialogRef.componentInstance.fieldChange.subscribe((control) => {
      if (control.field === 'decision' && control.value) {
        if (control.value === 2) {
          this.approveField.forEach((detail) => {
            if (detail.prop === 'statusDetails') {
              control.form.controls.statusDetails.setValidators(
                Validators.required
              );
              control.form.controls.statusDetails.updateValueAndValidity();
            }
          });
        } else {
          this.approveField.forEach((detail) => {
            if (detail.prop === 'statusDetails') {
              control.form.controls.statusDetails.setValidators(null);
              control.form.controls.statusDetails.updateValueAndValidity();
            }
          });
        }
      }
    });

    dialogRef.componentInstance.result.subscribe((result) => {
      if (result) {
        let objToPost = {
          approved: result.decision === 1,
          statusDetails: result.statusDetails,
        };

        this._service.review(element.id, objToPost, type).subscribe((res) => {
          if (res?.body) {
            element.applicationAuditStatus = res.body.applicationAuditStatus;
            element.applicationAuditStatusLocked =
              res.body.applicationAuditStatusLocked;
          }
          this.updateActionsByStatus(element);
          this.updateActionsByPrevStatus(element);
          let index = this.dataTableData.visibleRows.findIndex(
            (x) => x.id === element.id
          );
          index >= 0 ? (this.dataTableData.visibleRows[index] = element) : null;

          //if i added a comment, remove disable from comment section
          if (objToPost.statusDetails) {
            element.isStatusDetailsDisabled = false;
            element.statusDetails_icon = 'assets/icons/comment.svg';
          }
          this.snacks.openSnackBar('snackbar.edit_success');
          dialogRef.close();
        });
      }
    });
  }

            id: element.id,

            id: result.surveyorUser,

              (f) => f.id === result.surveyorUser

  private reportIssue(element?: any) {
    let issue;
    let disableName;
    let disablePhoneNumber;
    let disableDialog = false;

    const contactIssue = element?.issues?.find(
      (x) => x.type.type == ISSUE_TYPE.CONTACT
    );
    if (contactIssue) {
      issue = {
        id: contactIssue.id,
        isNameWrong: contactIssue.tags.includes(ISSUE_TAG.NAME_CHANGE),
        isMobileWrong: contactIssue.tags.includes(ISSUE_TAG.MOBILE_CHANGE),
        comment: contactIssue.comment,
      };
    }

    const allowedIssueTypes = element?.allowedIssueTypes.find(
      (x) => x.key == ISSUE_TYPE.CONTACT
    );
    if (allowedIssueTypes.allowed) {
      disableName = allowedIssueTypes.tags.find(
        (x) => x.key === ISSUE_TAG.NAME_CHANGE
      ).allowed;
      disablePhoneNumber = allowedIssueTypes.tags.find(
        (x) => x.key === ISSUE_TAG.MOBILE_CHANGE
      ).allowed;
      this.reportIssueField.find((x) => x.prop == 'isNameWrong').isDisabled =
        !disableName;
      this.reportIssueField.find((x) => x.prop == 'isMobileWrong').isDisabled =
        !disablePhoneNumber;
    } else {
      disableDialog = true;
    }

    if (this.canViewHistory) {
      this._service.getHistory(element.id).subscribe((dataTable) => {
        const dialogRef = this.dialog.open(HistoryDialogComponent, {
          panelClass: 'details-dialog',
          disableClose: true,
          width: '80%',
          data: {
            data: issue,
            fields: this.reportIssueField,
            modalTitle: 'home.households.report_issue_title',
            dataTable,
            showTable: dataTable.content.length > 0,
            isDisabled: disableDialog,
          },
        });
        this.sendReportIssueRequest(dialogRef, element);
      });
    } else {
      const dialogRef = this.dialog.open(DetailsDialogComponent, {
        panelClass: 'details-dialog',
        disableClose: true,
        width: '80%',
        data: {
          data: issue,
          fields: this.reportIssueField,
          modalTitle: 'home.households.report_issue_title',
          isDisabled: disableDialog,
        },
      });
      this.sendReportIssueRequest(dialogRef, element);
    }
  }

  private sendReportIssueRequest(dialogRef: MatDialogRef<any>, element: any) {
    dialogRef.componentInstance.result.subscribe((result) => {
      if (result) {
        let objToPost: ReportIssueModel = {
          type: ISSUE_TYPE.CONTACT,
          tags: [],
          comment: result.comment,
        };
        if (result.isNameWrong) {
          objToPost.tags.push(ISSUE_TAG.NAME_CHANGE);
        }
        if (result.isMobileWrong) {
          objToPost.tags.push(ISSUE_TAG.MOBILE_CHANGE);
        }
        if (!result.isNameWrong && !result.isMobileWrong) {
          return;
        }

        this._service.reportIssue(element.id, objToPost).subscribe((res) => {
          if (res?.body) {
            let index = this.dataTableData.visibleRows.findIndex(
              (x) => x.id === element.id
            );
            if (index >= 0) {
              const beneficiaryIssue = this.dataTableData.visibleRows[
                index
              ].issues.find((x) => x.type.type == ISSUE_TYPE.BENEFICIARY);

              this.dataTableData.visibleRows[index].issues = res.body.issues;

              if (beneficiaryIssue) {
                this.dataTableData.visibleRows[index].issues.push(
                  beneficiaryIssue
                );
              }
            }
            this.updateActionByIssues(this.dataTableData.visibleRows[index]);

            this.snacks.openSnackBar('snackbar.edit_success');
            dialogRef.close();
          }
        });
      }
    });
  }

  private sendResolveIssueRequest(
    dialogRef: MatDialogRef<any>,
    element: any,
    contactIssue
  ) {
    dialogRef.componentInstance.result.subscribe((result) => {
      if (result) {
        let objToPost: ResolveIssueModel = {
          mobileNumber: result.mobileNumber,
          firstName: result.firstName,
          fatherName: result.fatherName,
          firstNameEn: result.firstNameEn,
          fatherNameEn: result.fatherNameEn,
          lastName: result.lastName,
          lastNameEn: result.lastNameEn,
          aidCardIssue: contactIssue.id,
          metadata: {
            cancelPayments: element.cancelPayments,
            noChanges: element.noChanges
          },
        };
        if (element.noChanges) {

          let newDataConcat = `${element.firstName} ${element.fatherName} ${element.lastName} ${element.firstNameEn} ${element.fatherNameEn} ${element.lastNameEn} ${element.mobileNumber}`;

          if (element.oldDataConcat !== newDataConcat) {
            const confirmialogRef = this.dialog.open(ConfirmationDialogComponent, {
              width: '600px',
              data: {
                body: 'confirmation_dialog.discard_on_no_change',
                saveButtonText: 'confirmation_dialog.confirm',
                discardButtonText: 'confirmation_dialog.reject'
              }
            })
            confirmialogRef.afterClosed().subscribe((confirmResult) => {
              if (confirmResult) {
                this.callResolveContactIssueService(element, objToPost, dialogRef);
              }
            })
          } else {
            this.callResolveContactIssueService(element, objToPost, dialogRef);
          }
        } else {
          this.callResolveContactIssueService(element, objToPost, dialogRef);
        }
      }
    });
  }

  private callResolveContactIssueService(element: any, objToPost: ResolveIssueModel, dialogRef: MatDialogRef<any, any>) {
    return this._service.addressIssue(element.id, objToPost).subscribe(
      (res) => {
        if (res?.body) {
          let index = this.dataTableData.visibleRows.findIndex(
            (x) => x.id === element.id
          );
          if (index >= 0) {
            const contactIndex = this.dataTableData.visibleRows[index].issues.findIndex((x) => x.type.type == 'CONTACT');

            this.dataTableData.visibleRows[index].issues.splice(
              contactIndex,
              1
            );

            this.dataTableData.visibleRows[index].name = `${res.body.firstName} ${res.body.fatherName} ${res.body.lastName}`;
            this.dataTableData.visibleRows[index].mobileNumber =
              res.body.mobileNumber;
          }

          this.updateActionByIssues(this.dataTableData.visibleRows[index]);
          this.snacks.openSnackBar('snackbar.edit_success');
          dialogRef.close();
        }
      },
      (error) => {
        if (error.code === 'conflict') {
          let index = this.dataTableData.visibleRows.findIndex(
            (x) => x.id === element.id
          );
          if (index >= 0) {
            const contactIndex = this.dataTableData.visibleRows[index].issues.findIndex((x) => x.type.type == 'CONTACT');
            this.dataTableData.visibleRows[index].issues.splice(
              contactIndex,
              1
            );
          }
          this.updateActionByIssues(this.dataTableData.visibleRows[index]);
          dialogRef.close();
          this.snacks.openSnackBar(
            'snackbar.conflict_resolved',
            null,
            null,
            7000
          );
        }
      }
    );
  }

  private sendResolveBeneficiaryRequest(
    dialogRef: MatDialogRef<any>,
    element: any,
    beneficiaryIssue,
    beneficiaryId
  ) {
    dialogRef.componentInstance.result.subscribe((result) => {
      if (result) {
        let objToPost: ResolveIssueModel = {
          aidCardIssue: beneficiaryIssue.id,
          metadata: {
            cancelPayments: element.cancelPayments,
            changeBeneficiary: element.changeBeneficiary,
            alternateBeneficiaryId: beneficiaryId,
            noChanges: element.noChanges
          },
        };
        if (element.noChanges) {
          if (element.cancelPayments || element.changeBeneficiary) {
            const confirmialogRef = this.dialog.open(ConfirmationDialogComponent, {
              width: '600px',
              data: {
                body: 'confirmation_dialog.discard_on_no_change',
                saveButtonText: 'confirmation_dialog.confirm',
                discardButtonText: 'confirmation_dialog.reject'
              }
            })
            confirmialogRef.afterClosed().subscribe((confirmResult) => {
              if (confirmResult) {
                this.callResolveBeneficiaryIssueService(element, objToPost, dialogRef);
              }
            })
          } else {
            this.callResolveBeneficiaryIssueService(element, objToPost, dialogRef);
          }
          return;

        } else {
          if (!result.changeBeneficiary && result.cancelPayments) {
            this.dialog.open(WarningDialogComponent, {
              width: '400px',
              data: {
                arabic: 'home.households.no_cancel_payment_allowed',
                canClose: true,
              },
            });

            this.dialog.open(WarningDialogComponent, {
              width: '400px',
              data: {
                arabic: 'home.households.change_beneficiary_mandatory',
                canClose: true,
              },
            });
            return;
          }
        }

        this.callResolveBeneficiaryIssueService(element, objToPost, dialogRef);
      }
    });
  }

  searchFilter(events) {
    //The filter is a string of key:value seperated by comma
    this.filterObject = events;
    this.filter = this.frameworkHelper.getFilterString(events, this.booleanFilterFieldsThatAreActingAsDropdowns[0], this.booleanFilterFieldsThatAreActingAsDropdowns[1]);
    //save filter in global storage
    this._service.addFilter({ component: 'assigned', filter: events });

    //Clear filters
    if (events == null) {
      this.header.togglePanel();
    }
  }

  bulkAuthUnauth(toAuthorize: boolean) {

    if (!this.totalElements) {

      this.dialog.open(ConfirmationDialogComponent, {
        width: '600px',
        data: {
          body: 'home.households.bulkResultNoData',
          saveButtonText: 'confirmation_dialog.accept',
          withoutDiscard: true
        }
      });

      return;
    }

    const dialogRef = this.dialog.open(ConfirmationDialogComponent, {
      width: '600px',
      data: {
        body: toAuthorize ? 'home.households.auth_confirm' : 'home.households.unauth_confirm',
        saveButtonText: 'confirmation_dialog.confirm',
        discardButtonText: 'confirmation_dialog.reject'
      }
    });

    this.frameworkHelper.formatFilterObject(this.filterObject, this.booleanFilterFieldsThatAreActingAsDropdowns);

    dialogRef.afterClosed().subscribe((confirmResult) => {

      if (confirmResult) {

        this._service[toAuthorize ? 'doBulkAuthorize' : 'doBulkUnAuthorize'](this.filterObject).subscribe({

          next: (res) => {

            this._showBulkResult(res.body?.count, this.totalElements);
          },
          error: (e) => {

            this.snacks.openWarningSnackBar('generic.error');
          }
        });
      }
    })
  }

  private _showBulkResult(affectedRecordsNumber: number, totalNumber: number) {

    this.dialog.open(ConfirmationDialogComponent, {
      width: '600px',
      data: {
        body: affectedRecordsNumber ?
          this.translateService.instant(
            'home.households.bulkResult',
            { x: affectedRecordsNumber, totalNumber }
          ) :
          this.translateService.instant('home.households.bulkResultNoData'),
        saveButtonText: 'confirmation_dialog.accept',
        withoutDiscard: true
      }
    });
  }
