import { AuthService } from './../../services/auth.service';
import { ToolbarDrawerService } from './../../services/toolbar-drawer.service';
import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { Router } from '@angular/router';
import { DrawerNotification, DrawerLink } from '../../models/toolbar-drawer.model';

@Component({
  selector: 'app-toolbar-drawer',
  templateUrl: './toolbar-drawer.component.html',
  styleUrls: ['./toolbar-drawer.component.sass'],
})
export class ToolbarDrawerComponent implements OnInit {
  @Output() close = new EventEmitter<void>();
  config: any
 
  constructor(private router: Router, public toolbarDrawerService: ToolbarDrawerService) {}

  ngOnInit(): void {
    this.config = this.toolbarDrawerService.drawerConfig;
  }

  onClose() {
    this.close.emit();
    this.toolbarDrawerService.onClose();
  }

  navigateTo(url: string) {
    this.router.navigate([url]);
    this.onClose();
  }

  getImgSrc(){
    return `assets/images/news_sources/${this.config.content.source}-card.png`
  }
}
