import { Component, Injector, OnInit } from '@angular/core';
import { FilterConfig } from 'src/app/shared/models/filter-config.model';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { PaginatorConfig } from 'src/app/shared/models/paginator-config.model';
import { HouseholdsService } from '../households.service';
import { DataTableBaseClass } from 'src/app/shared/bases/data-table.base';
import { MatDialog } from '@angular/material/dialog';
import { ConfigTypeEnum } from 'src/app/shared/enums/config.enum';
import { DetailsDialogComponent } from 'src/app/shared/components/details-dialog/details-dialog.component';
import { LookupsEnum } from 'src/app/shared/models/lookups.enum';
import { UserModulesEnum, UserRolesActionsEnum } from 'src/app/shared/models/user-role.enum';
import { formatDate } from '@angular/common';
import { FieldType } from 'src/app/shared/models/fields-config.model';
import { UtilsService } from 'src/app/shared/services/utils.service';
import { UnassignedColumnDef, UnassignedDetails, FilterbyOptions } from './unassigned-households.config';
import { HomeService } from '../../home.service';

@Component({
  selector: 'wfps-unassigned-households',
  templateUrl: './unassigned-households.component.html',
  styleUrls: ['./unassigned-households.component.sass']
})
export class UnassignedHouseholdsComponent extends DataTableBaseClass<any> implements OnInit {

  dataTableColumnsDefinition = UnassignedColumnDef;
  details = UnassignedDetails;
  filterbyOptions = FilterbyOptions;
  paginatorConfig = new PaginatorConfig(ConfigTypeEnum.BE);
  filterConfig = new FilterConfig(ConfigTypeEnum.NONE);
  lookups: any = {};
  filterLookups: any = {};
  emptyTable = {
    icon: 'assets/icons/institution_empty.svg',
    text: 'home.households.empty_table'
  };
  moduleName = UserModulesEnum.HOUSEHOLDS;
  canSchedule: boolean = true;

  constructor(
    public _service: HouseholdsService,
    private dialog: MatDialog,
    public injector: Injector,
    private utilsService: UtilsService,
    private homeService: HomeService
  ) {
    super(injector);
  }

  ngOnInit(): void {
    super.ngOnInit();
    this.updateAccess();
    this.updateColumns();
    this.getFilterLookups();
    this.getListing(null, true);
  }

  private updateAccess() {
    this.canSchedule = this.authService.isActionAllowed(this.moduleName, UserRolesActionsEnum.SCHEDULE);
  }

  private updateColumns() {
    if (!this.canSchedule) {
      this.dataTableColumnsDefinition = this.dataTableColumnsDefinition.filter(f => f.property != 'assign')
    }
  }

  getListing(paginationObj: PageChangeEvent, start: boolean, filter?: string) {
    this._service.getUnassignedList(paginationObj, start, filter).subscribe(
      (data) => {
        if (data && data.body) {
          this.setDataTableData(data.body, start);
        }
      },
      (err) => { }
    );
  };

  getLookups(date: any, visit) {
    let dateDash = formatDate(date, 'dd-MM-yyyy', 'en');
    let dateSlash = formatDate(date, 'd/M/yyyy', 'en');
    this.homeService.getActiveSurveyorsByDate(dateDash).subscribe(s => {
      this.lookups[LookupsEnum.Surveyor] = s?.body;
      if (visit && visit.visitDate == dateSlash) {
        if (!this.lookups[LookupsEnum.Surveyor].find(f => f.id === visit.surveyorUser.id)) {
          this.lookups[LookupsEnum.Surveyor].push(visit.surveyorUser);
        }
      }
    })
  };

  private getFilterLookups() {
    let dropdownFields = this.filterbyOptions.filter(f => f.type == FieldType.dropdown);
    let keys = dropdownFields.map(m => { return m.lookupKey })
    this.utilsService.getLookupsService(keys).subscribe(s => {
      if (s.body) {
        keys.forEach(k => {
          if (k === LookupsEnum.Surveyor) {
            this.homeService.getAllSurveyors().subscribe(Surveyors => {
              this.filterLookups[LookupsEnum.Surveyor] = Surveyors.body;
            })
          }
          this.filterLookups[k] = s.body[k];
        });
      }
    });
  }

  public openDetails(element?: any, title?: string) {
    const dialogRef = this.dialog.open(DetailsDialogComponent, {
      panelClass: "details-dialog",
      disableClose: true,
      width: '80%',
      data: {
        data: element?.aidCardAuditVisit,
        fields: this.details,
        lookups: this.lookups,
        modalTitle: title
      },
    });

    dialogRef.componentInstance.fieldChange.subscribe((control) => {
      if (control.field === "visitDate" && control.value) {
        this.getLookups(control.value, element?.aidCardAuditVisit);
      }
    });

    dialogRef.componentInstance.result.subscribe((result) => {
      if (result) {
        let objToPost = {
          visitDate: formatDate(result.visitDate, 'dd/MM/yyyy', 'en'),
          aidCardApplication: {
            id: element.id
          },
          surveyorUser: {
            id: result.surveyorUser
          }
        };
        this._service.scheduleVisit(objToPost.aidCardApplication.id, objToPost).subscribe(res => {
          if (res?.body) {
            let index = this.dataTableData.visibleRows.findIndex(x => x.id === res.body.id);
            if (index >= 0) {
              this.dataTableData.visibleRows[index].aidCardAuditVisit =
                this.dataTableData.visibleRows[index].aidCardAuditVisit ? Object.assign(this.dataTableData.visibleRows[index].aidCardAuditVisit, res.body) : res.body;
              this.dataTableData.visibleRows[index].assign_icon = 'assets/icons/date-green.svg';
            }
          }
          this.snacks.openSnackBar("snackbar.edit_success");
          dialogRef.close();
        }, error => {
          if (error.code == "not_allowed") {
            // already assigned by another supervisor
            this.snacks.openWarningSnackBar("snackbar.already_assigned");
            this.getListing(null, true);
            dialogRef.close();
          } else {
            this.snacks.openWarningSnackBar("snackbar.edit_error");
          }

        })
      }
    });
  };

  onPage(pageChangeEvent: PageChangeEvent) {
    this.getListing(pageChangeEvent, false, this.filter);
  }

  actionClicked(ev) {
    switch (ev.name) {
      case 'assign':
        this.openDetails(ev.data, "home.households.surveyor");
        break;
      default:
        break;
    }
  };

  searchFilter(events) {
    //The filter is a string of key:value seperated by comma
    this.filter = this.frameworkHelper.getFilterString(events, 'suspended');

    //Call List API
    this.getListing(null, true, this.filter);

    //Clear filters
    if (events == null) {
      this.header.togglePanel();
    }
  };
}
