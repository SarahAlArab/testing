import { HttpClient } from '@angular/common/http';
import { Injectable, Injector } from '@angular/core';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';
import { HttpBaseClass } from '../bases/http-base.class';
import { FrameworkHelper } from '../helpers/framework.helper';
import { UserModule, UserRolesEnum } from '../models/user-role.enum';
import { User } from './../models/user.model';
import { LocalStorageService } from './local-storage.service';
import { SnackbarService } from './snackbar.service';
import { TenantService } from './tenant.service';
import { UtilsService } from './utils.service';
import { Microservice } from '../enums/microservice.enum';

@Injectable()
export class AuthService extends HttpBaseClass {
    public redirectTo: string;
    private loggedinUser: User;
    private frameworkHelper = FrameworkHelper;

    constructor(
        public http: HttpClient,
        public localStorageService: LocalStorageService,
        public router: Router,
        public injector: Injector,
        public tenantService: TenantService,
        public utilsService: UtilsService,
        private snacks: SnackbarService
    ) {
        super(http, injector);
    }

    login(username: string, password: string): Observable<any> {
        const loginObj = { username, password };
        return this.post('login', loginObj, undefined, true, this.settingsService.settings.authUrl +  Microservice.AUTH_V2);
    }

    sendOTP(username: string, code): Observable<any> {
        const loginObj = { username, code };
        return this.post('login/otp', loginObj, undefined, true, this.settingsService.settings.authUrl +  Microservice.AUTH_V2);
    }

    createMophUsers() {
        return this.post(`user/moph`, undefined, undefined, undefined, this.settingsService.settings.authUrl + Microservice.SECUIRITY);
    }
    
    createNptpMigrationUsers() {
        console.log('createNptpMigrationUsers', this.settingsService.settings.backofficeUrl + Microservice.SECUIRITY);
        return this.post(`user/nptpmig`, undefined, undefined, undefined, this.settingsService.settings.authUrl + Microservice.SECUIRITY);
    }

    onLogout() {
        this.clearUser();
        this.utilsService.clearAllCachedLookups();
        this.router.navigate(['/login']);
    }

    /**
  * GET user details
  */
    getProfile() {
        return this.get('user');
    }

    /**
     * PUT to update profile
     * @param obj user details
     */
    updateProfile(obj) {
        return this.put('user', obj);
    }

    setUser(user: User) {
        const savedUser = this.getUserFromLS();
        this.loggedinUser = savedUser ? Object.assign(savedUser, user) : user;
        this.saveUser(this.loggedinUser);

    }

    saveUser(user: User) {
        this.localStorageService.setItem(
            this.tenantService.tenantConfig?.config.local_storage_user_key,
            JSON.stringify(user)
        );
    }

    getUserFromLS(): User {
        const user = this.localStorageService.getItem(
            this.tenantService.tenantConfig?.config.local_storage_user_key
        );
        return JSON.parse(user);
    }

    getUser(): User {
        let us = this.loggedinUser;
        if (!us) {
            us = this.getUserFromLS();
            this.loggedinUser = us;
        }
        return us;
    }

    clearUser() {
        this.loggedinUser = null;
        this.localStorageService.removeItem(
            this.tenantService.tenantConfig?.config.local_storage_user_key
        );
    }

    getToken(): string {
        const user = this.getUserFromLS();
        return user ? user.accessToken : null;
    }

    fetchUser() {
        return this.get('user')
    }

    /**
    * @returns user role from user authorities
    */
    getUserRole() {
        const user = this.getUser();
        const roles = this.getSystemRoles();
        const role = roles.find(f => user.authorities?.map(m => m.key)?.includes(f));
        return role;
    }

    /**
    * @returns all roles possible for this project
    */
    getSystemRoles() {
        return Object.values(UserRolesEnum);
    }

    /**
    * @returns all roles possible for this project
    */
    getUserAuthorities() {
        const user = this.getUser();
        return user ? user.authorities : null;
    }


    /**
     * @returns navigate the user to the main screen
     */
    navigateToMainScreen() {
        
        const role = this.getUserRole();
        if (this.redirectTo && this.redirectTo !== '/') {
            const redirectTo = this.redirectTo;
            this.redirectTo = null;
            this.router.navigate([redirectTo])
            return;
        }      
        switch (role) {
            case UserRolesEnum.WFP_ADMIN:
            case UserRolesEnum.SUPERVISOR:
            case UserRolesEnum.SUPPORT_ADMIN_USER:
            case UserRolesEnum.NPTP_USER:
            case UserRolesEnum.CI_INSPECTOR_USER:
            case UserRolesEnum.MEHE_USER:
            case UserRolesEnum.SUPPORT_OPERATOR_USER:
            case UserRolesEnum.MOPH_USER:
                this.router.navigate(['/home']);
                break;
            default:
                this.onLogout();
                this.snacks.openWarningSnackBar("snackbar.user_not_allowed");
                break;
        }
    }

    isModuleAccessible(module: string): boolean {
        let authorities = this.getUserAuthorities();
        const ind = authorities.findIndex(i => i.key === module);
        return ind >= 0;
    }

    isPageAccessible(module: string): boolean {
        let res = true;
        if (module) {
            let authorities = this.getUserAuthorities();
            let page;
            if (module.includes('.')) {
                page = module.split('.');
            }
            else {
                page = [module];
            }
            page.forEach(p => {
                const ind = authorities?.findIndex(i => i.key === p);
                if (ind >= 0) {
                    authorities = authorities[ind].authorities;
                    res = res && true;
                }
                else {
                    res = false;
                }
            });
        }
        else {
            res = false;
        }
        return res;
    }

    isActionAllowed(module: UserModule, action: string): boolean {
        let authorities = this.getUserAuthorities();
        const mod = authorities.find(f => f.key === module.module);
        const ind = mod ? mod.authorities?.findIndex(i => i.key === `${module.key}_${action}`) : -1;
        return ind >= 0;
    }

}
