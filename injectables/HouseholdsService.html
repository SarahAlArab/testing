  getFilterVal(): any {
    return this._filterSource.getValue();
  }

  private _setFilterVal(filter: any): void {
    this._filterSource.next(filter);
  }

  addFilter(componentFilter: any) {
    let existingIndex = this.getFilterVal()?.findIndex(
      (f) => f.component === componentFilter.component
    );

    //replace existing filter
    if (existingIndex >= 0) {
      this.getFilterVal()[existingIndex] = componentFilter;
      this._setFilterVal(this.getFilterVal());
    }

    //add new filter
    else {
      let allFilters = [...this.getFilterVal(), componentFilter];
      this._setFilterVal(allFilters);
    }
  }

  clearFilter(componentName: any) {
    let filters = [...this.getFilterVal()].filter(
      (f) => f.component !== componentName
    );
    this._filterSource.next(filters);
  }

  getHistory(id: number): Observable<{ content: HistoryTableData[] }> {
    return this.get(`aid_card_audit/application/${id}/issue/history`).pipe(
      map((res: HttpResponse<HistoryTableData[]>) => {
        res.body.forEach((history) => {
          let { tag } = history;
          if (tag === ISSUE_TAG.NAME_CHANGE) {
            history.tag = 'إصلاح الخطأ في الاسم';
          } else if (tag === ISSUE_TAG.MOBILE_CHANGE) {
            history.tag = 'إصلاح الخطأ في رقم الهاتف';
          } else if (tag === ISSUE_TAG.BENEFICIARY_CHANGE) {
            history.tag = 'إصلاح اسم المستفيد (مستفيد بديل)';
          } else if (tag === ISSUE_TAG.CANCEL_PAYMENTS) {
            history.tag = '(إلغاء جميع المدفوعات السابقة) ';
          }
        });
        return { content: res.body };
      })
    );
  }

  getApplicationAuditHistory(
    pagination: PageChangeEvent,
    start: boolean,
    filter?: string
  ): Observable<any> {
    let allParams;
    let params: SpringPaginationParametersModel;
    //default filter is used since the api expect the below parameters as mandatory.
    let defaultFilter =
      'nameChange:false,headNameChanges:false,adminChanges:false,townChange:false,mobileNumberChange:false,familyNbrChange:false,idNumberChange:false';
    //pagination
    if (start) {
      params = Object.assign({}, SpringDefaultPaginationParametersModel);
    } else if (pagination) {
      params = { page: pagination.page, size: pagination.size };
    }
    //filter
    if (filter) {
      allParams = {
        sort: 'creationDateTime,desc',
        ...params,
        ...{ search: filter.replaceAll('=', ':') + ',' + defaultFilter },
      };
    } else {
      allParams = {
        sort: 'creationDateTime,desc',
        ...params,
        ...{ search: defaultFilter },
      };
    }

    return this.get(
      'audit_history/application/app-history',
      this.setAndGetParams(allParams),
      undefined,
      this.settingsService.settings.essnUrl
    );
  }
