import { HttpClient } from '@angular/common/http';
import { Injectable, Injector } from '@angular/core';
import { Observable, of } from 'rxjs';
import { HttpBaseClass } from 'src/app/shared/bases/http-base.class';
import { PageChangeEvent } from 'src/app/shared/models/page-change-event.interface';
import { SpringDefaultPaginationParametersModel, SpringPaginationParametersModel } from 'src/app/shared/models/pagination-parameters.model';

@Injectable()
export class InstitutionsService extends HttpBaseClass {

  constructor(
    public http: HttpClient,
    public injector: Injector,
  ) {
    super(http, injector);
  }

  getList(pagination: PageChangeEvent, start: boolean, filter?: string): Observable<any> {
    let allParams;
    let params: SpringPaginationParametersModel;
    if (start) {
      params = Object.assign({}, SpringDefaultPaginationParametersModel);
    }
    else if (pagination) {
      params = { page: pagination.page, size: pagination.size };
    }
   
    if (filter) {
      allParams = { sort: 'creationDateTime,desc', ...params, ...{ search: filter } };
    } else {
      allParams = { sort: 'creationDateTime,desc', ...params };
    }

    return this.get('survey_mgt/entity', this.setAndGetParams(allParams));
  }

  getRecordById(ref: number): Observable<any> {
    return this.get(`survey_mgt/entity/${ref}`);
  }

  editStatus(ref: string | number, status: any): Observable<any> {
    return this.put(`survey_mgt/entity/${ref}/${status}`);
  }

  editRecord(ref: string | number, obj: any): Observable<any> {
    return this.put(`survey_mgt/entity/${ref}`, obj);
  }

  addRecord(obj: any): Observable<any> {
    return this.post('survey_mgt/entity', obj);
  }
}
