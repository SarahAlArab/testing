export interface AssignSpotcheckPayload {

    fromDate: string;

export const prepareAssignSpotcheckPayload = (result: DialogResult, version: number): AssignSpotcheckPayload => {
    return {
        kadaa: { id: result.kadaa },
        applicationPercentage: +result.applicationPercentage,
        fromDate: formatDate(result.fromDate, "dd/MM/yyy", "en"),
        toDate: formatDate(result.toDate, "dd/MM/yyy", "en"),
        version,
        supervisorIds: result.supervisorIds
    }
}
