import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';
import { CanActivatePageGuard } from 'src/app/shared/guards/can-activate-page.guard';
import { UserModulesEnum, UserRolesActionsEnum } from 'src/app/shared/models/user-role.enum';
import { ApplicationsListComponent } from './applications-list/applications-list.component';
import { AssignedHouseholdsComponent } from './assigned-households/assigned-households.component';
import { AuditComponent } from './assigned-households/audit/audit.component';
import { EssnComponent } from './assigned-households/essn/essn.component';
import { SurveysListComponent } from './assigned-households/surveys-list/surveys-list.component';
import { CasesInspectionListComponent } from './cases-inspection/cases-inspection-list/cases-inspection-list.component';
import { CasesInspectionComponent } from './cases-inspection/cases-inspection.component';
import { CorrectionsComponent } from './corrections/corrections.component';
import { FuzzyMatchingDetailsComponent } from './fuzzy-matching/fuzzy-matching-details/fuzzy-matching-details.component';
import { FuzzyMatchingListComponent } from './fuzzy-matching/fuzzy-matching-list/fuzzy-matching-list.component';
import { FuzzyMatchingComponent } from './fuzzy-matching/fuzzy-matching.component';
import { HouseholdsComponent } from './households.component';
import { LandingHouseholdsComponent } from './landing-households/landing-households.component';
import { PaymentGenerationComponent } from './payment-generation/payment-generation.component';
import { StudentDataComponent } from './student-data/student-data.component';
import { HistoryProcessComponent } from './history-process/history-process.component';
import { SmsGenerationListComponent } from './sms/sms-generation-list/sms-generation-list.component';
import { SmsGenerationComponent } from './sms/sms-generation.component';
import { CampaignsListComponent } from './sms/campaigns/campaigns-list/campaigns-list.component';
import { CampaignsDetailsComponent } from './sms/campaigns/campaigns-details/campaigns-details.component';
import { CampaignsTemplatesListComponent } from './sms/campaigns-templates-list/campaigns-templates-list.component';
import { UnassignedHouseholdsComponent } from './unassigned-households/unassigned-households.component';
import { UnscheduledSpotchecksComponent } from './unscheduled-spotchecks/unscheduled-spotchecks.component';
import { PreviousChangesComponent } from './change-logs/previous-changes/previous-changes.component';
import { ChangeLogsListingComponent } from './change-logs/change-logs-listing/change-logs-listing.component';
import { GeneratedReportComponent } from './generated-reports/generated-report.component';
import { ChangeLogsComponent } from './change-logs/change-logs.component';
import { HouseholdReportsComponent } from './household-reports/household-reports.component';
import { InclusionExclusionComponent } from './inclusion-exclusion/inclusion-exclusion.component';
import { ReinitializeApplicationsComponent } from './reinitialize-apps/reinitialize-applications.component';
import { PaymentPlansComponent } from './payment-generation/payment-plans/payment-plans.component';
import { ExternalProgramComponent } from '../external-program/external-program.component';
import { MigrationComponent } from './migration-dialog-component/migration-dialog.component';

const routes: Routes = [
  {
    path: '', component: HouseholdsComponent,
    children: [
      { path: '', component: LandingHouseholdsComponent },
      {
        path: 'assigned', component: AssignedHouseholdsComponent,
        canActivate: [CanActivatePageGuard],
        data: {
          pageAccess: `${UserModulesEnum.HOUSEHOLDS.module}.${UserModulesEnum.HOUSEHOLDS.key}_ASSIGNED_${UserRolesActionsEnum.LIST}`,
          partialPageAccess: `ACCESS_APPLICATIONS`
        },
        children: [
          { path: '', component: SurveysListComponent },
          { path: ':id/audit', component: AuditComponent, data: { isCIData: false } },
          { path: ':id/essn', component: EssnComponent, data: { isCIData: false } },
        ]
      },
      {
        path: 'unassigned', component: UnassignedHouseholdsComponent,
        canActivate: [CanActivatePageGuard],
        data: { pageAccess: `${UserModulesEnum.HOUSEHOLDS.module}.${UserModulesEnum.HOUSEHOLDS.key}_${UserRolesActionsEnum.LIST}` }
      },
      {
        path: 'unscheduled-spotchecks', component: UnscheduledSpotchecksComponent,
        canActivate: [CanActivatePageGuard],
        data: { pageAccess: `${UserModulesEnum.HOUSEHOLDS.module}.${UserModulesEnum.HOUSEHOLDS.key}_${UserRolesActionsEnum.LIST}` }
      },
      {
        path: 'fuzzy-matching', component: FuzzyMatchingComponent,
        children: [
          { path: '', component: FuzzyMatchingListComponent },
          { path: ':id/moderate', component: FuzzyMatchingDetailsComponent },
        ],
        canActivate: [CanActivatePageGuard],
        data: { pageAccess: `${UserModulesEnum.DUPLICATE.module}.${UserModulesEnum.DUPLICATE.key}_${UserRolesActionsEnum.LIST}` }
      },
      {
        path: 'cases-inspection', component: CasesInspectionComponent,
        children: [
          { path: '', component: CasesInspectionListComponent },
          { path: ':id/audit', component: AuditComponent, data: { isCIData: true } },
          { path: ':id/visit', component: EssnComponent, data: { isCIData: true } }
        ]
      },
      {
        path: 'change-logs', component: ChangeLogsComponent,
        children: [
          { path: '', component: ChangeLogsListingComponent },
          { path: ':id/logs', component: PreviousChangesComponent },
          { path: ':id/audit', component: AuditComponent, data: { isCIData: false } },
          { path: ':id/essn', component: EssnComponent, data: { isCIData: false } },
        ],
      },
      {
        path: 'payments', component: PaymentGenerationComponent,
      },
      {
        path: 'studentData', component: StudentDataComponent,
      },
      {
        path: 'houseHoldReport', component: HouseholdReportsComponent,
      },
      {
        path: 'inclusion-exclusion', component: InclusionExclusionComponent,
      },
      {
        path: 'reinitialize-applications', component: ReinitializeApplicationsComponent,
      },
      { path: 'history-process', component: HistoryProcessComponent },
      { path: 'generated-reports', component: GeneratedReportComponent },
      {
        path: 'sms', component: SmsGenerationComponent,
        children: [
          { path: '', component: SmsGenerationListComponent },
          { path: 'campaigns', component: CampaignsListComponent },
          { path: 'campaigns/new', component: CampaignsDetailsComponent },
          { path: 'campaigns/:id', component: CampaignsDetailsComponent },
          { path: 'campaigns-template', component: CampaignsTemplatesListComponent }
        ],
      },
      {
        path: 'applications', component: ApplicationsListComponent
      },
      {
        path: 'corrections', component: CorrectionsComponent
      },
      {
        path: 'payment-plans', component: PaymentPlansComponent
      },
      {
        path: 'external-program', component: ExternalProgramComponent
      },
      {
        path: 'bulkAuthUnauthorization', component: SurveysListComponent,
        data: {
          showBulkAuthUnauthorization: true
        }
      },
      {path: 'migration', component: MigrationComponent}
    ]
  }];

@NgModule({
  imports: [RouterModule.forChild(routes)],
  exports: [RouterModule]
})
export class HouseholdsRoutingModule { }
